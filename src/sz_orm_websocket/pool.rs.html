1//! # 连接池管理
2//!
3//! 实现 WebSocket 连接池：限制最大连接数，LRU 淘汰最久未活跃的连接。
4//! 适用于高并发场景下的连接数控制。
5//!
6//! ## 主要类型
7//!
8//! - [`PoolConfig`] — 连接池配置
9//! - [`PooledConnection`] — 池中连接条目
10//! - [`ConnectionPool`] — LRU 连接池
11
12use std::collections::{HashMap, VecDeque};
13use std::sync::Arc;
14use tokio::sync::RwLock;
15
16/// 连接池配置
17#[derive(Debug, Clone, Copy)]
18pub struct PoolConfig {
19    /// 最大连接数
20    pub max_connections: usize,
21}
22
23impl Default for PoolConfig {
24    fn default() -> Self {
25        Self {
26            max_connections: 10_000,
27        }
28    }
29}
30
31impl PoolConfig {
32    pub fn new(max_connections: usize) -> Self {
33        Self { max_connections }
34    }
35
36    pub fn validate(&self) -> Result<(), String> {
37        if self.max_connections == 0 {
38            return Err("max_connections must be > 0".to_string());
39        }
40        Ok(())
41    }
42}

44/// 池中连接条目
45#[derive(Debug, Clone)]
46pub struct PooledConnection {
47    /// 连接 ID
48    pub connection_id: String,
49    /// 关联的用户 ID
50    pub user_id: Option<i64>,
51    /// 最后活跃时间戳（毫秒）
52    pub last_active_at: i64,
53    /// 创建时间戳（毫秒）
54    pub created_at: i64,
55    /// 已发送消息数
56    pub messages_sent: u64,
57    /// 已接收消息数
58    pub messages_received: u64,
59}
60
61impl PooledConnection {
62    pub fn new(connection_id: impl Into<String>, now_ms: i64) -> Self {
63        Self {
64            connection_id: connection_id.into(),
65            user_id: None,
66            last_active_at: now_ms,
67            created_at: now_ms,
68            messages_sent: 0,
69            messages_received: 0,
70        }
71    }
72
73    pub fn with_user(mut self, user_id: i64) -> Self {
74        self.user_id = Some(user_id);
75        self
76    }
77
78    /// 更新最后活跃时间
79    pub fn touch(&mut self, now_ms: i64) {
80        self.last_active_at = now_ms;
81    }
82
83    /// 记录发送消息
84    pub fn record_sent(&mut self) {
85        self.messages_sent += 1;
86    }
87
88    /// 记录接收消息
89    pub fn record_received(&mut self) {
90        self.messages_received += 1;
91    }
92
93    /// 空闲时长（毫秒）
94    pub fn idle_ms(&self, now_ms: i64) -> i64 {
95        now_ms - self.last_active_at
96    }
97
98    /// 存活时长（毫秒）
99    pub fn uptime_ms(&self, now_ms: i64) -> i64 {
100        now_ms - self.created_at
101    }
102}

104/// LRU 连接池
105#[derive(Debug)]
106pub struct ConnectionPool {
107    config: PoolConfig,
108    /// 连接表：connection_id -> PooledConnection
109    connections: Arc<RwLock<HashMap<String, PooledConnection>>>,
110    /// LRU 顺序队列：最近活跃的在前，最久未活跃的在后
111    lru_order: Arc<RwLock<VecDeque<String>>>,
112}
113
114/// 加入连接池的结果
115#[derive(Debug, PartialEq, Eq)]
116pub enum AdmitResult {
117    /// 连接被接纳
118    Admitted,
119    /// 连接已存在（重复添加）
120    AlreadyExists,
121    /// 池已满，淘汰了最久未活跃的连接以腾出空间
122    EvictedAndAdmitted { evicted_id: String },
123}
124
125impl ConnectionPool {
126    pub fn new(config: PoolConfig) -> Self {
127        Self {
128            config,
129            connections: Arc::new(RwLock::new(HashMap::new())),
130            lru_order: Arc::new(RwLock::new(VecDeque::new())),
131        }
132    }
133
134    /// 获取配置
135    pub fn config(&self) -> &PoolConfig {
136        &self.config
137    }
138
139    /// 尝试将连接加入池中。
140    /// - 若已存在，返回 AlreadyExists
141    /// - 若池已满，淘汰 LRU 末尾连接后接纳
142    /// - 否则接纳
143    pub async fn admit(
144        &self,
145        connection_id: impl Into<String>,
146        now_ms: i64,
147    ) -> AdmitResult {
148        let id = connection_id.into();
149        let mut connections = self.connections.write().await;
150        if connections.contains_key(&id) {
151            return AdmitResult::AlreadyExists;
152        }
153
154        let mut lru = self.lru_order.write().await;
155        // 池已满：淘汰 LRU 末尾
156        let evicted = if connections.len() >= self.config.max_connections {
157            // 从末尾找到一个仍存在的连接（可能已被 remove 清除）
158            let mut evicted_id = None;
159            while let Some(candidate) = lru.pop_back() {
160                if connections.contains_key(&candidate) {
161                    connections.remove(&candidate);
162                    evicted_id = Some(candidate);
163                    break;
164                }
165            }
166            evicted_id
167        } else {
168            None
169        };
170
171        // 插入新连接
172        connections.insert(id.clone(), PooledConnection::new(&id, now_ms));
173        lru.push_front(id.clone());
174
175        match evicted {
176            Some(evicted_id) => AdmitResult::EvictedAndAdmitted { evicted_id },
177            None => AdmitResult::Admitted,
178        }
179    }
180
181    /// 移除连接
182    pub async fn remove(&self, connection_id: &str) -> Option<PooledConnection> {
183        let mut connections = self.connections.write().await;
184        let removed = connections.remove(connection_id);
185        if removed.is_some() {
186            let mut lru = self.lru_order.write().await;
187            lru.retain(|id| id != connection_id);
188        }
189        removed
190    }
191
192    /// 更新连接活跃时间（移到 LRU 头部）
193    pub async fn touch(&self, connection_id: &str, now_ms: i64) -> bool {
194        let mut connections = self.connections.write().await;
195        if let Some(conn) = connections.get_mut(connection_id) {
196            conn.touch(now_ms);
197            drop(connections);
198            let mut lru = self.lru_order.write().await;
199            lru.retain(|id| id != connection_id);
200            lru.push_front(connection_id.to_string());
201            return true;
202        }
203        false
204    }
205
206    /// 记录发送消息
207    pub async fn record_sent(&self, connection_id: &str) -> bool {
208        let mut connections = self.connections.write().await;
209        if let Some(conn) = connections.get_mut(connection_id) {
210            conn.record_sent();
211            return true;
212        }
213        false
214    }
215
216    /// 记录接收消息
217    pub async fn record_received(&self, connection_id: &str) -> bool {
218        let mut connections = self.connections.write().await;
219        if let Some(conn) = connections.get_mut(connection_id) {
220            conn.record_received();
221            return true;
222        }
223        false
224    }
225
226    /// 获取连接
227    pub async fn get(&self, connection_id: &str) -> Option<PooledConnection> {
228        let connections = self.connections.read().await;
229        connections.get(connection_id).cloned()
230    }
231
232    /// 当前连接数
233    pub async fn count(&self) -> usize {
234        let connections = self.connections.read().await;
235        connections.len()
236    }
237
238    /// 是否已满
239    pub async fn is_full(&self) -> bool {
240        self.count().await >= self.config.max_connections
241    }
242
243    /// 按用户 ID 查询连接
244    pub async fn find_by_user(&self, user_id: i64) -> Vec<PooledConnection> {
245        let connections = self.connections.read().await;
246        let mut result: Vec<PooledConnection> = connections
247            .values()
248            .filter(|c| c.user_id == Some(user_id))
249            .cloned()
250            .collect();
251        result.sort_by(|a, b| a.connection_id.cmp(&b.connection_id));
252        result
253    }
254
255    /// 清理空闲超过指定时长的连接，返回被清理的数量
256    pub async fn evict_idle(&self, idle_threshold_ms: i64, now_ms: i64) -> usize {
257        let mut connections = self.connections.write().await;
258        let mut lru = self.lru_order.write().await;
259        let before = connections.len();
260        let to_remove: Vec<String> = connections
261            .iter()
262            .filter(|(_, c)| c.idle_ms(now_ms) >= idle_threshold_ms)
263            .map(|(id, _)| id.clone())
264            .collect();
265        for id in &to_remove {
266            connections.remove(id);
267        }
268        lru.retain(|id| !to_remove.contains(id));
269        before - connections.len()
270    }
271
272    /// 清空连接池
273    pub async fn clear(&self) {
274        let mut connections = self.connections.write().await;
275        let mut lru = self.lru_order.write().await;
276        connections.clear();
277        lru.clear();
278    }
279
280    /// 获取 LRU 顺序的连接 ID 列表（最近活跃在前）
281    pub async fn lru_order_list(&self) -> Vec<String> {
282        let lru = self.lru_order.read().await;
283        lru.iter().cloned().collect()
284    }
285}
