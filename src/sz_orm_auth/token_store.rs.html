1//! 刷新令牌存储（Refresh Token Store）
2//!
3//! 实现安全的刷新令牌管理，支持：
4//! - **令牌轮换（Rotation）**：每次使用刷新令牌时签发新令牌，旧令牌立即失效
5//! - **令牌撤销（Revocation）**：主动撤销单个令牌或整个令牌家族
6//! - **重放检测（Replay Detection）**：检测到已使用的刷新令牌被再次提交时，
7//!   撤销该令牌所属的整个家族（Token Family），防止令牌窃取
8//! - **家族追踪（Family Tracking）**：同一认证会话产生的所有刷新令牌属于同一家族
9//!
10//! ## 工作流程
11//!
12//! 1. 用户登录 -> `issue_family(access, refresh)` 创建新家族
13//! 2. 刷新令牌 -> `refresh(old_refresh, new_access, new_refresh)` 轮换令牌
14//! 3. 重复使用旧令牌 -> `refresh()` 返回 `TokenFamilyError::ReplayDetected`，
15//!    自动撤销整个家族
16//! 4. 登出 -> `revoke_token()` 或 `revoke_family()` 撤销令牌
17

24/// 令牌家族错误
25///
26/// 表示刷新令牌操作中的安全相关错误。
27#[derive(Debug)]
28pub enum TokenFamilyError {
29    /// 刷新令牌不存在或已被撤销
30    NotFound(String),
31    /// 刷新令牌已被使用（重放攻击检测）
32    ///
33    /// 当已使用的刷新令牌被再次提交时返回此错误。
34    /// 调用方应立即撤销该令牌所属的整个家族。
35    ReplayDetected(String),
36    /// 刷新令牌已过期
37    Expired(String),
38    /// 家族已被撤销
39    FamilyRevoked(String),
40}
41
42impl std::fmt::Display for TokenFamilyError {
43    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
44        match self {
45            TokenFamilyError::NotFound(msg) => write!(f, "Token not found: {}", msg),
46            TokenFamilyError::ReplayDetected(msg) => {
47                write!(f, "Replay detected (token already used): {}", msg)
48            }
49            TokenFamilyError::Expired(msg) => write!(f, "Token expired: {}", msg),
50            TokenFamilyError::FamilyRevoked(msg) => {
51                write!(f, "Token family revoked: {}", msg)
52            }
53        }
54    }
55}
56
57impl std::error::Error for TokenFamilyError {}
58
59impl From<TokenFamilyError> for AuthError {
60    fn from(e: TokenFamilyError) -> Self {
61        match e {
62            TokenFamilyError::NotFound(msg) => AuthError::TokenInvalid(msg),
63            TokenFamilyError::ReplayDetected(msg) => AuthError::TokenInvalid(msg),
64            TokenFamilyError::Expired(msg) => AuthError::TokenExpired(msg),
65            TokenFamilyError::FamilyRevoked(msg) => AuthError::TokenInvalid(msg),
66        }
67    }
68}

70/// 存储的令牌元数据
71#[derive(Debug, Clone)]
72pub struct StoredToken {
73    /// 令牌值（refresh token 字符串）
74    pub token: String,
75    /// 所属家族 ID
76    pub family_id: String,
77    /// 关联的用户 ID
78    pub user_id: i64,
79    /// 创建时间（Unix 秒）
80    pub created_at: i64,
81    /// 过期时间（Unix 秒）
82    pub expires_at: i64,
83    /// 是否已被使用（轮换后标记为 true）
84    pub used: bool,
85    /// 是否已被主动撤销
86    pub revoked: bool,
87}
88
89impl StoredToken {
90    /// 创建新的存储令牌
91    pub fn new(
92        token: impl Into<String>,
93        family_id: impl Into<String>,
94        user_id: i64,
95        expires_at: i64,
96    ) -> Self {
97        Self {
98            token: token.into(),
99            family_id: family_id.into(),
100            user_id,
101            created_at: current_secs(),
102            expires_at,
103            used: false,
104            revoked: false,
105        }
106    }
107
108    /// 是否已过期
109    pub fn is_expired(&self) -> bool {
110        current_secs() > self.expires_at
111    }
112
113    /// 是否有效（未使用、未撤销、未过期）
114    pub fn is_valid(&self) -> bool {
115        !self.used && !self.revoked && !self.is_expired()
116    }
117}

120#[derive(Debug)]
121struct FamilyInfo {

123    revoked: bool,
124    /// 家族中的所有令牌值
125    tokens: Vec<String>,
126}

128/// 刷新令牌存储
129///
130/// 管理刷新令牌的生命周期，支持轮换、撤销和重放检测。
131/// 使用 `Mutex<HashMap>` 进行线程安全存储。
132pub struct TokenStore {
133    /// 令牌值 -> 存储的令牌元数据
134    tokens: Mutex<HashMap<String, StoredToken>>,
135    /// 家族 ID -> 家族信息
136    families: Mutex<HashMap<String, FamilyInfo>>,
137    /// 刷新令牌默认有效期（秒）
138    default_refresh_lifetime: i64,
139}
140
141impl TokenStore {
142    /// 创建新的令牌存储，刷新令牌默认有效期 7 天
143    pub fn new() -> Self {
144        Self {
145            tokens: Mutex::new(HashMap::new()),
146            families: Mutex::new(HashMap::new()),
147            default_refresh_lifetime: 7 * 24 * 3600,
148        }
149    }
150
151    /// 配置刷新令牌默认有效期（秒）
152    pub fn with_refresh_lifetime(mut self, seconds: i64) -> Self {
153        self.default_refresh_lifetime = seconds;
154        self
155    }
156
157    /// 签发新的令牌家族（用户登录时调用）
158    ///
159    /// 创建一个新的令牌家族，并存储初始刷新令牌。
160    /// 返回创建的 `StoredToken` 供调用方返回给客户端。
161    pub fn issue_family(
162        &self,
163        refresh_token: impl Into<String>,
164        user_id: i64,
165    ) -> Result<StoredToken, AuthError> {
166        let token_value = refresh_token.into();
167        let family_id = generate_family_id();
168        let expires_at = current_secs() + self.default_refresh_lifetime;
169
170        let stored = StoredToken::new(token_value.clone(), family_id.clone(), user_id, expires_at);
171
172        self.tokens
173            .lock()
174            .unwrap()
175            .insert(token_value.clone(), stored.clone());
176
177        self.families.lock().unwrap().insert(
178            family_id.clone(),
179            FamilyInfo {
180                revoked: false,
181                tokens: vec![token_value],
182            },
183        );
184
185        Ok(stored)
186    }
187
188    /// 刷新令牌（轮换）
189    ///
190    /// 验证旧刷新令牌有效后，标记其为已使用，并签发新的刷新令牌（同一家族）。
191    ///
192    /// # 安全机制
193    ///
194    /// 1. 如果旧令牌已被使用 -> 返回 `ReplayDetected`，撤销整个家族
195    /// 2. 如果旧令牌已被撤销 -> 返回 `NotFound`
196    /// 3. 如果旧令牌已过期 -> 返回 `Expired`
197    /// 4. 如果家族已被撤销 -> 返回 `FamilyRevoked`
198    pub fn refresh(
199        &self,
200        old_refresh_token: &str,
201        new_refresh_token: impl Into<String>,
202    ) -> Result<StoredToken, TokenFamilyError> {
203        let new_token_value = new_refresh_token.into();
204        let now = current_secs();
205        let expires_at = now + self.default_refresh_lifetime;
206
207        // 第一阶段：读取并验证旧令牌状态（不加写锁，避免与 revoke 冲突）
208        let (family_id, user_id, is_used, is_revoked, is_expired, family_revoked) = {
209            let tokens = self.tokens.lock().unwrap();
210            let old_stored = match tokens.get(old_refresh_token) {
211                Some(t) => t,
212                None => {
213                    return Err(TokenFamilyError::NotFound(
214                        "Refresh token not found".to_string(),
215                    ))
216                }
217            };
218
219            let family_id = old_stored.family_id.clone();
220            let user_id = old_stored.user_id;
221            let is_used = old_stored.used;
222            let is_revoked = old_stored.revoked;
223            let is_expired = old_stored.is_expired();
224
225            let family_revoked = {
226                let families = self.families.lock().unwrap();
227                families
228                    .get(&family_id)
229                    .map(|f| f.revoked)
230                    .unwrap_or(false)
231            };
232
233            (
234                family_id,
235                user_id,
236                is_used,
237                is_revoked,
238                is_expired,
239                family_revoked,
240            )
241        };
242
243        // 检查家族是否已被撤销
244        if family_revoked {
245            return Err(TokenFamilyError::FamilyRevoked(format!(
246                "Family {} has been revoked",
247                family_id
248            )));
249        }
250
251        // 检查令牌是否已被撤销
252        if is_revoked {
253            return Err(TokenFamilyError::NotFound(
254                "Refresh token has been revoked".to_string(),
255            ));
256        }
257
258        // 检查令牌是否已过期
259        if is_expired {
260            return Err(TokenFamilyError::Expired(
261                "Refresh token has expired".to_string(),
262            ));
263        }
264
265        // 重放检测：令牌已被使用 -> 撤销整个家族
266        // 注意：此时未持有 tokens 锁，revoke_family_internal 可以安全获取锁
267        if is_used {
268            self.revoke_family_internal(&family_id);
269            return Err(TokenFamilyError::ReplayDetected(format!(
270                "Refresh token already used (family {} revoked)",
271                family_id
272            )));
273        }
274
275        // 第二阶段：标记旧令牌为已使用，并创建新令牌
276        let new_stored =
277            StoredToken::new(new_token_value.clone(), family_id.clone(), user_id, expires_at);
278
279        {
280            let mut tokens = self.tokens.lock().unwrap();
281            // 再次检查令牌状态（防止 TOCTOU：在两次加锁之间令牌可能被撤销或使用）
282            let old = match tokens.get_mut(old_refresh_token) {
283                Some(t) => t,
284                None => {
285                    return Err(TokenFamilyError::NotFound(
286                        "Refresh token not found".to_string(),
287                    ))
288                }
289            };
290
291            if old.used {
292                // 在释放锁的窗口内被使用 -> 重放
293                drop(tokens);
294                self.revoke_family_internal(&family_id);
295                return Err(TokenFamilyError::ReplayDetected(format!(
296                    "Refresh token already used (family {} revoked)",
297                    family_id
298                )));
299            }
300            if old.revoked {
301                return Err(TokenFamilyError::NotFound(
302                    "Refresh token has been revoked".to_string(),
303                ));
304            }
305
306            old.used = true;
307            tokens.insert(new_token_value.clone(), new_stored.clone());
308        }
309
310        // 将新令牌添加到家族
311        {
312            let mut families = self.families.lock().unwrap();
313            if let Some(family) = families.get_mut(&family_id) {
314                family.tokens.push(new_token_value);
315            }
316        }
317
318        Ok(new_stored)
319    }

321    /// 撤销单个令牌
322    ///
323    /// 标记令牌为已撤销，但不影响家族中的其他令牌。
324    /// 适用于用户登出单个设备的场景。
325    pub fn revoke_token(&self, token: &str) -> Result<(), TokenFamilyError> {
326        let mut tokens = self.tokens.lock().unwrap();
327        let stored = tokens
328            .get_mut(token)
329            .ok_or_else(|| TokenFamilyError::NotFound("Token not found".to_string()))?;
330        stored.revoked = true;
331        Ok(())
332    }
333
334    /// 撤销整个令牌家族
335    ///
336    /// 撤销家族中的所有令牌。适用于：
337    /// - 用户修改密码
338    /// - 检测到重放攻击
339    /// - 管理员强制下线
340    pub fn revoke_family(&self, family_id: &str) -> Result<usize, TokenFamilyError> {
341        // 先验证家族存在
342        {
343            let families = self.families.lock().unwrap();
344            if !families.contains_key(family_id) {
345                return Err(TokenFamilyError::NotFound("Family not found".to_string()));
346            }
347        }
348        // 实际撤销由内部方法处理
349        Ok(self.revoke_family_internal(family_id))
350    }
351
352    /// 撤销家族的内部实现（不加锁冲突）
353    ///
354    /// 返回撤销的令牌数量。
355    fn revoke_family_internal(&self, family_id: &str) -> usize {
356        let token_values: Vec<String> = {
357            let mut families = self.families.lock().unwrap();
358            if let Some(family) = families.get_mut(family_id) {
359                family.revoked = true;
360                family.tokens.clone()
361            } else {
362                return 0;
363            }
364        };
365
366        let mut tokens = self.tokens.lock().unwrap();
367        let mut count = 0;
368        for tv in &token_values {
369            if let Some(stored) = tokens.get_mut(tv) {
370                stored.revoked = true;
371                count += 1;
372            }
373        }
374        count
375    }

377    /// 撤销用户的所有令牌
378    ///
379    /// 撤销属于指定用户的所有令牌家族。
380    /// 适用于用户修改密码、账户被禁用等场景。
381    pub fn revoke_user(&self, user_id: i64) -> usize {
382        let family_ids: Vec<String> = {
383            let tokens = self.tokens.lock().unwrap();
384            tokens
385                .values()
386                .filter(|t| t.user_id == user_id)
387                .map(|t| t.family_id.clone())
388                .collect::<std::collections::HashSet<_>>()
389                .into_iter()
390                .collect()
391        };
392
393        let mut total = 0;
394        for fid in family_ids {
395            total += self.revoke_family_internal(&fid);
396        }
397        total
398    }

400    /// 验证令牌是否有效
401    pub fn is_valid(&self, token: &str) -> bool {
402        let tokens = self.tokens.lock().unwrap();
403        tokens.get(token).map(|t| t.is_valid()).unwrap_or(false)
404    }

406    /// 获取令牌信息
407    pub fn get_token(&self, token: &str) -> Option<StoredToken> {
408        self.tokens.lock().unwrap().get(token).cloned()
409    }

411    /// 获取家族中的所有令牌
412    pub fn family_tokens(&self, family_id: &str) -> Vec<StoredToken> {
413        let token_values: Vec<String> = {
414            let families = self.families.lock().unwrap();
415            families
416                .get(family_id)
417                .map(|f| f.tokens.clone())
418                .unwrap_or_default()
419        };
420
421        let tokens = self.tokens.lock().unwrap();
422        token_values
423            .iter()
424            .filter_map(|tv| tokens.get(tv).cloned())
425            .collect()
426    }

428    /// 检查家族是否已被撤销
429    pub fn is_family_revoked(&self, family_id: &str) -> bool {
430        self.families
431            .lock()
432            .unwrap()
433            .get(family_id)
434            .map(|f| f.revoked)
435            .unwrap_or(false)
436    }

438    /// 清理已过期和已撤销的令牌
439    ///
440    /// 返回清理的令牌数量。
441    pub fn cleanup(&self) -> usize {
442        let mut tokens = self.tokens.lock().unwrap();
443        let before = tokens.len();
444        tokens.retain(|_, t| !t.is_expired() && !t.revoked);
445        before - tokens.len()
446    }
447
448    /// 返回当前存储的令牌数量
449    pub fn token_count(&self) -> usize {
450        self.tokens.lock().unwrap().len()
451    }
452
453    /// 返回当前存储的家族数量
454    pub fn family_count(&self) -> usize {
455        self.families.lock().unwrap().len()
456    }

465/// 生成随机家族 ID（32 字节十六进制）
466fn generate_family_id() -> String {
467    use std::collections::hash_map::DefaultHasher;
468    use std::hash::{Hash, Hasher};
469    let mut hasher = DefaultHasher::new();
470    current_nanos().hash(&mut hasher);
471    let seed = hasher.finish();
472    format!("fam_{:016x}", seed)
473}
474
475fn current_secs() -> i64 {
476    SystemTime::now()
477        .duration_since(UNIX_EPOCH)
478        .unwrap_or_default()
479        .as_secs() as i64
480}
481
482fn current_nanos() -> u128 {
483    SystemTime::now()
484        .duration_since(UNIX_EPOCH)
485        .unwrap_or_default()
486        .as_nanos()
487}

769        assert!(store.get_token("r3").unwrap().used);

771        assert!(!store.get_token("r4").unwrap().used);
772        assert!(store.is_valid("r4"));
