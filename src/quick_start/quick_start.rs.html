1//! 快速入门 — QueryBuilder 基础用法
2//!
3//! 演示如何使用 QueryBuilder 构建 SELECT/INSERT/UPDATE/DELETE SQL。
4//! 无需数据库连接，仅生成 SQL 字符串。
5//!
6//! 运行：`cargo run -p sz-orm-examples --bin quick_start`
7
8use std::collections::HashMap;
9
10use sz_orm_core::dialect::get_dialect;
11use sz_orm_core::{DbType, Model, QueryBuilder, TimestampFields, Value};
12
13#[derive(Debug, Clone, Default)]
14#[allow(dead_code)]
15struct User {
16    id: i64,
17    name: String,
18    email: String,
19}
20
21impl Model for User {
22    type PrimaryKey = i64;
23    fn table_name() -> &'static str {
24        "users"
25    }
26    fn pk(&self) -> Self::PrimaryKey {
27        self.id
28    }
29    fn set_pk(&mut self, pk: Self::PrimaryKey) {
30        self.id = pk;
31    }
32    fn timestamp_fields() -> Option<TimestampFields> {
33        Some(TimestampFields::with_both("created_at", "updated_at"))
34    }
35}
36
37fn main() {
38    let make_dialect = || get_dialect(DbType::MySQL).expect("MySQL 方言可用");
39
40    // ===== SELECT =====
41    let select_sql = QueryBuilder::<User>::new(make_dialect())
42        .table("users")
43        .select(vec!["id", "name", "email"])
44        .where_cond("status = 'active'")
45        .order_by("created_at")
46        .order_desc("id")
47        .limit(10)
48        .build_select();
49    println!("SELECT:\n{}\n", select_sql);
50
51    // ===== WHERE 复合条件 =====
52    let complex_sql = QueryBuilder::<User>::new(make_dialect())
53        .table("users")
54        .select(vec!["id", "name"])
55        .where_cond("status = 'active'")
56        .or_where("role = 'admin'")
57        .where_in("id", vec![Value::I64(1), Value::I64(2), Value::I64(3)])
58        .where_between("age", Value::I64(18), Value::I64(65))
59        .where_null("deleted_at")
60        .page(3, 20)
61        .build_select();
62    println!("复杂 WHERE:\n{}\n", complex_sql);
63
64    // ===== JOIN =====
65    let join_sql = QueryBuilder::<User>::new(make_dialect())
66        .table("users")
67        .select(vec!["users.id", "posts.title"])
68        .join_inner("posts", "users.id", "posts.user_id")
69        .join_left("profiles", "users.id", "profiles.user_id")
70        .where_cond("users.status = 'active'")
71        .build_select();
72    println!("JOIN:\n{}\n", join_sql);
73
74    // ===== 聚合 =====
75    let count_sql = QueryBuilder::<User>::new(make_dialect())
76        .table("users")
77        .where_cond("status = 'active'")
78        .build_count();
79    println!("COUNT:\n{}\n", count_sql);
80
81    // ===== INSERT =====
82    let mut data = HashMap::new();
83    data.insert("name".to_string(), Value::String("Alice".to_string()));
84    data.insert(
85        "email".to_string(),
86        Value::String("alice@example.com".to_string()),
87    );
88    data.insert("age".to_string(), Value::I64(25));
89
90    let insert_sql = QueryBuilder::<User>::new(make_dialect())
91        .table("users")
92        .build_insert(&data);
93    println!("INSERT:\n{}\n", insert_sql);
94
95    // ===== UPDATE =====
96    let mut update_data = HashMap::new();
97    update_data.insert("name".to_string(), Value::String("Bob".to_string()));
98
99    let update_sql = QueryBuilder::<User>::new(make_dialect())
100        .table("users")
101        .where_cond("id = 1")
102        .build_update(&update_data);
103    println!("UPDATE:\n{}\n", update_sql);
104
105    // ===== DELETE =====
106    let delete_sql = QueryBuilder::<User>::new(make_dialect())
107        .table("users")
108        .where_cond("id = 1")
109        .build_delete();
110    println!("DELETE:\n{}\n", delete_sql);
111
112    // ===== 校验 =====
113    let validate_result = QueryBuilder::<User>::new(make_dialect())
114        .table("users")
115        .select(vec!["id", "name"])
116        .where_cond("id = 1")
117        .validate();
118    println!("校验结果: {:?}", validate_result);
119}
