1//! 多租户 — TenantModel + TenantScope
2//!
3//! 演示如何让 Model 自动获得 `tenant_id = ?` 的查询过滤。
4//! 通过实现 TenantModel trait，配合 (TenantScope, M) 全局作用域。
5//!
6//! 运行：`cargo run -p sz-orm-examples --bin multi_tenant`
7
8use sz_orm_core::hooks::{GlobalScope, HookContext, TenantModel, TenantScope};
9use sz_orm_core::{Model, Value};
10
11// ===== 多租户 Model 定义 =====
12
13#[derive(Debug, Clone, Default)]
14#[allow(dead_code)]
15struct Order {
16    id: i64,
17    tenant_id: i64,
18    amount: f64,
19    status: String,
20}
21
22impl Model for Order {
23    type PrimaryKey = i64;
24    fn table_name() -> &'static str {
25        "orders"
26    }
27    fn pk(&self) -> Self::PrimaryKey {
28        self.id
29    }
30    fn set_pk(&mut self, pk: Self::PrimaryKey) {
31        self.id = pk;
32    }
33}
34
35impl TenantModel for Order {
36    fn tenant_field() -> &'static str {
37        "tenant_id"
38    }
39    fn tenant_id(&self) -> i64 {
40        self.tenant_id
41    }
42    fn set_tenant_id(&mut self, tenant_id: i64) {
43        self.tenant_id = tenant_id;
44    }
45}
46
47fn main() {
48    println!("=== 多租户 Model ===");
49    println!("表名:        {}", Order::table_name());
50    println!("租户字段:    {}", Order::tenant_field());
51
52    let mut order = Order::default();
53    order.set_tenant_id(42);
54    println!("租户 ID:     {}", order.tenant_id());
55
56    println!("\n=== TenantScope 应用（有租户上下文）===");
57    let ctx_with_tenant = HookContext::new().with_tenant(42);
58    let scope_result = <(TenantScope, Order) as GlobalScope>::apply_scope(&ctx_with_tenant);
59    println!("ctx.tenant_id = {:?}", ctx_with_tenant.tenant_id);
60    println!("追加 WHERE:    {:?}", scope_result);
61    // 预期: Some(("tenant_id = ?", [I64(42)]))
62
63    println!("\n=== TenantScope 应用（无租户上下文，跨租户查询）===");
64    let ctx_no_tenant = HookContext::new();
65    let scope_result = <(TenantScope, Order) as GlobalScope>::apply_scope(&ctx_no_tenant);
66    println!("ctx.tenant_id = {:?}", ctx_no_tenant.tenant_id);
67    println!("追加 WHERE:    {:?}", scope_result);
68    // 预期: None（不追加条件，调用方自行保证安全）
69
70    println!("\n=== 多租户使用模式（伪代码）===");
71    println!(
72        r#"// 1. 中间件注入租户 ID 到 HookContext
73async fn tenant_middleware(req: Request, next: Next) -> Response {{
74    let tenant_id = extract_tenant_from_jwt(&req);
75    let ctx = HookContext::new().with_tenant(tenant_id);
76    // 将 ctx 注入到请求扩展中...
77    next.run(req).await
78}}
79
80// 2. 查询时自动追加 tenant_id = ?
81//    SELECT * FROM orders WHERE status = 'paid' AND tenant_id = ?
82//    (TenantScope::apply_scope 返回 ("tenant_id = ?", [I64(42)]))
83
84// 3. 超级管理员跨租户查询（不设置 tenant_id）
85//    SELECT * FROM orders WHERE status = 'paid'
86//    (TenantScope::apply_scope 返回 None)
87
88// 4. 写入时确保租户隔离
89let mut order = Order::default();
90order.set_tenant_id(ctx.tenant_id.unwrap());
91order.amount = 100.0;
92// INSERT INTO orders (tenant_id, amount, ...) VALUES (42, 100.0, ...)
93"#
94    );
95
96    println!("=== Value 类型验证 ===");
97    if let Some((sql, params)) =
98        <(TenantScope, Order) as GlobalScope>::apply_scope(&ctx_with_tenant)
99    {
100        println!("SQL 片段:  {}", sql);
101        println!("参数数量:  {}", params.len());
102        for (i, p) in params.iter().enumerate() {
103            println!("  参数[{}]: as_i64() = {:?}", i, p.as_i64());
104        }
105        // 验证 Value::I64 包装
106        let v = Value::I64(42);
107        assert_eq!(v.as_i64(), Some(42));
108        println!("Value::I64(42).as_i64() == Some(42) ✓");
109    }
110}
