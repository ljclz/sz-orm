1//! # SZ-ORM Audit — SQL 审计日志
2//!
3//! 提供 SQL 执行审计记录，对 password/token/credit_card 等敏感关键词进行
4//! 大小写不敏感脱敏，确保审计日志不泄露敏感信息。
5//!
6//! ## 主要类型
7//!
8//! - [`SqlAuditContext`] — 审计上下文（SQL/用户/时间戳）
9//! - [`SqlAuditor`] — 审计执行器
10
11use serde::{Deserialize, Serialize};
12use std::sync::Mutex;
13
14#[derive(Debug, Clone, Serialize, Deserialize)]
15pub struct SqlAuditContext {
16    pub sql: String,
17    pub user: String,
18    pub timestamp: i64,
19}
20
21/// Sensitive keywords that should be masked in audit logs. Matching is
22/// case-insensitive on the ASCII bytes of the SQL string.
23const SENSITIVE_KEYWORDS: &[&str] = &[
24    "password",
25    "pwd",
26    "passwd",
27    "secret",
28    "token",
29    "api_key",
30    "apikey",
31    "access_key",
32    "accesskey",
33    "session",
34    "credit_card",
35    "creditcard",
36    "cvv",
37    "ssn",
38];
39
40pub struct SqlAuditor {
41    logs: Mutex<Vec<SqlAuditContext>>,
42}
43
44impl SqlAuditor {
45    pub fn new() -> Self {
46        Self {
47            logs: Mutex::new(vec![]),
48        }
49    }
50
51    /// Log an audit entry. The SQL is masked for sensitive keywords before
52    /// being stored in the in-memory buffer.
53    pub fn log(&self, ctx: &SqlAuditContext) {
54        let masked_sql = mask_sensitive(&ctx.sql);
55        let entry = SqlAuditContext {
56            sql: masked_sql,
57            user: ctx.user.clone(),
58            timestamp: ctx.timestamp,
59        };
60        let mut logs = self.logs.lock().unwrap();
61        logs.push(entry);
62    }
63
64    /// Return a snapshot of all stored audit entries.
65    pub fn get_logs(&self) -> Vec<SqlAuditContext> {
66        let logs = self.logs.lock().unwrap();
67        logs.iter().cloned().collect()
68    }
69
70    /// Flush all stored audit entries to a JSON file at `path`.
71    /// Returns the number of entries written.
72    pub fn flush(&self, path: &str) -> Result<usize, String> {
73        let logs = self.logs.lock().unwrap();
74        let snapshot: Vec<&SqlAuditContext> = logs.iter().collect();
75        let json = serde_json::to_string_pretty(&snapshot).map_err(|e| e.to_string())?;
76        std::fs::write(path, json).map_err(|e| e.to_string())?;
77        Ok(logs.len())
78    }
79
80    /// Mask all sensitive keywords in `sql` with `******`. Matching is
81    /// case-insensitive.
82    pub fn mask_sensitive(&self, sql: &str) -> String {
83        mask_sensitive(sql)
84    }
85}
86
87impl Default for SqlAuditor {
88    fn default() -> Self {
89        Self::new()
90    }
91}

93/// Mask all sensitive keywords in `sql` with `******`. Matching is
94/// case-insensitive over the ASCII bytes of the string.
95fn mask_sensitive(sql: &str) -> String {
96    let lower = sql.to_ascii_lowercase();
97    let mut result = String::with_capacity(sql.len());
98    let mut i = 0;
99    let bytes = sql.as_bytes();
100    let lower_bytes = lower.as_bytes();
101    while i < bytes.len() {
102        let mut matched_len: Option<usize> = None;
103        for keyword in SENSITIVE_KEYWORDS {
104            let kw_bytes = keyword.as_bytes();
105            if i + kw_bytes.len() <= bytes.len() && &lower_bytes[i..i + kw_bytes.len()] == kw_bytes
106            {
107                // Only treat as a keyword match if it's not part of a longer identifier.
108                // Boundary check: previous and next char must be non-alphanumeric/underscore
109                let prev_ok = i == 0 || !is_ident_char(bytes[i - 1]);
110                let next_idx = i + kw_bytes.len();
111                let next_ok = next_idx >= bytes.len() || !is_ident_char(bytes[next_idx]);
112                if prev_ok && next_ok {
113                    matched_len = Some(kw_bytes.len());
114                    break;
115                }
116            }
117        }
118        if let Some(kw_len) = matched_len {
119            result.push_str("******");
120            i += kw_len;
121        } else {
122            // Push one char (handles UTF-8 properly since we step by char).
123            let ch = sql[i..].chars().next().unwrap();
124            result.push(ch);
125            i += ch.len_utf8();
126        }
127    }
128    result
129}
130
131/// Returns true if `b` is an ASCII identifier character (alphanumeric or '_').
132fn is_ident_char(b: u8) -> bool {
133    b.is_ascii_alphanumeric() || b == b'_'
134}

138// ============================================================================

140/// 审计规则：允许/拒绝列表，用于决定是否记录某条 SQL 审计日志。
141///
142/// 规则评估顺序：
143/// 1. 若 SQL 命中拒绝列表中的任一模式 → 不记录
144/// 2. 若允许列表为空 → 记录所有未命中拒绝列表的 SQL
145/// 3. 若允许列表非空 → 仅记录命中允许列表的 SQL
146#[derive(Debug, Clone, Default)]
147pub struct AuditRules {
148    /// 允许列表模式（大小写不敏感子串匹配），为空表示允许所有
149    allow_patterns: Vec<String>,
150    /// 拒绝列表模式（大小写不敏感子串匹配），命中则拒绝
151    deny_patterns: Vec<String>,
152}
153
154impl AuditRules {
155    pub fn new() -> Self {
156        Self::default()
157    }
158
159    /// 添加允许模式（大小写不敏感子串匹配）
160    pub fn allow(mut self, pattern: impl Into<String>) -> Self {
161        self.allow_patterns.push(pattern.into().to_ascii_lowercase());
162        self
163    }
164
165    /// 添加拒绝模式（大小写不敏感子串匹配）
166    pub fn deny(mut self, pattern: impl Into<String>) -> Self {
167        self.deny_patterns.push(pattern.into().to_ascii_lowercase());
168        self
169    }
170
171    /// 判断给定 SQL 是否应该被审计记录
172    pub fn should_audit(&self, sql: &str) -> bool {
173        let lower = sql.to_ascii_lowercase();
174        // 拒绝列表优先
175        for pat in &self.deny_patterns {
176            if lower.contains(pat) {
177                return false;
178            }
179        }
180        // 允许列表为空 → 允许所有
181        if self.allow_patterns.is_empty() {
182            return true;
183        }
184        // 允许列表非空 → 仅允许命中项
185        self.allow_patterns.iter().any(|pat| lower.contains(pat))
186    }
187
188    /// 返回允许模式数量
189    pub fn allow_count(&self) -> usize {
190        self.allow_patterns.len()
191    }
192
193    /// 返回拒绝模式数量
194    pub fn deny_count(&self) -> usize {
195        self.deny_patterns.len()
196    }
197}

200// 审计日志轮转策略（按大小/时间）
201// ============================================================================
202
203/// 审计日志轮转策略配置。
204///
205/// - `max_entries`：内存中最多保留的日志条数，超过后自动轮转（旧日志清空或落盘）
206/// - `max_age_ms`：日志最大存活时间（毫秒），超过后触发轮转
207#[derive(Debug, Clone)]
208pub struct RotationPolicy {
209    /// 最大条目数（0 表示不限制）
210    pub max_entries: usize,
211    /// 最大存活时间毫秒（0 表示不限制）
212    pub max_age_ms: i64,
213}
214
215impl RotationPolicy {
216    /// 创建不限制的轮转策略
217    pub fn none() -> Self {
218        Self {
219            max_entries: 0,
220            max_age_ms: 0,
221        }
222    }
223
224    /// 按条目数轮转
225    pub fn by_size(max_entries: usize) -> Self {
226        Self {
227            max_entries,
228            max_age_ms: 0,
229        }
230    }
231
232    /// 按时间轮转（毫秒）
233    pub fn by_age(max_age_ms: i64) -> Self {
234        Self {
235            max_entries: 0,
236            max_age_ms,
237        }
238    }
239
240    /// 同时按条目数和时间轮转
241    pub fn by_size_and_age(max_entries: usize, max_age_ms: i64) -> Self {
242        Self {
243            max_entries,
244            max_age_ms,
245        }
246    }
247
248    /// 判断是否需要轮转
249    fn needs_rotation(&self, entry_count: usize, oldest_ts: i64, now_ts: i64) -> bool {
250        if self.max_entries > 0 && entry_count >= self.max_entries {
251            return true;
252        }
253        if self.max_age_ms > 0 && oldest_ts > 0 && (now_ts - oldest_ts) > self.max_age_ms {
254            return true;
255        }
256        false
257    }
258}
259
260impl Default for RotationPolicy {
261    fn default() -> Self {
262        Self::none()
263    }
264}

269
270/// 带轮转策略和审计规则的增强审计器。
271///
272/// 在 `SqlAuditor` 基础上增加：
273/// - 日志轮转（按大小/时间自动清理旧日志）
274/// - 审计规则（允许/拒绝列表过滤）
275pub struct RotatingAuditor {
276    logs: Mutex<Vec<SqlAuditContext>>,
277    rules: AuditRules,
278    policy: RotationPolicy,
279    /// 已轮转（清理）的次数
280    rotations: Mutex<usize>,
281}
282
283impl RotatingAuditor {
284    pub fn new(policy: RotationPolicy, rules: AuditRules) -> Self {
285        Self {
286            logs: Mutex::new(vec![]),
287            rules,
288            policy,
289            rotations: Mutex::new(0),
290        }
291    }
292
293    /// 创建仅按大小轮转的审计器，无规则过滤
294    pub fn with_max_entries(max_entries: usize) -> Self {
295        Self::new(RotationPolicy::by_size(max_entries), AuditRules::new())
296    }
297
298    /// 创建仅按时间轮转的审计器，无规则过滤
299    pub fn with_max_age(max_age_ms: i64) -> Self {
300        Self::new(RotationPolicy::by_age(max_age_ms), AuditRules::new())
301    }
302
303    /// 记录审计日志，自动应用规则过滤和轮转策略
304    pub fn log(&self, ctx: &SqlAuditContext) -> bool {
305        // 规则过滤
306        if !self.rules.should_audit(&ctx.sql) {
307            return false;
308        }
309        let masked_sql = mask_sensitive(&ctx.sql);
310        let entry = SqlAuditContext {
311            sql: masked_sql,
312            user: ctx.user.clone(),
313            timestamp: ctx.timestamp,
314        };
315        let mut logs = self.logs.lock().unwrap();
316
317        // 在添加新条目前检查轮转（确保新条目不被立即清除）
318        let now = ctx.timestamp;
319        let oldest = logs.first().map(|e| e.timestamp).unwrap_or(now);
320        if self.policy.needs_rotation(logs.len(), oldest, now) {
321            logs.clear();
322            *self.rotations.lock().unwrap() += 1;
323        }
324
325        logs.push(entry);
326        true
327    }
328
329    /// 返回当前日志快照
330    pub fn get_logs(&self) -> Vec<SqlAuditContext> {
331        self.logs.lock().unwrap().clone()
332    }
333
334    /// 返回已轮转次数
335    pub fn rotation_count(&self) -> usize {
336        *self.rotations.lock().unwrap()
337    }
338
339    /// 手动触发轮转（清空当前日志）
340    pub fn rotate(&self) -> usize {
341        let mut logs = self.logs.lock().unwrap();
342        let count = logs.len();
343        logs.clear();
344        *self.rotations.lock().unwrap() += 1;
345        count
346    }
347
348    /// 返回当前日志条数
349    pub fn len(&self) -> usize {
350        self.logs.lock().unwrap().len()
351    }
352
353    /// 是否为空
354    pub fn is_empty(&self) -> bool {
355        self.logs.lock().unwrap().is_empty()
356    }
357}

363/// 异步审计写入器：通过后台线程异步写入审计日志，避免阻塞主线程。
364///
365/// 使用 `std::sync::mpsc` 通道将日志发送到后台线程，后台线程负责存储。
366/// 关闭时调用 `shutdown` 等待后台线程退出并返回所有已写入的日志。
367pub struct AsyncAuditWriter {
368    sender: std::sync::mpsc::Sender<AsyncCommand>,
369    handle: Mutex<Option<std::thread::JoinHandle<Vec<SqlAuditContext>>>>,
370}
371
372enum AsyncCommand {
373    Log(SqlAuditContext),
374    Shutdown,
375}
376
377impl AsyncAuditWriter {
378    /// 创建异步写入器，启动后台线程
379    pub fn new() -> Self {
380        let (sender, receiver) = std::sync::mpsc::channel::<AsyncCommand>();
381        let handle = std::thread::spawn(move || {
382            let mut logs: Vec<SqlAuditContext> = Vec::new();
383            for cmd in receiver {
384                match cmd {
385                    AsyncCommand::Log(ctx) => {
386                        let masked_sql = mask_sensitive(&ctx.sql);
387                        logs.push(SqlAuditContext {
388                            sql: masked_sql,
389                            user: ctx.user,
390                            timestamp: ctx.timestamp,
391                        });
392                    }
393                    AsyncCommand::Shutdown => break,
394                }
395            }
396            logs
397        });
398        Self {
399            sender,
400            handle: Mutex::new(Some(handle)),
401        }
402    }
403
404    /// 异步记录审计日志（非阻塞）
405    pub fn log(&self, ctx: &SqlAuditContext) -> Result<(), String> {
406        self.sender
407            .send(AsyncCommand::Log(ctx.clone()))
408            .map_err(|e| format!("AsyncAuditWriter channel closed: {}", e))
409    }
410
411    /// 关闭后台线程并返回所有已写入的日志
412    pub fn shutdown(&self) -> Result<Vec<SqlAuditContext>, String> {
413        let _ = self.sender.send(AsyncCommand::Shutdown);
414        let mut handle_guard = self.handle.lock().unwrap();
415        if let Some(handle) = handle_guard.take() {
416            handle
417                .join()
418                .map_err(|e| format!("Thread panicked: {:?}", e))
419        } else {
420            Err("Already shut down".to_string())
421        }
422    }
423}
424
425impl Default for AsyncAuditWriter {
426    fn default() -> Self {
427        Self::new()
428    }
429}

435/// 审计日志查询过滤器
436#[derive(Debug, Clone, Default)]
437pub struct AuditQuery {
438    /// 按用户名过滤（精确匹配，None 表示不过滤）
439    pub user: Option<String>,
440    /// 时间范围起始（毫秒时间戳，None 表示不限制下限）
441    pub from_ts: Option<i64>,
442    /// 时间范围结束（毫秒时间戳，None 表示不限制上限）
443    pub to_ts: Option<i64>,
444    /// SQL 关键词过滤（大小写不敏感子串匹配，None 表示不过滤）
445    pub sql_contains: Option<String>,
446    /// 限制返回条数（0 表示不限制）
447    pub limit: usize,
448}
449
450impl AuditQuery {
451    pub fn new() -> Self {
452        Self::default()
453    }
454
455    /// 按用户名过滤
456    pub fn by_user(mut self, user: impl Into<String>) -> Self {
457        self.user = Some(user.into());
458        self
459    }
460
461    /// 按时间范围过滤（毫秒时间戳）
462    pub fn by_time_range(mut self, from: i64, to: i64) -> Self {
463        self.from_ts = Some(from);
464        self.to_ts = Some(to);
465        self
466    }
467
468    /// 按 SQL 关键词过滤（大小写不敏感）
469    pub fn by_sql_contains(mut self, keyword: impl Into<String>) -> Self {
470        self.sql_contains = Some(keyword.into());
471        self
472    }
473
474    /// 限制返回条数
475    pub fn with_limit(mut self, limit: usize) -> Self {
476        self.limit = limit;
477        self
478    }
479
480    /// 对日志列表执行查询过滤
481    pub fn filter(&self, logs: &[SqlAuditContext]) -> Vec<SqlAuditContext> {
482        let keyword_lower = self.sql_contains.as_ref().map(|s| s.to_ascii_lowercase());
483        let mut result: Vec<SqlAuditContext> = logs
484            .iter()
485            .filter(|entry| {
486                if let Some(u) = &self.user {
487                    if entry.user != *u {
488                        return false;
489                    }
490                }
491                if let Some(from) = self.from_ts {
492                    if entry.timestamp < from {
493                        return false;
494                    }
495                }
496                if let Some(to) = self.to_ts {
497                    if entry.timestamp > to {
498                        return false;
499                    }
500                }
501                if let Some(kw) = &keyword_lower {
502                    if !entry.sql.to_ascii_lowercase().contains(kw) {
503                        return false;
504                    }
505                }
506                true
507            })
508            .cloned()
509            .collect();
510        if self.limit > 0 && result.len() > self.limit {
511            result.truncate(self.limit);
512        }
513        result
514    }
515}

517/// 从 `SqlAuditor` 的日志中按条件查询
518pub fn query_logs(auditor: &SqlAuditor, query: &AuditQuery) -> Vec<SqlAuditContext> {
519    let logs = auditor.get_logs();
520    query.filter(&logs)
521}
