1//! Search 错误类型
2
3use std::fmt;
4
5/// Search 操作错误
6#[derive(Debug)]
7pub enum SearchError {
8    /// 索引未找到
9    NotFound(String),
10    /// 文档未找到
11    DocNotFound { index: String, id: String },
12    /// 查询语法错误
13    InvalidQuery(String),
14    /// 索引已存在
15    IndexAlreadyExists(String),
16    /// 索引配置无效
17    InvalidConfig(String),
18    /// 序列化/反序列化错误
19    Serialization(String),
20    /// 查询执行错误
21    Query(String),
22    /// 连接错误
23    Connection(String),
24}
25
26impl fmt::Display for SearchError {
27    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
28        match self {
29            SearchError::NotFound(msg) => write!(f, "not found: {}", msg),
30            SearchError::DocNotFound { index, id } => {
31                write!(f, "document not found: {}/{}", index, id)
32            }
33            SearchError::InvalidQuery(msg) => write!(f, "invalid query: {}", msg),
34            SearchError::IndexAlreadyExists(msg) => {
35                write!(f, "index already exists: {}", msg)
36            }
37            SearchError::InvalidConfig(msg) => write!(f, "invalid config: {}", msg),
38            SearchError::Serialization(msg) => write!(f, "serialization error: {}", msg),
39            SearchError::Query(msg) => write!(f, "query error: {}", msg),
40            SearchError::Connection(msg) => write!(f, "connection error: {}", msg),
41        }
42    }
43}
44
45impl std::error::Error for SearchError {}
46
47impl From<serde_json::Error> for SearchError {
48    fn from(err: serde_json::Error) -> Self {
49        SearchError::Serialization(err.to_string())
50    }
51}
52
53impl From<std::io::Error> for SearchError {
54    fn from(err: std::io::Error) -> Self {
55        SearchError::Query(err.to_string())
56    }
57}
