1//! 钩子系统 — 软删除 + 全局作用域
2//!
3//! 演示 HookContext、HookRegistry、ScopeRegistry 的用法。
4//! Hookable + SoftDelete + GlobalScope 让 Model 自动获得：
5//! - 查询时自动追加 `deleted_at IS NULL`
6//! - 删除时改为 `UPDATE SET deleted_at = NOW()`
7//!
8//! 运行：`cargo run -p sz-orm-examples --bin hooks_soft_delete`
9
10use std::sync::Arc;
11
12use sz_orm_core::hooks::{
13    GlobalScope, HookContext, HookEvent, HookRegistry, Hookable, ScopeRegistry, SoftDelete,
14    SoftDeleteScope,
15};
16use sz_orm_core::{DbError, Model, TimestampFields, Value};
17
18// ===== 定义带软删除的 Model =====
19
20#[derive(Debug, Clone, Default)]
21#[allow(dead_code)]
22struct Product {
23    id: i64,
24    name: String,
25    price: f64,
26    deleted_at: Option<String>,
27}
28
29impl Model for Product {
30    type PrimaryKey = i64;
31    fn table_name() -> &'static str {
32        "products"
33    }
34    fn pk(&self) -> Self::PrimaryKey {
35        self.id
36    }
37    fn set_pk(&mut self, pk: Self::PrimaryKey) {
38        self.id = pk;
39    }
40    fn timestamp_fields() -> Option<TimestampFields> {
41        None
42    }
43    fn soft_delete_field() -> Option<&'static str> {
44        Some("deleted_at")
45    }
46}
47
48impl SoftDelete for Product {
49    fn soft_delete_field() -> &'static str {
50        "deleted_at"
51    }
52    fn is_deleted(&self) -> bool {
53        self.deleted_at.is_some()
54    }
55}
56
57impl Hookable for Product {
58    fn before_insert(_ctx: &mut HookContext) -> Result<(), DbError> {
59        println!("  [钩子] before_insert: 即将插入商品");
60        Ok(())
61    }
62    fn after_insert(_ctx: &HookContext, _id: &Self::PrimaryKey) -> Result<(), DbError> {
63        println!("  [钩子] after_insert: 商品已插入, id={}", _id);
64        Ok(())
65    }
66    fn before_delete(_ctx: &mut HookContext, _id: &Self::PrimaryKey) -> Result<(), DbError> {
67        println!("  [钩子] before_delete: 即将删除商品 id={}", _id);
68        Ok(())
69    }
70}
71
72fn main() {
73    println!("=== HookContext 构建器 ===");
74    let ctx = HookContext::new()
75        .with_tenant(42)
76        .with_operator(1)
77        .with_timestamp(1700000000);
78    println!("租户:    {:?}", ctx.tenant_id);
79    println!("操作人:  {:?}", ctx.operator_id);
80    println!("时间戳:  {}", ctx.timestamp);
81
82    let mut ctx = ctx;
83    ctx.set_meta("source", "api");
84    ctx.set_meta("ip", "127.0.0.1");
85    println!("元数据:  {:?}", ctx.metadata);
86
87    println!("\n=== 软删除全局作用域 ===");
88    // (SoftDeleteScope, Product) 实现了 GlobalScope
89    let scope_result = <(SoftDeleteScope, Product) as GlobalScope>::apply_scope(&ctx);
90    println!("追加 WHERE: {:?}", scope_result);
91    // 预期: Some(("deleted_at IS NULL", []))
92
93    println!("\n=== HookRegistry 运行时钩子 ===");
94    let registry = HookRegistry::new();
95
96    // 注册运行时钩子
97    let call_count = Arc::new(std::sync::atomic::AtomicU32::new(0));
98    let counter = Arc::clone(&call_count);
99    registry.register(
100        HookEvent::BeforeInsert,
101        Arc::new(move |_ctx| {
102            counter.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
103            Ok(())
104        }),
105    );
106
107    println!(
108        "初始调用次数: {}",
109        call_count.load(std::sync::atomic::Ordering::SeqCst)
110    );
111    registry.dispatch(HookEvent::BeforeInsert, &ctx).unwrap();
112    registry.dispatch(HookEvent::BeforeInsert, &ctx).unwrap();
113    println!(
114        "dispatch 后:  {}",
115        call_count.load(std::sync::atomic::Ordering::SeqCst)
116    );
117
118    println!("\n=== ScopeRegistry 作用域控制 ===");
119    let scope_reg = ScopeRegistry::new();
120    println!("soft_delete 启用: {}", scope_reg.is_enabled("soft_delete"));
121    scope_reg.disable("soft_delete");
122    println!("禁用后:           {}", scope_reg.is_enabled("soft_delete"));
123
124    println!("\n=== without_scope 临时禁用 ===");
125    let result = scope_reg.without_scope("tenant", || {
126        println!("  闭包内 tenant 状态: {}", scope_reg.is_enabled("tenant"));
127        42
128    });
129    println!("闭包返回: {}", result);
130    println!("闭包外 tenant 状态: {}", scope_reg.is_enabled("tenant"));
131
132    println!("\n=== Hookable trait 默认钩子调用 ===");
133    println!("执行 before_insert:");
134    let mut ctx2 = HookContext::new();
135    Product::before_insert(&mut ctx2).unwrap();
136
137    println!("执行 after_insert:");
138    Product::after_insert(&ctx2, &100).unwrap();
139
140    println!("执行 before_delete:");
141    Product::before_delete(&mut ctx2, &100).unwrap();
142
143    println!("\n=== Value 类型示例（作用域参数）===");
144    let v = Value::I64(42);
145    println!("Value::I64(42): as_i64() = {:?}", v.as_i64());
146}
