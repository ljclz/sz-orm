1//! 动态 Filter（Hibernate @Filter / @FilterDef 风格）
2//!
3//! 对应文档 6.8 节改进项 40（@Filter 动态 Filter）。
4//!
5//! # 核心概念
6//!
7//! - **FilterDef**：Filter 定义（名称 + 参数列表 + WHERE 子句模板）
8//! - **FilterRegistry**：Filter 注册表（管理 FilterDef + 启用状态 + 参数）
9//! - **apply_filters**：将所有启用的 Filter 追加到 SQL 的 WHERE 子句
10//!
11//! 与 GlobalScope 的区别：
12//! - GlobalScope 是 Model 类型系统层面的"编译期"作用域
13//! - @Filter 是运行时可启用/禁用的动态 Filter，支持参数化
14//!
15//! # 设计灵感
16//!
17//! - Hibernate `@FilterDef` + `@Filter`
18//! - JPA `EntityGraph`
19//! - Rails `default_scope` + `unscoped`
20//!
21//! # 使用示例
22//!
23//! ```no_run
24//! use sz_orm_core::dynamic_filter::{FilterDef, FilterRegistry, FilterParam};
25//! use sz_orm_core::Value;
26//! use std::collections::HashMap;
27//!
28//! // 1. 定义 Filter（类似 @FilterDef + @Filter）
29//! let filter = FilterDef::new("active_users")
30//!     .with_condition("status = :status")
31//!     .with_param(FilterParam::new("status", "active"));
32//!
33//! // 2. 注册并启用
34//! let mut registry = FilterRegistry::new();
35//! registry.register(filter);
36//! registry.enable("active_users", HashMap::new());
37//!
38//! // 3. 应用到 SQL
39//! let sql = registry.apply("SELECT * FROM users");
40//! // → SELECT * FROM users WHERE status = 'active'
41//! ```

50/// Filter 参数定义
51///
52/// 描述 Filter WHERE 子句中 `:param_name` 占位符的类型与默认值。
53#[derive(Debug, Clone)]
54pub struct FilterParam {
55    /// 参数名（对应 `:param_name` 占位符）
56    pub name: String,
57    /// 默认值（启用 Filter 但未传参时使用）
58    pub default_value: Option<crate::Value>,
59    /// 参数描述（用于文档）
60    pub description: String,
61}
62
63impl FilterParam {
64    /// 创建参数定义
65    pub fn new(name: impl Into<String>, default_value: impl Into<crate::Value>) -> Self {
66        Self {
67            name: name.into(),
68            default_value: Some(default_value.into()),
69            description: String::new(),
70        }
71    }
72
73    /// 创建无默认值的参数
74    pub fn required(name: impl Into<String>) -> Self {
75        Self {
76            name: name.into(),
77            default_value: None,
78            description: String::new(),
79        }
80    }
81
82    /// 添加描述
83    pub fn with_description(mut self, desc: impl Into<String>) -> Self {
84        self.description = desc.into();
85        self

93/// Filter 定义
94///
95/// 类似 Hibernate `@FilterDef` + `@Filter` 的组合。
96/// 定义一个具名 Filter，包含 WHERE 子句模板和参数列表。
97///
98/// WHERE 子句模板中使用 `:param_name` 作为参数占位符。
99#[derive(Debug, Clone)]
100pub struct FilterDef {
101    /// Filter 名称（唯一标识）
102    pub name: String,
103    /// WHERE 子句模板（如 `"status = :status"`、`"dept_id IN :dept_ids"`）
104    pub condition: String,
105    /// 参数列表
106    pub params: Vec<FilterParam>,
107    /// 描述（用于文档/调试）
108    pub description: String,
109    /// 默认表名（可选，用于多表 SQL 时指定应用到哪个表）
110    pub table: Option<String>,
111}
112
113impl FilterDef {
114    /// 创建 Filter 定义
115    pub fn new(name: impl Into<String>) -> Self {
116        Self {
117            name: name.into(),
118            condition: String::new(),
119            params: Vec::new(),
120            description: String::new(),
121            table: None,
122        }
123    }
124
125    /// 设置 WHERE 子句模板
126    pub fn with_condition(mut self, condition: impl Into<String>) -> Self {
127        self.condition = condition.into();
128        self
129    }
130
131    /// 添加参数
132    pub fn with_param(mut self, param: FilterParam) -> Self {
133        self.params.push(param);
134        self
135    }
136
137    /// 设置描述
138    pub fn with_description(mut self, desc: impl Into<String>) -> Self {
139        self.description = desc.into();
140        self
141    }
142
143    /// 设置目标表名
144    pub fn with_table(mut self, table: impl Into<String>) -> Self {
145        self.table = Some(table.into());
146        self
147    }
148
149    /// 查找参数定义
150    pub fn find_param(&self, name: &str) -> Option<&FilterParam> {
151        self.params.iter().find(|p| p.name == name)
152    }
153}

156// FilterError — Filter 错误类型
157// ============================================================================
158
159/// Filter 错误类型
160#[derive(Debug)]
161pub enum FilterError {
162    /// Filter 未注册
163    NotRegistered(String),
164    /// 缺少必需参数
165    MissingParam {
166        /// Filter 名
167        filter: String,
168        /// 参数名
169        param: String,
170    },
171    /// 参数类型不匹配
172    ParamTypeMismatch {
173        /// Filter 名
174        filter: String,
175        /// 参数名
176        param: String,
177        /// 期望类型
178        expected: String,
179    },
180    /// Filter 已启用
181    AlreadyEnabled(String),
182    /// Filter 未启用
183    NotEnabled(String),
184    /// 条件模板解析错误
185    TemplateError(String),
186}
187
188impl std::fmt::Display for FilterError {
189    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
190        match self {
191            FilterError::NotRegistered(n) => write!(f, "Filter `{}` not registered", n),
192            FilterError::MissingParam { filter, param } => {
193                write!(f, "Filter `{}` missing required param: `{}`", filter, param)
194            }
195            FilterError::ParamTypeMismatch {
196                filter,
197                param,
198                expected,
199            } => write!(
200                f,
201                "Filter `{}` param `{}` type mismatch, expected: {}",
202                filter, param, expected
203            ),
204            FilterError::AlreadyEnabled(n) => write!(f, "Filter `{}` already enabled", n),
205            FilterError::NotEnabled(n) => write!(f, "Filter `{}` not enabled", n),
206            FilterError::TemplateError(msg) => write!(f, "Filter template error: {}", msg),
207        }
208    }
209}
210
211impl std::error::Error for FilterError {}

233/// Filter 注册表 — 管理 FilterDef + 启用状态 + 运行时参数
234///
235/// 线程安全：内部使用 RwLock，可在多线程环境下共享。
236///
237/// # 示例
238///
239/// ```
240/// use sz_orm_core::dynamic_filter::{FilterDef, FilterRegistry, FilterParam};
241/// use sz_orm_core::Value;
242/// use std::collections::HashMap;
243///
244/// let mut registry = FilterRegistry::new();
245///
246/// // 1. 注册 Filter
247/// let filter = FilterDef::new("active_only")
248///     .with_condition("status = :status")
249///     .with_param(FilterParam::new("status", "active"));
250/// registry.register(filter);
251///
252/// // 2. 启用 Filter（可覆盖默认参数）
253/// let mut params = HashMap::new();
254/// params.insert("status".to_string(), Value::String("pending".to_string()));
255/// registry.enable("active_only", params).unwrap();
256///
257/// // 3. 应用到 SQL
258/// let sql = registry.apply("SELECT * FROM orders");
259/// assert!(sql.contains("status = 'pending'"));
260/// ```
261pub struct FilterRegistry {
262    /// 已注册的 Filter 定义
263    defs: RwLock<HashMap<String, FilterDef>>,
264    /// 已启用的 Filter 实例（按启用顺序）
265    enabled: RwLock<Vec<EnabledFilter>>,
266}
267
268impl Default for FilterRegistry {
269    fn default() -> Self {
270        Self::new()
271    }
272}
273
274impl FilterRegistry {
275    /// 创建空注册表
276    pub fn new() -> Self {
277        Self {
278            defs: RwLock::new(HashMap::new()),
279            enabled: RwLock::new(Vec::new()),
280        }
281    }

283    /// 注册 Filter 定义
284    pub fn register(&self, def: FilterDef) {
285        if let Ok(mut defs) = self.defs.write() {
286            defs.insert(def.name.clone(), def);
287        }
288    }
289
290    /// 注销 Filter 定义
291    pub fn unregister(&self, name: &str) -> bool {
292        if let Ok(mut defs) = self.defs.write() {
293            defs.remove(name).is_some()
294        } else {
295            false
296        }
297    }
298
299    /// 启用 Filter
300    ///
301    /// `params` 中的值会覆盖 FilterDef 中的默认值；缺失参数则使用默认值；
302    /// 若必需参数（无默认值）缺失，返回 `Err`。
303    pub fn enable(&self, name: &str, params: HashMap<String, crate::Value>) -> FilterResult<()> {
304        // 检查是否已注册
305        let def = {
306            let defs = self
307                .defs
308                .read()
309                .map_err(|_| FilterError::NotRegistered(name.to_string()))?;
310            defs.get(name)
311                .ok_or_else(|| FilterError::NotRegistered(name.to_string()))?
312                .clone()
313        };
314
315        // 检查是否已启用
316        {
317            let enabled = self
318                .enabled
319                .read()
320                .map_err(|_| FilterError::NotRegistered(name.to_string()))?;
321            if enabled.iter().any(|e| e.name == name) {
322                return Err(FilterError::AlreadyEnabled(name.to_string()));
323            }
324        }
325
326        // 合并参数：params > default_value
327        let mut merged = HashMap::new();
328        for param_def in &def.params {
329            if let Some(v) = params.get(&param_def.name) {
330                merged.insert(param_def.name.clone(), v.clone());
331            } else if let Some(dv) = &param_def.default_value {
332                merged.insert(param_def.name.clone(), dv.clone());
333            } else {
334                return Err(FilterError::MissingParam {
335                    filter: name.to_string(),
336                    param: param_def.name.clone(),
337                });
338            }
339        }
340
341        if let Ok(mut enabled) = self.enabled.write() {
342            enabled.push(EnabledFilter {
343                name: name.to_string(),
344                params: merged,
345            });
346        }
347        Ok(())
348    }
349
350    /// 禁用 Filter
351    pub fn disable(&self, name: &str) -> FilterResult<()> {
352        if let Ok(mut enabled) = self.enabled.write() {
353            let before = enabled.len();
354            enabled.retain(|e| e.name != name);
355            if enabled.len() == before {
356                Err(FilterError::NotEnabled(name.to_string()))
357            } else {
358                Ok(())
359            }
360        } else {
361            Err(FilterError::NotEnabled(name.to_string()))
362        }
363    }
364
365    /// 检查 Filter 是否已启用
366    pub fn is_enabled(&self, name: &str) -> bool {
367        self.enabled
368            .read()
369            .map(|e| e.iter().any(|f| f.name == name))
370            .unwrap_or(false)
371    }
372
373    /// 已启用的 Filter 数量
374    pub fn enabled_count(&self) -> usize {
375        self.enabled.read().map(|e| e.len()).unwrap_or(0)
376    }
377
378    /// 已注册的 Filter 数量
379    pub fn registered_count(&self) -> usize {
380        self.defs.read().map(|d| d.len()).unwrap_or(0)
381    }
382
383    /// 列出所有已注册 Filter 名称
384    pub fn registered_names(&self) -> Vec<String> {
385        self.defs
386            .read()
387            .map(|d| d.keys().cloned().collect())
388            .unwrap_or_default()
389    }
390
391    /// 列出所有已启用 Filter 名称
392    pub fn enabled_names(&self) -> Vec<String> {
393        self.enabled
394            .read()
395            .map(|e| e.iter().map(|f| f.name.clone()).collect())
396            .unwrap_or_default()
397    }

399    /// 清空所有已启用 Filter（不影响已注册定义）
400    pub fn clear_enabled(&self) {
401        if let Ok(mut e) = self.enabled.write() {
402            e.clear();
403        }
404    }

406    /// 将所有已启用的 Filter 应用到 SQL
407    ///
408    /// 按启用顺序依次追加 WHERE 子句（用 AND 连接）。
409    ///
410    /// # 安全性
411    ///
412    /// 本方法使用 PostgreSQL 方言的转义规则作为默认行为。
413    /// **生产环境请使用 [`FilterRegistry::apply_with_dialect`]** 以获得方言感知的转义。
414    pub fn apply(&self, sql: &str) -> String {
415        // v0.2.2 修复 H-1：默认使用 PostgreSQL 方言（保持向后兼容）
416        self.apply_with_dialect(sql, &crate::dialect::PostgreSqlDialect)
417    }
418
419    /// v0.2.2 修复 H-1：方言感知的 Filter 应用
420    ///
421    /// 与 [`FilterRegistry::apply`] 的区别：使用 `dialect.escape_string()` 转义参数值，
422    /// 确保在所有方言下都安全（特别是 MySQL 默认配置下 backslash 转义）。
423    pub fn apply_with_dialect(&self, sql: &str, dialect: &dyn crate::dialect::Dialect) -> String {
424        let (clauses, has_error) = self.collect_clauses_with_dialect(dialect);
425        if has_error || clauses.is_empty() {
426            return sql.to_string();
427        }
428        append_filter_clauses(sql, &clauses)
429    }
430
431    /// 方言感知的子句收集
432    fn collect_clauses_with_dialect(
433        &self,
434        dialect: &dyn crate::dialect::Dialect,
435    ) -> (Vec<String>, bool) {
436        let mut clauses = Vec::new();
437        let mut has_error = false;
438
439        let enabled = match self.enabled.read() {
440            Ok(e) => e,
441            Err(_) => return (clauses, true),
442        };
443        let defs = match self.defs.read() {
444            Ok(d) => d,
445            Err(_) => return (clauses, true),
446        };
447
448        for ef in enabled.iter() {
449            if let Some(def) = defs.get(&ef.name) {
450                let rendered = render_condition_with_dialect(&def.condition, &ef.params, dialect);
451                if let Some(rendered) = rendered {
452                    if !rendered.trim().is_empty() {
453                        clauses.push(rendered);
454                    }
455                } else {
456                    has_error = true;
457                }
458            }
459        }
460
461        (clauses, has_error)
462    }
463}

465// ============================================================================
466// render_condition — 渲染 WHERE 子句模板
467// ============================================================================
468
469/// 渲染 WHERE 子句模板，将 `:param_name` 替换为实际值
470///
471/// - `:param_name` → 替换为参数值的 SQL 字面量
472/// - 未知参数 → 返回 None（表示渲染失败）
473///
474/// # 安全性
475///
476/// 本函数使用 PostgreSQL 方言的转义规则（`'` → `''`）作为默认行为。
477/// 对 PostgreSQL/SQLite/Oracle/SQL Server 默认配置安全，
478/// 但对 MySQL 默认配置（backslash 是转义字符）不安全。
479/// **生产环境请使用 [`render_condition_with_dialect`]** 以获得方言感知的转义。
480///
481/// # 示例
482///
483/// ```
484/// use sz_orm_core::dynamic_filter::render_condition;
485/// use sz_orm_core::Value;
486/// use std::collections::HashMap;
487///
488/// let mut params = HashMap::new();
489/// params.insert("status".to_string(), Value::String("active".to_string()));
490/// params.insert("min_age".to_string(), Value::I64(18));
491///
492/// let rendered = render_condition("status = :status AND age >= :min_age", &params);
493/// assert_eq!(rendered.unwrap(), "status = 'active' AND age >= 18");
494/// ```
495pub fn render_condition(template: &str, params: &HashMap<String, crate::Value>) -> Option<String> {
496    // v0.2.2 修复 H-1：默认使用 PostgreSQL 方言（保持向后兼容）
497    render_condition_with_dialect(template, params, &crate::dialect::PostgreSqlDialect)
498}
499
500/// v0.2.2 修复 H-1：方言感知的 WHERE 子句模板渲染
501///
502/// 与 [`render_condition`] 的区别：使用 `dialect.escape_string()` 转义字符串参数，
503/// 确保在所有方言下都安全（特别是 MySQL 默认配置下 backslash 转义）。
504///
505/// # 示例
506///
507/// ```
508/// use sz_orm_core::dynamic_filter::render_condition_with_dialect;
509/// use sz_orm_core::dialect::MySqlDialect;
510/// use sz_orm_core::Value;
511/// use std::collections::HashMap;
512///
513/// let mut params = HashMap::new();
514/// params.insert("name".to_string(), Value::String("hello\\nworld".to_string()));
515///
516/// let rendered = render_condition_with_dialect("name = :name", &params, &MySqlDialect);
517/// // MySQL 方言下反斜杠会被转义为 \\\\
518/// assert!(rendered.unwrap().contains("hello\\\\nworld"));
519/// ```
520pub fn render_condition_with_dialect(
521    template: &str,
522    params: &HashMap<String, crate::Value>,
523    dialect: &dyn crate::dialect::Dialect,
524) -> Option<String> {
525    let mut result = String::with_capacity(template.len() + 32);
526    let bytes = template.as_bytes();
527    let mut i = 0;
528
529    while i < bytes.len() {
530        if bytes[i] == b':' {
531            // 收集参数名（字母/数字/下划线）
532            let mut j = i + 1;
533            while j < bytes.len() && (bytes[j].is_ascii_alphanumeric() || bytes[j] == b'_') {
534                j += 1;
535            }
536            if j > i + 1 {
537                let param_name = &template[i + 1..j];
538                // 未知参数时返回 None
539                let value = params.get(param_name)?;
540                result.push_str(&value.to_param_with_dialect(dialect));
541                i = j;
542                continue;
543            }
544        }
545        result.push(bytes[i] as char);
546        i += 1;
547    }
548
549    Some(result)
550}

552// ============================================================================
553// append_filter_clauses — 追加 Filter 子句到 SQL
554// ============================================================================
555
556/// 将 Filter 子句追加到 SQL 的 WHERE 部分
557///
558/// 逻辑与 data_permission::append_where_clauses 相同，但避免循环依赖。
559fn append_filter_clauses(sql: &str, clauses: &[String]) -> String {
560    if clauses.is_empty() {
561        return sql.to_string();
562    }
563
564    let combined = clauses.join(" AND ");
565    let upper = sql.to_uppercase();
566
567    let where_pos = find_keyword_pos(&upper, "WHERE");
568    let group_by_pos = find_keyword_pos(&upper, "GROUP BY");
569    let order_by_pos = find_keyword_pos(&upper, "ORDER BY");
570    let limit_pos = find_keyword_pos(&upper, "LIMIT");
571    let having_pos = find_keyword_pos(&upper, "HAVING");
572
573    let end_pos = [group_by_pos, order_by_pos, limit_pos, having_pos]
574        .iter()
575        .filter_map(|x| *x)
576        .min();
577
578    if let Some(wp) = where_pos {
579        let insert_pos = end_pos.unwrap_or(sql.len());
580        let before = &sql[..wp + 5];
581        let existing_clause = &sql[wp + 5..insert_pos];
582        let after = &sql[insert_pos..];
583
584        let trimmed_existing = existing_clause.trim();
585        if trimmed_existing.is_empty() {
586            format!("{} {}{}", before, combined, after)
587        } else {
588            format!(
589                "{} ({} ) AND ({}){}",
590                before, trimmed_existing, combined, after
591            )
592        }
593    } else {
594        let insert_pos = end_pos.unwrap_or(sql.len());
595        let before = &sql[..insert_pos];
596        let after = &sql[insert_pos..];
597        let trimmed = before.trim_end();
598        let sep = if trimmed.is_empty() { "" } else { " " };
599        format!("{}{}WHERE {}{}", trimmed, sep, combined, after)
600    }
601}

603/// 在 SQL 中查找指定关键字的位置（独立词匹配，大小写不敏感）
604fn find_keyword_pos(sql: &str, keyword: &str) -> Option<usize> {
605    let upper_sql = sql.to_uppercase();
606    let kw_upper = keyword.to_uppercase();
607    let kw_len = kw_upper.len();
608    if kw_len == 0 || upper_sql.len() < kw_len {
609        return None;
610    }
611
612    let bytes = upper_sql.as_bytes();
613    let kw_bytes = kw_upper.as_bytes();
614
615    let mut i = 0;
616    while i + kw_len <= bytes.len() {
617        if &bytes[i..i + kw_len] == kw_bytes {
618            let prev_ok = i == 0 || !bytes[i - 1].is_ascii_alphanumeric() && bytes[i - 1] != b'_';
619            let next_idx = i + kw_len;
620            let next_ok = next_idx >= bytes.len()
621                || !bytes[next_idx].is_ascii_alphanumeric() && bytes[next_idx] != b'_';
622            if prev_ok && next_ok {
623                return Some(i);
624            }
625        }
626        i += 1;
627    }
628    None
629}

742    /// 验证 render_condition 对抗 SQL 注入：
743    /// 用户输入 `' OR 1=1 --` 经过 escape_string + to_param 后，
744    /// 应被包成一个完整的字符串字面量，不会被解析为 SQL 代码。
745    #[test]
746    fn test_render_condition_defends_against_sql_injection() {
747        let mut params = HashMap::new();
748        // 模拟恶意用户输入
749        params.insert("name".to_string(), Value::String("' OR 1=1 --".to_string()));
750
751        let rendered = render_condition("name = :name", &params).unwrap();
752        // 整个值应是一个字符串字面量，不包含未转义的 SQL 代码
753        // 验证：原 `'` 被转义为 `''`，整个值被包在 `'...'` 中
754        assert_eq!(rendered, "name = ''' OR 1=1 --'");
755        // 关键：渲染结果中不应出现"裸露的 OR 1=1"作为 SQL 代码
756        // （即 OR 1=1 应在字符串字面量内部，而非 SQL 操作符）
757        // 通过查找第一个 `=` 后的内容来验证（值部分）
758        // 注意：不能用 split('=').nth(1) 因为值中可能包含 `=`（如 1=1）
759        let eq_pos = rendered.find('=').unwrap();
760        let after_eq = rendered[eq_pos + 1..].trim();
761        assert!(
762            after_eq.starts_with('\''),
763            "value should start with quote, got: {}",
764            after_eq
765        );
766        assert!(
767            after_eq.ends_with('\''),
768            "value should end with quote, got: {}",
769            after_eq
770        );
771        // 验证引号配对：开头的 ''' 中第 1 个 ' 是字面量开始，第 2-3 个 '' 是转义的 '
772        // 最后的 ' 是字面量结束。共 4 个 ' = 1 (start) + 2 (escaped) + 1 (end)
773        let quote_count = after_eq.matches('\'').count();
774        assert_eq!(
775            quote_count, 4,
776            "expected 4 quotes (1 start + 2 escaped + 1 end), got {}",
777            quote_count
778        );
779    }
