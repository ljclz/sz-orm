1//! 脏字段追踪（Dirty Attributes）+ @DynamicInsert / @DynamicUpdate
2//!
3//! 对应文档 6.8 节改进项 24（Dirty Attributes 脏字段追踪）+ 48（@DynamicInsert/@DynamicUpdate）。
4//!
5//! # 核心概念
6//!
7//! - **DirtyTracker**：追踪字段变更状态（原始值快照 vs 当前值），计算哪些字段被修改
8//! - **is_dirty()**：判断是否有脏字段
9//! - **get_dirty_fields()**：返回所有脏字段名
10//! - **mark_clean()**：将当前值重新作为基准（写入完成后调用）
11//! - **get_original()**：获取字段的原始值（写入失败回滚时使用）
12//! - **build_dynamic_update()**：仅生成脏字段的 UPDATE SQL（对应 Hibernate `@DynamicUpdate`）
13//! - **build_dynamic_insert()**：仅生成非 null 字段的 INSERT SQL（对应 Hibernate `@DynamicInsert`）
14//!
15//! # 设计灵感
16//!
17//! - Hibernate `@DynamicInsert` / `@DynamicUpdate`
18//! - Doctrine `ChangeTrackingPolicy::DEFERRED_EXPLICIT`
19//! - Yii2 `ActiveRecord::getDirtyAttributes()`
20//! - Laravel Eloquent `getDirty()` / `getOriginal()`
21//! - MyBatis-Plus `whereEntity` 仅含非 null 字段
22//!
23//! # 使用示例
24//!
25//! ```
26//! use sz_orm_core::dirty_attributes::DirtyTracker;
27//! use sz_orm_core::Value;
28//! use std::collections::HashMap;
29//!
30//! // 1. 加载用户后建立快照
31//! let mut row: HashMap<String, Value> = HashMap::new();
32//! row.insert("id".to_string(), Value::I64(1));
33//! row.insert("name".to_string(), Value::String("alice".to_string()));
34//! row.insert("age".to_string(), Value::I64(25));
35//! let mut tracker = DirtyTracker::new(row);
36//!
37//! // 2. 修改字段
38//! tracker.set("age", Value::I64(26));
39//!
40//! // 3. 检查脏字段
41//! assert!(tracker.is_dirty());
42//! assert_eq!(tracker.get_dirty_fields(), vec!["age"]);
43//! assert_eq!(tracker.get_original("age"), Some(&Value::I64(25)));
44//! ```

50// ============================================================================
51// DirtyTracker — 脏字段追踪器
52// ============================================================================
53
54/// 脏字段追踪器
55///
56/// 通过维护「原始值快照」与「当前值」两份 HashMap，比较得到脏字段集合。
57///
58/// # 设计要点
59///
60/// - **快照机制**：`new()` 时将所有字段视为「原始值」
61/// - **修改追踪**：`set()` 修改字段时，仅写入 `current`，不动 `original`
62/// - **脏字段判定**：`original[field] != current[field]` 则该字段为脏
63/// - **新字段处理**：`set()` 一个 original 中不存在的字段时，自动视为脏字段
64///   （等价于 `original[field] = Null`）
65/// - **mark_clean**：写入成功后调用，将 `original` 同步为 `current`
66/// - **rollback**：写入失败时调用 `rollback()`，将 `current` 还原为 `original`
67#[derive(Debug, Clone)]
68pub struct DirtyTracker {
69    /// 原始值快照（加载时/上次 mark_clean 后的状态）
70    original: HashMap<String, Value>,
71    /// 当前值
72    current: HashMap<String, Value>,
73}
74
75impl DirtyTracker {
76    /// 创建脏字段追踪器，传入初始字段值作为快照
77    pub fn new(initial: HashMap<String, Value>) -> Self {
78        let original = initial.clone();
79        Self {
80            original,
81            current: initial,
82        }
83    }
84
85    /// 创建空追踪器
86    pub fn empty() -> Self {
87        Self {
88            original: HashMap::new(),
89            current: HashMap::new(),
90        }
91    }
92
93    /// 设置字段值（修改 current，不动 original）
94    ///
95    /// 若字段原本不存在（original 中无），则视为新增脏字段。
96    pub fn set(&mut self, field: impl Into<String>, value: Value) {
97        self.current.insert(field.into(), value);
98    }
99
100    /// 批量设置字段值
101    pub fn set_many(&mut self, fields: HashMap<String, Value>) {
102        for (k, v) in fields {
103            self.current.insert(k, v);
104        }
105    }
106
107    /// 获取当前值
108    pub fn get(&self, field: &str) -> Option<&Value> {
109        self.current.get(field)
110    }
111
112    /// 获取原始值（写入失败时可用于回滚业务层）
113    pub fn get_original(&self, field: &str) -> Option<&Value> {
114        self.original.get(field)
115    }
116
117    /// 获取所有字段的当前值（克隆）
118    pub fn current(&self) -> &HashMap<String, Value> {
119        &self.current
120    }
121
122    /// 获取所有字段的原始值（克隆）
123    pub fn original(&self) -> &HashMap<String, Value> {
124        &self.original
125    }
126
127    /// 是否存在脏字段
128    pub fn is_dirty(&self) -> bool {
129        self.original.len() != self.current.len() || self.dirty_fields_iter().next().is_some()
130    }
131
132    /// 判断指定字段是否为脏
133    pub fn is_field_dirty(&self, field: &str) -> bool {
134        match (self.original.get(field), self.current.get(field)) {
135            (None, None) => false,
136            (None, Some(_)) => true, // 新增字段视为脏
137            (Some(_), None) => true, // 删除字段视为脏
138            (Some(o), Some(c)) => o != c,
139        }
140    }
141
142    /// 获取所有脏字段名（按字段名字典序排列，保证测试稳定）
143    pub fn get_dirty_fields(&self) -> Vec<String> {
144        let mut dirty: Vec<String> = self.dirty_fields_iter().cloned().collect();
145        dirty.sort();
146        dirty
147    }
148
149    /// 获取脏字段及其当前值
150    pub fn get_dirty_attributes(&self) -> HashMap<String, Value> {
151        let mut result = HashMap::new();
152        for field in self.dirty_fields_iter() {
153            if let Some(v) = self.current.get(field) {
154                result.insert(field.clone(), v.clone());
155            }
156        }
157        result
158    }
159
160    /// 标记所有字段为干净（写入成功后调用）
161    ///
162    /// 将 `original` 同步为 `current`，下一次 `is_dirty()` 返回 false。
163    pub fn mark_clean(&mut self) {
164        self.original = self.current.clone();
165    }
166
167    /// 回滚：将 `current` 还原为 `original`（写入失败时调用）
168    pub fn rollback(&mut self) {
169        self.current = self.original.clone();
170    }
171
172    /// 重置追踪器，丢弃所有数据
173    pub fn clear(&mut self) {
174        self.original.clear();
175        self.current.clear();
176    }
177
178    /// 内部：脏字段迭代器（避免重复实现）
179    fn dirty_fields_iter(&self) -> impl Iterator<Item = &String> {
180        // current 中所有不在 original 或值不同的字段
181        let keys: Vec<&String> = self.current.keys().collect();
182        keys.into_iter()
183            .filter(move |k| match self.original.get(*k) {
184                None => true, // 新增字段
185                Some(o) => self.current.get(*k).map(|c| c != o).unwrap_or(true),
186            })
187            .chain(
188                // original 中存在但 current 中不存在的字段（被删除）
189                self.original
190                    .keys()
191                    .filter(move |k| !self.current.contains_key(*k)),
192            )
193    }
194}

196// ============================================================================
197// build_dynamic_update — 仅生成脏字段的 UPDATE SQL
198// ============================================================================
199
200/// 生成仅含脏字段的 UPDATE SQL（对应 Hibernate `@DynamicUpdate`）
201///
202/// 生成的 SQL 形如：
203/// ```sql
204/// UPDATE `table` SET `col1` = ?, `col2` = ? WHERE `pk` = ?
205/// ```
206///
207/// # 参数
208/// - `dialect`：数据库方言
209/// - `table`：表名
210/// - `pk_column`：主键列名
211/// - `pk_value`：主键值
212/// - `tracker`：脏字段追踪器
213///
214/// # 返回
215/// - `Some(sql)`：存在脏字段时返回 UPDATE SQL
216/// - `None`：无脏字段，无需更新
217///
218/// # 示例
219///
220/// ```
221/// use sz_orm_core::dirty_attributes::{DirtyTracker, build_dynamic_update};
222/// use sz_orm_core::{DbType, get_dialect, Value};
223/// use std::collections::HashMap;
224///
225/// let dialect = get_dialect(DbType::MySQL).unwrap();
226/// let mut row = HashMap::new();
227/// row.insert("id".to_string(), Value::I64(1));
228/// row.insert("name".to_string(), Value::String("alice".to_string()));
229/// row.insert("age".to_string(), Value::I64(25));
230/// let mut tracker = DirtyTracker::new(row);
231/// tracker.set("age", Value::I64(26));
232///
233/// let sql = build_dynamic_update(&*dialect, "users", "id", &Value::I64(1), &tracker).unwrap();
234/// assert!(sql.contains("UPDATE `users` SET"));
235/// assert!(sql.contains("`age` = 26"));
236/// assert!(!sql.contains("`name`")); // 未修改字段不出现
237/// ```
238pub fn build_dynamic_update(
239    dialect: &dyn Dialect,
240    table: &str,
241    pk_column: &str,
242    pk_value: &Value,
243    tracker: &DirtyTracker,
244) -> Option<String> {
245    let dirty = tracker.get_dirty_attributes();
246    if dirty.is_empty() {
247        return None;
248    }
249
250    let quoted_table = dialect.quote(table);
251    let quoted_pk = dialect.quote(pk_column);
252
253    // 按字段名字典序排列，保证 SQL 输出稳定
254    let mut fields: Vec<&String> = dirty.keys().collect();
255    fields.sort();
256
257    let sets: Vec<String> = fields
258        .iter()
259        .map(|k| {
260            format!(
261                "{} = {}",
262                dialect.quote(k),
263                dirty[*k].to_param_with_dialect(dialect)
264            )
265        })
266        .collect();
267    let sets_sql = sets.join(", ");
268
269    Some(format!(
270        "UPDATE {} SET {} WHERE {} = {}",
271        quoted_table,
272        sets_sql,
273        quoted_pk,
274        pk_value.to_param_with_dialect(dialect),
275    ))
276}

278// ============================================================================
279// build_dynamic_insert — 仅生成非 null 字段的 INSERT SQL
280// ============================================================================
281
282/// 生成仅含非 null 字段的 INSERT SQL（对应 Hibernate `@DynamicInsert`）
283///
284/// 生成的 SQL 形如：
285/// ```sql
286/// INSERT INTO `table` (`col1`, `col2`) VALUES (?, ?)
287/// ```
288///
289/// `Value::Null` 字段会被排除，让数据库使用列默认值。
290///
291/// # 参数
292/// - `dialect`：数据库方言
293/// - `table`：表名
294/// - `data`：要插入的字段（Null 字段会被过滤）
295///
296/// # 返回
297/// - `Some(sql)`：存在非 null 字段时返回 INSERT SQL
298/// - `None`：所有字段均为 Null，无法生成 INSERT
299///
300/// # 示例
301///
302/// ```
303/// use sz_orm_core::dirty_attributes::build_dynamic_insert;
304/// use sz_orm_core::{DbType, get_dialect, Value};
305/// use std::collections::HashMap;
306///
307/// let dialect = get_dialect(DbType::MySQL).unwrap();
308/// let mut data = HashMap::new();
309/// data.insert("name".to_string(), Value::String("alice".to_string()));
310/// data.insert("age".to_string(), Value::I64(25));
311/// data.insert("bio".to_string(), Value::Null); // Null 字段被排除
312///
313/// let sql = build_dynamic_insert(&*dialect, "users", &data).unwrap();
314/// assert!(sql.contains("INSERT INTO `users`"));
315/// assert!(sql.contains("`name`"));
316/// assert!(sql.contains("`age`"));
317/// assert!(!sql.contains("`bio`"));
318/// ```
319pub fn build_dynamic_insert(
320    dialect: &dyn Dialect,
321    table: &str,
322    data: &HashMap<String, Value>,
323) -> Option<String> {
324    // 过滤 Null 字段
325    let non_null: Vec<(&String, &Value)> = data
326        .iter()
327        .filter(|(_, v)| !matches!(v, Value::Null))
328        .collect();
329
330    if non_null.is_empty() {
331        return None;
332    }
333
334    // 按字段名字典序排列
335    let mut sorted = non_null.clone();
336    sorted.sort_by(|a, b| a.0.cmp(b.0));
337
338    let quoted_table = dialect.quote(table);
339    let columns: Vec<String> = sorted.iter().map(|(k, _)| dialect.quote(k)).collect();
340    let values: Vec<String> = sorted
341        .iter()
342        .map(|(_, v)| v.to_param_with_dialect(dialect).to_string())
343        .collect();
344
345    Some(format!(
346        "INSERT INTO {} ({}) VALUES ({})",
347        quoted_table,
348        columns.join(", "),
349        values.join(", "),
350    ))
351}

738        assert_eq!(col_count, 3);

756    fn test_workflow_load_modify_save() {
757        let dialect = get_dialect(DbType::MySQL).unwrap();
758
759        // 模拟从数据库加载
760        let mut row = HashMap::new();
761        row.insert("id".to_string(), Value::I64(1));
762        row.insert("name".to_string(), Value::String("alice".to_string()));
763        row.insert("age".to_string(), Value::I64(25));
764        row.insert(
765            "updated_at".to_string(),
766            Value::String("2026-01-01".to_string()),
767        );
768
769        let mut tracker = DirtyTracker::new(row);
770
771        // 修改字段
772        tracker.set("age", Value::I64(26));
773        tracker.set("updated_at", Value::String("2026-07-19".to_string()));
774
775        // 生成 UPDATE
776        let sql = build_dynamic_update(&*dialect, "users", "id", &Value::I64(1), &tracker).unwrap();
777        assert!(sql.contains("`age` = 26"));
778        assert!(sql.contains("`updated_at` = '2026-07-19'"));
779        let set_clause = sql.split("WHERE").next().unwrap();
780        assert!(!set_clause.contains("`name`"));
781        assert!(!set_clause.contains("`id` ="));
782
783        // 模拟写入成功，标记干净
784        tracker.mark_clean();
785        assert!(!tracker.is_dirty());
786
787        // 再次修改
788        tracker.set("name", Value::String("bob".to_string()));
789        assert!(tracker.is_dirty());
790        assert_eq!(tracker.get_dirty_fields(), vec!["name"]);
791    }
