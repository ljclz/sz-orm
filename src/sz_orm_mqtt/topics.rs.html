3#[derive(Debug, Clone)]
4pub struct TopicFilter {
5    pattern: String,
6    levels: Vec<String>,
7}
8
9impl TopicFilter {
10    pub fn new(pattern: impl Into<String>) -> Result<Self, MqttError> {
11        let pattern = pattern.into();
12        if pattern.is_empty() {
13            return Err(MqttError::Topic("Topic filter cannot be empty".to_string()));
14        }
15        let levels: Vec<String> = pattern.split('/').map(|s| s.to_string()).collect();
16        for (i, level) in levels.iter().enumerate() {
17            if level.contains('#') && (level.len() != 1 || i != levels.len() - 1) {
18                return Err(MqttError::Topic(format!(
19                    "Invalid topic filter: '#' must occupy the entire last level: {}",
20                    pattern
21                )));
22            }
23            if level.contains('+') && level.len() != 1 {
24                return Err(MqttError::Topic(format!(
25                    "Invalid topic filter: '+' must occupy an entire level: {}",
26                    pattern
27                )));
28            }
29        }
30        Ok(Self { pattern, levels })
31    }
32
33    pub fn pattern(&self) -> &str {
34        &self.pattern
35    }
36
37    pub fn levels(&self) -> &[String] {
38        &self.levels
39    }
40
41    pub fn has_wildcard(&self) -> bool {
42        self.pattern.contains('#') || self.pattern.contains('+')
43    }
44
45    pub fn matches(&self, topic: &str) -> bool {
46        topic_matches(topic, &self.pattern)
47    }
48}

50impl From<&str> for TopicFilter {
51    fn from(s: &str) -> Self {
52        TopicFilter::new(s).unwrap_or_else(|_| TopicFilter {
53            pattern: s.to_string(),
54            levels: s.split('/').map(|l| l.to_string()).collect(),
55        })
56    }
57}

59impl From<String> for TopicFilter {
60    fn from(s: String) -> Self {
61        TopicFilter::from(s.as_str())
62    }
63}

65pub fn topic_matches(topic: &str, filter: &str) -> bool {
66    if topic.is_empty() || filter.is_empty() {
67        return false;
68    }
69
70    let topic_levels: Vec<&str> = topic.split('/').collect();
71    let filter_levels: Vec<&str> = filter.split('/').collect();
72
73    let mut topic_idx = 0;
74    let mut filter_idx = 0;
75
76    while filter_idx < filter_levels.len() {
77        let filter_part = filter_levels[filter_idx];
78
79        if filter_part == "#" {
80            return true;
81        }
82
83        if filter_part == "+" {
84            if topic_idx >= topic_levels.len() {
85                return false;
86            }
87            topic_idx += 1;
88            filter_idx += 1;
89            continue;
90        }
91
92        if topic_idx >= topic_levels.len() {
93            return false;
94        }
95
96        if filter_part != topic_levels[topic_idx] {
97            return false;
98        }
99
100        topic_idx += 1;
101        filter_idx += 1;
102    }
103
104    topic_idx == topic_levels.len()
105}
