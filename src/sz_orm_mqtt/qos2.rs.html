1//! # QoS 2 四次握手状态机
2//!
3//! 实现 MQTT QoS 2（Exactly Once）消息传输的完整四次握手流程：
4//!
5//! ```text
6//! 发送方                接收方
7//!   |  PUBLISH(packet_id)  ->    |
8//!   |  <-  PUBREC(packet_id)     |
9//!   |  PUBREL(packet_id)   ->    |
10//!   |  <-  PUBCOMP(packet_id)    |
11//! ```
12//!
13//! 发送方状态：Init -> Published -> Received -> Released -> Complete
14//! 接收方状态：Init -> Received -> Released -> Complete
15//!
16//! 任何阶段异常可通过 [`PacketState`] 查询并支持重传。
17
18use std::collections::HashMap;
19use std::sync::Arc;
20use tokio::sync::RwLock;
21
22/// QoS 2 四次握手阶段
23#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
24pub enum PacketState {
25    /// 初始：尚未发送/接收 PUBLISH
26    Init,
27    /// 已发送/接收 PUBLISH
28    Published,
29    /// 已发送/接收 PUBREC（发送方收到，接收方已发送）
30    Received,
31    /// 已发送/接收 PUBREL
32    Released,
33    /// 已发送/接收 PUBCOMP，握手完成
34    Complete,
35}
36
37impl PacketState {
38    /// 是否处于握手进行中（未完成）
39    pub fn in_progress(&self) -> bool {
40        !matches!(self, PacketState::Complete)
41    }
42
43    /// 是否已完成
44    pub fn is_complete(&self) -> bool {
45        matches!(self, PacketState::Complete)
46    }
47}
48
49/// 单个 QoS 2 数据包的握手跟踪记录
50#[derive(Debug, Clone)]
51pub struct Qos2Packet {
52    /// 数据包 ID（MQTT 协议中 1-65535）
53    pub packet_id: u16,
54    /// 主题
55    pub topic: String,
56    /// 当前状态
57    pub state: PacketState,
58    /// 重试次数
59    pub retries: u8,
60}
61
62impl Qos2Packet {
63    pub fn new(packet_id: u16, topic: impl Into<String>) -> Self {
64        Self {
65            packet_id,
66            topic: topic.into(),
67            state: PacketState::Init,
68            retries: 0,
69        }
70    }
71
72    /// 推进到下一状态，返回是否成功
73    pub fn advance(&mut self) -> bool {
74        let next = match self.state {
75            PacketState::Init => PacketState::Published,
76            PacketState::Published => PacketState::Received,
77            PacketState::Received => PacketState::Released,
78            PacketState::Released => PacketState::Complete,
79            PacketState::Complete => return false,
80        };
81        self.state = next;
82        true
83    }
84
85    /// 重置回 Init（用于重传 PUBLISH）
86    pub fn reset(&mut self) {
87        self.state = PacketState::Init;
88        self.retries = self.retries.saturating_add(1);
89    }
90}

92/// 发送方视角的 QoS 2 状态跟踪表
93#[derive(Debug, Default)]
94pub struct SenderState {
95    packets: Arc<RwLock<HashMap<u16, Qos2Packet>>>,
96    /// 下一个未使用的 packet_id（自增）
97    next_id: Arc<RwLock<u16>>,
98}
99
100impl SenderState {
101    pub fn new() -> Self {
102        Self {
103            packets: Arc::new(RwLock::new(HashMap::new())),
104            next_id: Arc::new(RwLock::new(1)),
105        }
106    }
107
108    /// 分配一个新的 packet_id（1-65535 循环使用）
109    pub async fn allocate_id(&self) -> u16 {
110        let mut next = self.next_id.write().await;
111        let id = *next;
112        *next = if *next == u16::MAX { 1 } else { *next + 1 };
113        id
114    }
115
116    /// 注册一个新的 QoS 2 发送（PUBLISH 已发送）
117    pub async fn register_publish(
118        &self,
119        packet_id: u16,
120        topic: impl Into<String>,
121    ) -> Result<(), String> {
122        let mut packets = self.packets.write().await;
123        if packets.contains_key(&packet_id) {
124            return Err(format!("packet_id {} already in use", packet_id));
125        }
126        let mut packet = Qos2Packet::new(packet_id, topic);
127        packet.state = PacketState::Published;
128        packets.insert(packet_id, packet);
129        Ok(())
130    }
131
132    /// 标记收到 PUBREC（PUBLISH 已被对端确认）
133    pub async fn receive_pubrec(&self, packet_id: u16) -> Result<(), String> {
134        let mut packets = self.packets.write().await;
135        let packet = packets
136            .get_mut(&packet_id)
137            .ok_or_else(|| format!("unknown packet_id {}", packet_id))?;
138        if packet.state != PacketState::Published {
139            return Err(format!(
140                "unexpected PUBREC for packet {} in state {:?}",
141                packet_id, packet.state
142            ));
143        }
144        packet.state = PacketState::Received;
145        Ok(())
146    }
147
148    /// 标记已发送 PUBREL
149    pub async fn send_pubrel(&self, packet_id: u16) -> Result<(), String> {
150        let mut packets = self.packets.write().await;
151        let packet = packets
152            .get_mut(&packet_id)
153            .ok_or_else(|| format!("unknown packet_id {}", packet_id))?;
154        if packet.state != PacketState::Received {
155            return Err(format!(
156                "unexpected PUBREL send for packet {} in state {:?}",
157                packet_id, packet.state
158            ));
159        }
160        packet.state = PacketState::Released;
161        Ok(())
162    }
163
164    /// 标记收到 PUBCOMP，握手完成
165    pub async fn receive_pubcomp(&self, packet_id: u16) -> Result<(), String> {
166        let mut packets = self.packets.write().await;
167        let packet = packets
168            .get_mut(&packet_id)
169            .ok_or_else(|| format!("unknown packet_id {}", packet_id))?;
170        if packet.state != PacketState::Released {
171            return Err(format!(
172                "unexpected PUBCOMP for packet {} in state {:?}",
173                packet_id, packet.state
174            ));
175        }
176        packet.state = PacketState::Complete;
177        // 完成后从跟踪表中移除
178        packets.remove(&packet_id);
179        Ok(())
180    }
181
182    /// 查询某 packet 的当前状态
183    pub async fn state_of(&self, packet_id: u16) -> Option<PacketState> {
184        let packets = self.packets.read().await;
185        packets.get(&packet_id).map(|p| p.state)
186    }
187
188    /// 当前进行中（未完成）的握手数量
189    pub async fn in_progress_count(&self) -> usize {
190        let packets = self.packets.read().await;
191        packets.values().filter(|p| p.state.in_progress()).count()
192    }
193
194    /// 当前跟踪的所有 packet_id 列表
195    pub async fn tracked_ids(&self) -> Vec<u16> {
196        let packets = self.packets.read().await;
197        let mut ids: Vec<u16> = packets.keys().copied().collect();
198        ids.sort();
199        ids
200    }
201}

203/// 接收方视角的 QoS 2 状态跟踪表
204#[derive(Debug, Default)]
205pub struct ReceiverState {
206    packets: Arc<RwLock<HashMap<u16, Qos2Packet>>>,
207}
208
209impl ReceiverState {
210    pub fn new() -> Self {
211        Self {
212            packets: Arc::new(RwLock::new(HashMap::new())),
213        }
214    }
215
216    /// 收到 PUBLISH：注册并返回 PUBREC 应发送
217    pub async fn receive_publish(
218        &self,
219        packet_id: u16,
220        topic: impl Into<String>,
221    ) -> Result<(), String> {
222        let mut packets = self.packets.write().await;
223        if packets.contains_key(&packet_id) {
224            // 重复 PUBLISH：可能是对端重传，保持原状态返回
225            return Err(format!("duplicate PUBLISH for packet_id {}", packet_id));
226        }
227        let mut packet = Qos2Packet::new(packet_id, topic);
228        packet.state = PacketState::Published;
229        packets.insert(packet_id, packet);
230        Ok(())
231    }
232
233    /// 标记已发送 PUBREC
234    pub async fn send_pubrec(&self, packet_id: u16) -> Result<(), String> {
235        let mut packets = self.packets.write().await;
236        let packet = packets
237            .get_mut(&packet_id)
238            .ok_or_else(|| format!("unknown packet_id {}", packet_id))?;
239        if packet.state != PacketState::Published {
240            return Err(format!(
241                "unexpected PUBREC send for packet {} in state {:?}",
242                packet_id, packet.state
243            ));
244        }
245        packet.state = PacketState::Received;
246        Ok(())
247    }
248
249    /// 收到 PUBREL：标记并返回 PUBCOMP 应发送
250    pub async fn receive_pubrel(&self, packet_id: u16) -> Result<(), String> {
251        let mut packets = self.packets.write().await;
252        let packet = packets
253            .get_mut(&packet_id)
254            .ok_or_else(|| format!("unknown packet_id {}", packet_id))?;
255        if packet.state != PacketState::Received {
256            return Err(format!(
257                "unexpected PUBREL for packet {} in state {:?}",
258                packet_id, packet.state
259            ));
260        }
261        packet.state = PacketState::Released;
262        Ok(())
263    }
264
265    /// 标记已发送 PUBCOMP，握手完成并清除
266    pub async fn send_pubcomp(&self, packet_id: u16) -> Result<(), String> {
267        let mut packets = self.packets.write().await;
268        let packet = packets
269            .get_mut(&packet_id)
270            .ok_or_else(|| format!("unknown packet_id {}", packet_id))?;
271        if packet.state != PacketState::Released {
272            return Err(format!(
273                "unexpected PUBCOMP send for packet {} in state {:?}",
274                packet_id, packet.state
275            ));
276        }
277        packet.state = PacketState::Complete;
278        packets.remove(&packet_id);
279        Ok(())
280    }
281
282    /// 查询状态
283    pub async fn state_of(&self, packet_id: u16) -> Option<PacketState> {
284        let packets = self.packets.read().await;
285        packets.get(&packet_id).map(|p| p.state)
286    }
287
288    /// 当前进行中的握手数量
289    pub async fn in_progress_count(&self) -> usize {
290        let packets = self.packets.read().await;
291        packets.values().filter(|p| p.state.in_progress()).count()
292    }
293}

295/// 完整的 QoS 2 四次握手模拟器，用于测试与教学
296pub struct Qos2HandshakeSimulator {
297    sender: SenderState,
298    receiver: ReceiverState,
299}
300
301impl Qos2HandshakeSimulator {
302    pub fn new() -> Self {
303        Self {
304            sender: SenderState::new(),
305            receiver: ReceiverState::new(),
306        }
307    }
308
309    /// 模拟一次完整的 QoS 2 消息发送流程
310    ///
311    /// 返回 Ok(()) 表示握手完成，Err 描述失败原因
312    pub async fn simulate(&self, packet_id: u16, topic: &str) -> Result<(), String> {
313        // 1. 发送方 PUBLISH
314        self.sender.register_publish(packet_id, topic).await?;
315        // 2. 接收方接收 PUBLISH
316        self.receiver.receive_publish(packet_id, topic).await?;
317        // 3. 接收方发送 PUBREC
318        self.receiver.send_pubrec(packet_id).await?;
319        // 4. 发送方接收 PUBREC
320        self.sender.receive_pubrec(packet_id).await?;
321        // 5. 发送方发送 PUBREL
322        self.sender.send_pubrel(packet_id).await?;
323        // 6. 接收方接收 PUBREL
324        self.receiver.receive_pubrel(packet_id).await?;
325        // 7. 接收方发送 PUBCOMP
326        self.receiver.send_pubcomp(packet_id).await?;
327        // 8. 发送方接收 PUBCOMP
328        self.sender.receive_pubcomp(packet_id).await?;
329        Ok(())
330    }
331
332    pub fn sender(&self) -> &SenderState {
333        &self.sender
334    }
335
336    pub fn receiver(&self) -> &ReceiverState {
337        &self.receiver
338    }
339}

342    fn default() -> Self {
343        Self::new()
344    }
