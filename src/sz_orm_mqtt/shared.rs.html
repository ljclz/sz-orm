1//! # 共享订阅（Shared Subscriptions）
2//!
3//! 实现 MQTT 5.0 共享订阅功能：多个客户端以同一组名订阅同一过滤器时，
4//! broker 将匹配的消息在该组内负载均衡分发（而非广播给每个订阅者）。
5//!
6//! 共享订阅语法：`$share/{group_name}/{topic_filter}`
7//!
8//! ## 主要类型
9//!
10//! - [`SharedSubscription`] — 共享订阅条目
11//! - [`SharedSubscriber`] — 组内订阅者
12//! - [`SharedSubscriptionRegistry`] — 共享订阅注册表
13//! - [`LoadBalanceStrategy`] — 负载均衡策略
14
15use crate::broker::QoS;
16use crate::topics::topic_matches;
17use std::collections::HashMap;
18use std::sync::Arc;
19use tokio::sync::RwLock;
20
21/// 共享订阅前缀
22pub const SHARED_PREFIX: &str = "$share/";
23
24/// 负载均衡策略
25#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
26pub enum LoadBalanceStrategy {
27    /// 轮询（Round Robin）—— 默认
28    #[default]
29    RoundRobin,
30    /// 随机
31    Random,
32    /// 最少未处理（选择当前未确认消息最少的订阅者）
33    LeastPending,
34}
35
36/// 共享订阅中的单个订阅者
37#[derive(Debug, Clone, PartialEq, Eq)]
38pub struct SharedSubscriber {
39    /// 客户端 ID
40    pub client_id: String,
41    /// 订阅 QoS
42    pub qos: QoS,
43}
44
45impl SharedSubscriber {
46    pub fn new(client_id: impl Into<String>, qos: QoS) -> Self {
47        Self {
48            client_id: client_id.into(),
49            qos,
50        }
51    }
52}

54/// 共享订阅组
55#[derive(Debug, Clone)]
56pub struct SharedSubscription {
57    /// 组名
58    pub group_name: String,
59    /// 主题过滤器（不含 `$share/{group}/` 前缀）
60    pub topic_filter: String,
61    /// 组内订阅者列表
62    pub subscribers: Vec<SharedSubscriber>,
63    /// 负载均衡策略
64    pub strategy: LoadBalanceStrategy,
65    /// 轮询索引（仅 RoundRobin 使用）
66    pub rr_index: usize,
67}
68
69impl SharedSubscription {
70    pub fn new(
71        group_name: impl Into<String>,
72        topic_filter: impl Into<String>,
73    ) -> Self {
74        Self {
75            group_name: group_name.into(),
76            topic_filter: topic_filter.into(),
77            subscribers: Vec::new(),
78            strategy: LoadBalanceStrategy::default(),
79            rr_index: 0,
80        }
81    }
82
83    /// 设置负载均衡策略
84    pub fn with_strategy(mut self, strategy: LoadBalanceStrategy) -> Self {
85        self.strategy = strategy;
86        self
87    }
88
89    /// 添加订阅者
90    pub fn add_subscriber(&mut self, subscriber: SharedSubscriber) -> bool {
91        if self.subscribers.iter().any(|s| s.client_id == subscriber.client_id) {
92            return false; // 已存在
93        }
94        self.subscribers.push(subscriber);
95        true
96    }
97
98    /// 移除订阅者
99    pub fn remove_subscriber(&mut self, client_id: &str) -> bool {
100        let before = self.subscribers.len();
101        self.subscribers.retain(|s| s.client_id != client_id);
102        self.subscribers.len() != before
103    }
104
105    /// 更新订阅者 QoS（若已存在则更新，不存在则添加）
106    pub fn upsert_subscriber(&mut self, client_id: &str, qos: QoS) {
107        if let Some(s) = self.subscribers.iter_mut().find(|s| s.client_id == client_id) {
108            s.qos = qos;
109        } else {
110            self.subscribers.push(SharedSubscriber::new(client_id, qos));
111        }
112    }
113
114    /// 当前订阅者数量
115    pub fn subscriber_count(&self) -> usize {
116        self.subscribers.len()
117    }
118
119    /// 是否包含某订阅者
120    pub fn contains(&self, client_id: &str) -> bool {
121        self.subscribers.iter().any(|s| s.client_id == client_id)
122    }
123
124    /// 主题是否匹配过滤器
125    pub fn matches(&self, topic: &str) -> bool {
126        topic_matches(topic, &self.topic_filter)
127    }
128
129    /// 选择下一个接收消息的订阅者（按策略）
130    /// 返回选中的订阅者索引；组为空时返回 None
131    pub fn select_next(&mut self) -> Option<usize> {
132        if self.subscribers.is_empty() {
133            return None;
134        }
135        let idx = match self.strategy {
136            LoadBalanceStrategy::RoundRobin => {
137                let i = self.rr_index % self.subscribers.len();
138                self.rr_index = self.rr_index.wrapping_add(1);
139                i
140            }
141            LoadBalanceStrategy::Random => {
142                // 简单确定性伪随机：基于 subscriber_count 与 rr_index
143                // 不引入 rand 依赖
144                let seed = self.rr_index.wrapping_add(self.subscribers.len());
145                seed % self.subscribers.len()
146            }
147            LoadBalanceStrategy::LeastPending => {
148                // 无 pending 跟踪时退化为 RoundRobin
149                let i = self.rr_index % self.subscribers.len();
150                self.rr_index = self.rr_index.wrapping_add(1);
151                i
152            }
153        };
154        Some(idx)
155    }
156}

158/// 解析共享订阅主题过滤器。
159/// 输入形如 `$share/group_name/topic/filter`，返回 `(group_name, topic_filter)`。
160/// 若输入不以 `$share/` 开头，返回 None。
161pub fn parse_shared_filter(filter: &str) -> Option<(&str, &str)> {
162    let rest = filter.strip_prefix(SHARED_PREFIX)?;
163    let slash = rest.find('/')?;
164    let group = &rest[..slash];
165    let topic = &rest[slash + 1..];
166    if group.is_empty() || topic.is_empty() {
167        return None;
168    }
169    Some((group, topic))
170}
171
172/// 判断是否为共享订阅过滤器
173pub fn is_shared_filter(filter: &str) -> bool {
174    parse_shared_filter(filter).is_some()
175}

177/// 共享订阅注册表
178#[derive(Debug, Default)]
179pub struct SharedSubscriptionRegistry {
180    /// key: "{group_name}/{topic_filter}"
181    subscriptions: Arc<RwLock<HashMap<String, SharedSubscription>>>,
182}
183
184impl SharedSubscriptionRegistry {
185    pub fn new() -> Self {
186        Self {
187            subscriptions: Arc::new(RwLock::new(HashMap::new())),
188        }
189    }
190
191    fn key(group: &str, filter: &str) -> String {
192        format!("{}/{}", group, filter)
193    }
194
195    /// 注册共享订阅（若组不存在则创建，存在则添加订阅者）
196    pub async fn subscribe(
197        &self,
198        group: &str,
199        filter: &str,
200        client_id: &str,
201        qos: QoS,
202    ) -> Result<(), String> {
203        // 校验过滤器合法性（复用 TopicFilter 校验逻辑）
204        let _ = crate::topics::TopicFilter::new(filter)
205            .map_err(|e| format!("invalid topic filter: {}", e))?;
206
207        let mut subs = self.subscriptions.write().await;
208        let key = Self::key(group, filter);
209        let entry = subs.entry(key).or_insert_with(|| {
210            SharedSubscription::new(group, filter)
211        });
212        entry.upsert_subscriber(client_id, qos);
213        Ok(())
214    }
215
216    /// 取消某客户端在某组+过滤器下的订阅
217    /// 返回是否成功移除（false 表示订阅者不存在）
218    pub async fn unsubscribe(
219        &self,
220        group: &str,
221        filter: &str,
222        client_id: &str,
223    ) -> bool {
224        let mut subs = self.subscriptions.write().await;
225        let key = Self::key(group, filter);
226        if let Some(sub) = subs.get_mut(&key) {
227            let removed = sub.remove_subscriber(client_id);
228            // 若组已空，移除整个条目
229            if sub.subscriber_count() == 0 {
230                subs.remove(&key);
231            }
232            return removed;
233        }
234        false
235    }
236
237    /// 移除某客户端在所有共享订阅组中的订阅（用于客户端断开连接）
238    /// 返回被移除的组数量
239    pub async fn unsubscribe_all(&self, client_id: &str) -> usize {
240        let mut subs = self.subscriptions.write().await;
241        let mut removed_groups = 0;
242        let mut empty_keys = Vec::new();
243        for sub in subs.values_mut() {
244            if sub.remove_subscriber(client_id) {
245                removed_groups += 1;
246            }
247            if sub.subscriber_count() == 0 {
248                empty_keys.push(Self::key(&sub.group_name, &sub.topic_filter));
249            }
250        }
251        for key in empty_keys {
252            subs.remove(&key);
253        }
254        removed_groups
255    }
256
257    /// 查询某组+过滤器下的订阅者数量
258    pub async fn subscriber_count(&self, group: &str, filter: &str) -> usize {
259        let subs = self.subscriptions.read().await;
260        let key = Self::key(group, filter);
261        subs.get(&key).map(|s| s.subscriber_count()).unwrap_or(0)
262    }
263
264    /// 查询所有匹配指定主题的共享订阅组（用于消息分发）
265    /// 返回 (group_name, topic_filter, subscriber_count) 列表
266    pub async fn groups_matching(&self, topic: &str) -> Vec<(String, String, usize)> {
267        let subs = self.subscriptions.read().await;
268        let mut result: Vec<(String, String, usize)> = subs
269            .values()
270            .filter(|s| s.matches(topic))
271            .map(|s| (s.group_name.clone(), s.topic_filter.clone(), s.subscriber_count()))
272            .collect();
273        result.sort_by(|a, b| a.0.cmp(&b.0).then(a.1.cmp(&b.1)));
274        result
275    }
276
277    /// 为消息选择接收者：对每个匹配的共享订阅组，按策略选一个订阅者。
278    /// 返回 (client_id, qos) 列表。
279    pub async fn select_recipients(&self, topic: &str) -> Vec<(String, QoS)> {
280        let mut subs = self.subscriptions.write().await;
281        let mut recipients = Vec::new();
282        // 收集匹配的 key（避免持有可变借用同时遍历）
283        let matching_keys: Vec<String> = subs
284            .iter()
285            .filter(|(_, s)| s.matches(topic))
286            .map(|(k, _)| k.clone())
287            .collect();
288        for key in matching_keys {
289            if let Some(sub) = subs.get_mut(&key) {
290                if let Some(idx) = sub.select_next() {
291                    let s = &sub.subscribers[idx];
292                    recipients.push((s.client_id.clone(), s.qos));
293                }
294            }
295        }
296        recipients.sort_by(|a, b| a.0.cmp(&b.0));
297        recipients
298    }
299
300    /// 当前注册的共享订阅组数量
301    pub async fn group_count(&self) -> usize {
302        let subs = self.subscriptions.read().await;
303        subs.len()
304    }
305
306    /// 列出所有组信息
307    pub async fn list_groups(&self) -> Vec<(String, String, usize)> {
308        let subs = self.subscriptions.read().await;
309        let mut result: Vec<(String, String, usize)> = subs
310            .values()
311            .map(|s| (s.group_name.clone(), s.topic_filter.clone(), s.subscriber_count()))
312            .collect();
313        result.sort_by(|a, b| a.0.cmp(&b.0).then(a.1.cmp(&b.1)));
314        result
315    }
316}

558        reg.subscribe("g1", "t", "c1", QoS::AtMostOnce).await.unwrap();
559        reg.subscribe("g1", "t", "c2", QoS::AtMostOnce).await.unwrap();
