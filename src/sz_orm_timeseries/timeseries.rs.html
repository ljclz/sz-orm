8/// TimescaleDB 时序扩展 trait
9#[async_trait]
10pub trait TimeseriesExt: Send + Sync {
11    /// 创建 hypertable（TimescaleDB 核心概念：按时间自动分区）
12    ///
13    /// 等价 SQL：`SELECT create_hypertable('table', 'time_column')`
14    async fn create_hypertable(&self, table: &str, time_column: &str)
15        -> Result<(), TimescaleError>;
16
17    /// 插入单个指标
18    async fn insert_metric(&self, metric: &Metric) -> Result<(), TimescaleError>;
19
20    /// 批量插入指标
21    async fn insert_metrics(&self, metrics: &[Metric]) -> Result<(), TimescaleError> {
22        for m in metrics {
23            self.insert_metric(m).await?;
24        }
25        Ok(())
26    }
27
28    /// 范围查询：查询指定指标在时间范围内的所有数据点
29    ///
30    /// 等价 SQL：`SELECT * FROM metric WHERE timestamp BETWEEN start AND end`
31    async fn query_range(
32        &self,
33        metric: &str,
34        start: DateTime<Utc>,
35        end: DateTime<Utc>,
36    ) -> Result<Vec<Metric>, TimescaleError>;
37
38    /// 时间桶聚合：按时间窗口聚合数据
39    ///
40    /// 等价 SQL：`SELECT time_bucket('1h', timestamp), AGG(value) FROM metric GROUP BY 1`
41    async fn time_bucket_aggregate(
42        &self,
43        metric: &str,
44        bucket: &str,
45        aggregation: Aggregation,
46        start: DateTime<Utc>,
47        end: DateTime<Utc>,
48    ) -> Result<Vec<TimeBucket>, TimescaleError>;
49
50    /// 创建连续聚合（物化视图，自动刷新）
51    ///
52    /// 等价 SQL：`CREATE MATERIALIZED VIEW ... WITH (timescaledb.continuous) AS ...`
53    async fn create_continuous_aggregate(
54        &self,
55        view_name: &str,
56        query: &str,
57    ) -> Result<(), TimescaleError>;
58
59    /// 降采样：将高频数据聚合为低频数据
60    async fn downsample(&self, config: &DownsampleConfig) -> Result<(), TimescaleError>;
61
62    /// 删除指定指标的所有数据
63    async fn drop_metric(&self, metric: &str) -> Result<(), TimescaleError>;
64}

66/// Provider 类型
67#[derive(Debug, Clone)]
68pub enum TimeseriesProvider {
69    /// 内存实现
70    Memory,
71    /// Stub 实现（生成 SQL）
72    Stub,
73    /// 真实 TimescaleDB（需启用 `real-timescale` feature）
74    #[cfg(feature = "real-timescale")]
75    RealTimescale(RealTimescaleConfig),
76}
77
78/// 真实 TimescaleDB 配置
79#[cfg(feature = "real-timescale")]
80#[derive(Debug, Clone, Default)]
81pub struct RealTimescaleConfig {
82    pub host: String,
83    pub port: u16,
84    pub database: String,
85    pub username: String,
86    pub password: String,
87}

89/// Wrapper enum
90pub enum TimeseriesWrapper {
91    Memory(crate::memory::MemoryTimeseries),
92    Stub(crate::stub::StubTimeseries),
93    #[cfg(feature = "real-timescale")]
94    RealTimescale(crate::real_timescale::RealTimescale),
95}

97#[async_trait]
98impl TimeseriesExt for TimeseriesWrapper {
99    async fn create_hypertable(
100        &self,
101        table: &str,
102        time_column: &str,
103    ) -> Result<(), TimescaleError> {
104        match self {
105            TimeseriesWrapper::Memory(p) => p.create_hypertable(table, time_column).await,
106            TimeseriesWrapper::Stub(p) => p.create_hypertable(table, time_column).await,
107            #[cfg(feature = "real-timescale")]
108            TimeseriesWrapper::RealTimescale(p) => p.create_hypertable(table, time_column).await,
109        }
110    }
111
112    async fn insert_metric(&self, metric: &Metric) -> Result<(), TimescaleError> {
113        match self {
114            TimeseriesWrapper::Memory(p) => p.insert_metric(metric).await,
115            TimeseriesWrapper::Stub(p) => p.insert_metric(metric).await,
116            #[cfg(feature = "real-timescale")]
117            TimeseriesWrapper::RealTimescale(p) => p.insert_metric(metric).await,
118        }
119    }
120
121    async fn query_range(
122        &self,
123        metric: &str,
124        start: DateTime<Utc>,
125        end: DateTime<Utc>,
126    ) -> Result<Vec<Metric>, TimescaleError> {
127        match self {
128            TimeseriesWrapper::Memory(p) => p.query_range(metric, start, end).await,
129            TimeseriesWrapper::Stub(p) => p.query_range(metric, start, end).await,
130            #[cfg(feature = "real-timescale")]
131            TimeseriesWrapper::RealTimescale(p) => p.query_range(metric, start, end).await,
132        }
133    }
134
135    async fn time_bucket_aggregate(
136        &self,
137        metric: &str,
138        bucket: &str,
139        aggregation: Aggregation,
140        start: DateTime<Utc>,
141        end: DateTime<Utc>,
142    ) -> Result<Vec<TimeBucket>, TimescaleError> {
143        match self {
144            TimeseriesWrapper::Memory(p) => {
145                p.time_bucket_aggregate(metric, bucket, aggregation, start, end)
146                    .await
147            }
148            TimeseriesWrapper::Stub(p) => {
149                p.time_bucket_aggregate(metric, bucket, aggregation, start, end)
150                    .await
151            }
152            #[cfg(feature = "real-timescale")]
153            TimeseriesWrapper::RealTimescale(p) => {
154                p.time_bucket_aggregate(metric, bucket, aggregation, start, end)
155                    .await
156            }
157        }
158    }
159
160    async fn create_continuous_aggregate(
161        &self,
162        view_name: &str,
163        query: &str,
164    ) -> Result<(), TimescaleError> {
165        match self {
166            TimeseriesWrapper::Memory(p) => p.create_continuous_aggregate(view_name, query).await,
167            TimeseriesWrapper::Stub(p) => p.create_continuous_aggregate(view_name, query).await,
168            #[cfg(feature = "real-timescale")]
169            TimeseriesWrapper::RealTimescale(p) => {
170                p.create_continuous_aggregate(view_name, query).await
171            }
172        }
173    }
174
175    async fn downsample(&self, config: &DownsampleConfig) -> Result<(), TimescaleError> {
176        match self {
177            TimeseriesWrapper::Memory(p) => p.downsample(config).await,
178            TimeseriesWrapper::Stub(p) => p.downsample(config).await,
179            #[cfg(feature = "real-timescale")]
180            TimeseriesWrapper::RealTimescale(p) => p.downsample(config).await,
181        }
182    }
183
184    async fn drop_metric(&self, metric: &str) -> Result<(), TimescaleError> {
185        match self {
186            TimeseriesWrapper::Memory(p) => p.drop_metric(metric).await,
187            TimeseriesWrapper::Stub(p) => p.drop_metric(metric).await,
188            #[cfg(feature = "real-timescale")]
189            TimeseriesWrapper::RealTimescale(p) => p.drop_metric(metric).await,
190        }
191    }

194/// Builder
195pub struct TimeseriesBuilder {
196    provider: TimeseriesProvider,
197}
198
199impl TimeseriesBuilder {
200    pub fn new(provider: TimeseriesProvider) -> Self {
201        Self { provider }
202    }
203
204    pub fn build(self) -> Result<TimeseriesWrapper, TimescaleError> {
205        match self.provider {
206            TimeseriesProvider::Memory => Ok(TimeseriesWrapper::Memory(
207                crate::memory::MemoryTimeseries::new(),
208            )),
209            TimeseriesProvider::Stub => {
210                Ok(TimeseriesWrapper::Stub(crate::stub::StubTimeseries::new()))
211            }
212            #[cfg(feature = "real-timescale")]
213            TimeseriesProvider::RealTimescale(config) => {
214                let real = crate::real_timescale::RealTimescale::new(config)?;
215                Ok(TimeseriesWrapper::RealTimescale(real))
216            }
217        }
218    }
219}
