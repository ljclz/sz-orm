11/// 内存 TimescaleDB 实现
12pub struct MemoryTimeseries {
13    /// 指标存储：metric_name -> `Vec<Metric>`
14    storage: Mutex<HashMap<String, Vec<Metric>>>,
15    /// hypertable 注册表
16    hypertables: Mutex<Vec<(String, String)>>,
17    /// 连续聚合视图
18    continuous_aggregates: Mutex<Vec<(String, String)>>,
19}
20
21impl MemoryTimeseries {
22    pub fn new() -> Self {
23        Self {
24            storage: Mutex::new(HashMap::new()),
25            hypertables: Mutex::new(Vec::new()),
26            continuous_aggregates: Mutex::new(Vec::new()),
27        }
28    }
29
30    /// 解析时间桶字符串（如 "1h"、"5m"、"1d"）为秒数
31    fn parse_bucket(bucket: &str) -> Result<i64, TimescaleError> {
32        if bucket.is_empty() {
33            return Err(TimescaleError::InvalidConfig(
34                "bucket cannot be empty".to_string(),
35            ));
36        }
37        let (num_str, unit) = bucket.split_at(bucket.len() - 1);
38        let num: i64 = num_str.parse().map_err(|_| {
39            TimescaleError::InvalidConfig(format!("invalid bucket number: {}", num_str))
40        })?;
41        let seconds = match unit {
42            "s" => num,
43            "m" => num * 60,
44            "h" => num * 3600,
45            "d" => num * 86400,
46            _ => {
47                return Err(TimescaleError::InvalidConfig(format!(
48                    "unsupported bucket unit: {}",
49                    unit
50                )))
51            }
52        };
53        Ok(seconds)
54    }
55}

58    fn default() -> Self {

63#[async_trait]
64impl TimeseriesExt for MemoryTimeseries {
65    async fn create_hypertable(
66        &self,
67        table: &str,
68        time_column: &str,
69    ) -> Result<(), TimescaleError> {
70        self.hypertables
71            .lock()
72            .unwrap()
73            .push((table.to_string(), time_column.to_string()));
74        // 确保 storage 中有该 metric 的 entry
75        self.storage
76            .lock()
77            .unwrap()
78            .entry(table.to_string())
79            .or_default();
80        Ok(())
81    }
82
83    async fn insert_metric(&self, metric: &Metric) -> Result<(), TimescaleError> {
84        let mut storage = self.storage.lock().unwrap();
85        storage
86            .entry(metric.name.clone())
87            .or_default()
88            .push(metric.clone());
89        Ok(())
90    }
91
92    async fn query_range(
93        &self,
94        metric: &str,
95        start: DateTime<Utc>,
96        end: DateTime<Utc>,
97    ) -> Result<Vec<Metric>, TimescaleError> {
98        // M-17 修复：校验时间范围（start < end 且跨度 <= MAX_QUERY_RANGE_SECS）
99        crate::validate_time_range(start, end)?;
100        let storage = self.storage.lock().unwrap();
101        let data = storage.get(metric).cloned().unwrap_or_default();
102        Ok(data
103            .into_iter()
104            .filter(|m| m.timestamp >= start && m.timestamp < end)
105            .collect())
106    }
107
108    async fn time_bucket_aggregate(
109        &self,
110        metric: &str,
111        bucket: &str,
112        aggregation: Aggregation,
113        start: DateTime<Utc>,
114        end: DateTime<Utc>,
115    ) -> Result<Vec<TimeBucket>, TimescaleError> {
116        // M-17 修复：校验时间范围（start < end 且跨度 <= MAX_QUERY_RANGE_SECS）
117        crate::validate_time_range(start, end)?;
118        let bucket_secs = Self::parse_bucket(bucket)?;
119        let bucket_duration = chrono::Duration::seconds(bucket_secs);
120
121        // 获取范围内的数据
122        let data = self.query_range(metric, start, end).await?;
123
124        // 按桶分组
125        let mut buckets: HashMap<DateTime<Utc>, Vec<f64>> = HashMap::new();
126        for m in data {
127            let elapsed = (m.timestamp - start).num_seconds();
128            let bucket_idx = elapsed / bucket_secs;
129            let bucket_start = start + chrono::Duration::seconds(bucket_idx * bucket_secs);
130            buckets.entry(bucket_start).or_default().push(m.value);
131        }
132
133        // 生成连续的桶（即使没有数据）
134        let mut result = Vec::new();
135        let mut current = start;
136        while current < end {
137            let values = buckets.remove(&current).unwrap_or_default();
138            let mut bucket = TimeBucket::from_values(current, &values);
139            // 如果桶内无数据，按聚合类型填充 0 或跳过
140            if values.is_empty() {
141                // 空桶：count=0，其他字段保持 0.0
142                bucket = TimeBucket {
143                    bucket_start: current,
144                    count: 0,
145                    sum: 0.0,
146                    min: 0.0,
147                    max: 0.0,
148                    avg: 0.0,
149                };
150            }
151            // 应用聚合（用于验证聚合函数工作）
152            let _agg_value = aggregation.apply(&values);
153            result.push(bucket);
154            current += bucket_duration;
155        }
156
157        Ok(result)
158    }
159
160    async fn create_continuous_aggregate(
161        &self,
162        view_name: &str,
163        query: &str,
164    ) -> Result<(), TimescaleError> {
165        self.continuous_aggregates
166            .lock()
167            .unwrap()
168            .push((view_name.to_string(), query.to_string()));
169        Ok(())
170    }
171
172    async fn downsample(&self, config: &DownsampleConfig) -> Result<(), TimescaleError> {
173        // 内存实现：从源指标读取所有数据，按 interval 聚合，写入目标指标
174        let storage = self.storage.lock().unwrap();
175        let source_data = storage
176            .get(&config.source_metric)
177            .cloned()
178            .unwrap_or_default();
179        drop(storage);
180
181        if source_data.is_empty() {
182            return Ok(());
183        }
184
185        let bucket_secs = Self::parse_bucket(&config.interval)?;
186        let start = source_data
187            .iter()
188            .map(|m| m.timestamp)
189            .min()
190            .unwrap_or_else(Utc::now);
191
192        // 按桶分组并聚合
193        let mut buckets: HashMap<i64, Vec<f64>> = HashMap::new();
194        for m in &source_data {
195            let elapsed = (m.timestamp - start).num_seconds();
196            let bucket_idx = elapsed / bucket_secs;
197            buckets.entry(bucket_idx).or_default().push(m.value);
198        }
199
200        // 生成降采样后的指标
201        let mut downsampled = Vec::new();
202        for (idx, values) in buckets {
203            let bucket_start = start + chrono::Duration::seconds(idx * bucket_secs);
204            let agg_value = config.aggregation.apply(&values);
205            downsampled.push(Metric::new(
206                config.target_metric.clone(),
207                bucket_start,
208                agg_value,
209            ));
210        }
211
212        // 写入目标指标
213        let mut storage = self.storage.lock().unwrap();
214        let target = storage.entry(config.target_metric.clone()).or_default();
215        target.extend(downsampled);
216
217        Ok(())
218    }
219
220    async fn drop_metric(&self, metric: &str) -> Result<(), TimescaleError> {
221        let mut storage = self.storage.lock().unwrap();
222        if storage.remove(metric).is_none() {
223            return Err(TimescaleError::NotFound(metric.to_string()));
224        }
225        Ok(())
226    }
227}

328        let start = Utc.with_ymd_and_hms(2026, 1, 1, 0, 0, 0).unwrap();
329        // start == end：拒绝
330        assert!(crate::validate_time_range(start, start).is_err());

333        assert!(crate::validate_time_range(start, end).is_err());
334        // 正常范围：允许
335        let end = start + chrono::Duration::days(30);
336        assert!(crate::validate_time_range(start, end).is_ok());
337        // 366 天：允许
338        let end = start + chrono::Duration::seconds(crate::MAX_QUERY_RANGE_SECS);
339        assert!(crate::validate_time_range(start, end).is_ok());
340        // 366 天 + 1 秒：拒绝
341        let end = start + chrono::Duration::seconds(crate::MAX_QUERY_RANGE_SECS + 1);
342        assert!(crate::validate_time_range(start, end).is_err());

346    async fn test_time_bucket_aggregate() {
347        let ts = MemoryTimeseries::new();
348        // 每 10 分钟一个数据点，4 个点
349        for i in 0..4 {
350            ts.insert_metric(&make_metric("cpu", i * 10, 0.1 * (i + 1) as f64))
351                .await
352                .unwrap();
353        }
354
355        let start = Utc.with_ymd_and_hms(2026, 7, 20, 0, 0, 0).unwrap();
356        let end = Utc.with_ymd_and_hms(2026, 7, 20, 1, 0, 0).unwrap();
357        let buckets = ts
358            .time_bucket_aggregate("cpu", "30m", Aggregation::Avg, start, end)
359            .await
360            .unwrap();
361        // 1 小时按 30 分钟分桶 = 2 个桶
362        assert_eq!(buckets.len(), 2);
363        // 第一个桶（0-30min）：包含 0, 10, 20 分钟的数据
364        assert_eq!(buckets[0].count, 3);
365        // (0.1+0.2+0.3)/3 = 0.2
366        assert!((buckets[0].avg - 0.2).abs() < 1e-10);
367        // 第二个桶（30-60min）：包含 30 分钟的数据
368        assert_eq!(buckets[1].count, 1);
369        assert!((buckets[1].avg - 0.4).abs() < 1e-10);
370    }
