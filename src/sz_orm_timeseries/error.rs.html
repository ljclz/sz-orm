1//! TimescaleDB 错误类型
2
3use std::fmt;
4
5/// TimescaleDB 操作错误
6#[derive(Debug)]
7pub enum TimescaleError {
8    /// 指标未找到
9    NotFound(String),
10    /// 时间范围无效（start >= end）
11    InvalidTimeRange { start: String, end: String },
12    /// 聚合类型不支持
13    UnsupportedAggregation(String),
14    /// SQL 执行错误
15    Query(String),
16    /// 连接错误
17    Connection(String),
18    /// 配置错误
19    InvalidConfig(String),
20}
21
22impl fmt::Display for TimescaleError {
23    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
24        match self {
25            TimescaleError::NotFound(msg) => write!(f, "not found: {}", msg),
26            TimescaleError::InvalidTimeRange { start, end } => {
27                write!(f, "invalid time range: start {} >= end {}", start, end)
28            }
29            TimescaleError::UnsupportedAggregation(msg) => {
30                write!(f, "unsupported aggregation: {}", msg)
31            }
32            TimescaleError::Query(msg) => write!(f, "query error: {}", msg),
33            TimescaleError::Connection(msg) => write!(f, "connection error: {}", msg),
34            TimescaleError::InvalidConfig(msg) => write!(f, "invalid config: {}", msg),
35        }
36    }
37}
38
39impl std::error::Error for TimescaleError {}
40
41impl From<std::io::Error> for TimescaleError {
42    fn from(err: std::io::Error) -> Self {
43        TimescaleError::Query(err.to_string())
44    }
45}
46
47impl From<chrono::ParseError> for TimescaleError {
48    fn from(err: chrono::ParseError) -> Self {
49        TimescaleError::InvalidConfig(format!("time parse error: {}", err))
50    }
51}
