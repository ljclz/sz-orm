1//! SQL 安全工具：标识符与时间桶校验
2//!
3//! v0.2.2 引入：为 `stub` / `memory` / `real_timescale` 提供统一的 SQL 注入防护原语。
4//! 所有需要拼接 SQL 标识符（表名/列名/视图名/指标名）的位置必须先经
5//! `validate_identifier` 校验；时间桶（time_bucket）参数必须经
6//! `validate_time_bucket` 校验。
7

10/// 校验 SQL 标识符（表名/列名/视图名/指标名）
11///
12/// 仅允许 ASCII 字母数字 + 下划线，不以数字开头，长度 1-63（PostgreSQL 限制）。
13/// 拒绝任何 SQL 元字符（引号、分号、空格、注释、引号转义等），杜绝 SQL 注入。
14pub fn validate_identifier(name: &str, kind: &str) -> Result<(), TimescaleError> {
15    if name.is_empty() || name.len() > 63 {
16        return Err(TimescaleError::InvalidConfig(format!(
17            "invalid {}: empty or too long (max 63 chars): {:?}",
18            kind, name
19        )));
20    }
21    let mut chars = name.chars();
22    let first = chars.next().expect("non-empty checked above");
23    if !first.is_ascii_alphabetic() && first != '_' {
24        return Err(TimescaleError::InvalidConfig(format!(
25            "invalid {}: must start with ASCII letter or underscore, got {:?}",
26            kind, name
27        )));
28    }
29    if !chars.all(|c| c.is_ascii_alphanumeric() || c == '_') {
30        return Err(TimescaleError::InvalidConfig(format!(
31            "invalid {}: only ASCII alphanumeric and underscore allowed, got {:?}",
32            kind, name
33        )));
34    }
35    Ok(())
36}

38/// 校验 time_bucket 参数（如 "5m" / "1h" / "1d"）
39///
40/// 仅允许 `<正整数><单位>` 格式，单位仅支持 TimescaleDB 标准单位：
41/// s/m/h/d/w。拒绝任何其他字符，防止 SQL 注入。
42pub fn validate_time_bucket(bucket: &str) -> Result<(), TimescaleError> {
43    if bucket.is_empty() || bucket.len() > 16 {
44        return Err(TimescaleError::InvalidConfig(format!(
45            "invalid time_bucket: empty or too long (max 16 chars): {:?}",
46            bucket
47        )));
48    }
49    let mut chars = bucket.chars();
50    // 必须以数字开头
51    let first = chars.next().expect("non-empty checked above");
52    if !first.is_ascii_digit() {
53        return Err(TimescaleError::InvalidConfig(format!(
54            "invalid time_bucket: must start with digit, got {:?}",
55            bucket
56        )));
57    }
58    // 中间必须是数字
59    let digits_end = bucket
60        .find(|c: char| !c.is_ascii_digit())
61        .unwrap_or(bucket.len());
62    let _digits = &bucket[..digits_end];
63    // 剩余部分必须是单个合法单位字符
64    let unit = &bucket[digits_end..];
65    const ALLOWED_UNITS: &[&str] = &["s", "m", "h", "d", "w"];
66    if !ALLOWED_UNITS.contains(&unit) {
67        return Err(TimescaleError::InvalidConfig(format!(
68            "invalid time_bucket: unit {:?} not allowed, allowed: {:?} (got {:?})",
69            unit, ALLOWED_UNITS, bucket
70        )));
71    }
72    Ok(())
73}

75/// 校验 `CREATE MATERIALIZED VIEW ... AS <query>` 中的 query 参数
76///
77/// v0.2.2 修复 P0-9（第二次审查发现）：原实现直接拼接 query 到 SQL，存在 SQL 注入风险。
78/// 本函数对 query 做严格白名单式校验：
79///
80/// 1. 长度限制：1..=4096 字节（足够覆盖连续聚合的 SELECT 语句）
81/// 2. 必须以 `SELECT` 关键字开头（大小写不敏感，前后允许空白）
82/// 3. 禁止分号 `;`（防止多语句注入）
83/// 4. 禁止行注释 `--` 和块注释 `/*` `*/`
84/// 5. 禁止危险 DDL/DML 关键字（DROP/DELETE/UPDATE/INSERT/ALTER/TRUNCATE/CREATE/GRANT/REVOKE/EXEC/MERGE）
85///    —— 在 word boundary 处匹配，避免误伤合法列名（如 `updated_at`）
86///
87/// 注意：本函数是**防御性校验**，不替代完整 SQL 解析器。对于复杂的连续聚合 query
88/// （含子查询、JOIN、CTE 等），建议调用方在应用层使用 `sqlparser` crate 做 AST 级校验。
89pub fn validate_continuous_aggregate_query(query: &str) -> Result<(), TimescaleError> {
90    if query.is_empty() {
91        return Err(TimescaleError::InvalidConfig(
92            "invalid continuous aggregate query: empty".to_string(),
93        ));
94    }
95    if query.len() > 4096 {
96        return Err(TimescaleError::InvalidConfig(format!(
97            "invalid continuous aggregate query: too long (max 4096 chars, got {})",
98            query.len()
99        )));
100    }
101    let trimmed = query.trim_start();
102    let upper_trimmed = trimmed.to_ascii_uppercase();
103    // 必须以 SELECT 或 WITH（CTE）开头（大小写不敏感）
104    // SELECT 后必须跟空白或 (；WITH 后必须跟空白
105    let starts_with_select = upper_trimmed.starts_with("SELECT")
106        && trimmed
107            .as_bytes()
108            .get(6)
109            .map(|&b| b == b' ' || b == b'\t' || b == b'\n' || b == b'\r' || b == b'(')
110            .unwrap_or(true);
111    let starts_with_with = upper_trimmed.starts_with("WITH")
112        && trimmed
113            .as_bytes()
114            .get(4)
115            .map(|&b| b == b' ' || b == b'\t' || b == b'\n' || b == b'\r')
116            .unwrap_or(true);
117    if !starts_with_select && !starts_with_with {
118        return Err(TimescaleError::InvalidConfig(format!(
119            "invalid continuous aggregate query: must start with SELECT or WITH keyword, got {:?}",
120            &query[..query.len().min(40)]
121        )));
122    }
123    // 禁止分号（多语句注入防护）
124    if query.contains(';') {
125        return Err(TimescaleError::InvalidConfig(
126            "invalid continuous aggregate query: semicolon ';' is forbidden (multi-statement injection prevention)".to_string(),
127        ));
128    }
129    // 禁止行注释和块注释
130    if query.contains("--") || query.contains("/*") || query.contains("*/") {
131        return Err(TimescaleError::InvalidConfig(
132            "invalid continuous aggregate query: SQL comments (--, /*, */) are forbidden"
133                .to_string(),
134        ));
135    }
136    // 禁止危险 DDL/DML 关键字（word boundary 匹配，大小写不敏感）
137    let upper = query.to_ascii_uppercase();
138    const FORBIDDEN_KEYWORDS: &[&str] = &[
139        "DROP", "DELETE", "UPDATE", "INSERT", "ALTER", "TRUNCATE", "CREATE", "GRANT", "REVOKE",
140        "EXEC", "MERGE", "CALL", "VACUUM", "REINDEX", "CLUSTER", "ATTACH", "DETACH",
141    ];
142    for kw in FORBIDDEN_KEYWORDS {
143        if contains_keyword_word_boundary(&upper, kw) {
144            return Err(TimescaleError::InvalidConfig(format!(
145                "invalid continuous aggregate query: forbidden SQL keyword {:?} detected (DDL/DML in continuous aggregate query is not allowed)",
146                kw
147            )));
148        }
149    }
150    Ok(())
151}

153/// 在 `haystack`（已 upper-case）中检查是否包含 `keyword`（已 upper-case）作为独立单词
154///
155/// 使用 word boundary 判定：关键字前后字符必须是非字母数字下划线（或字符串边界）。
156/// 这样 `updated_at` 不会被误判为含 `UPDATE`，`inserted_at` 不会被误判为含 `INSERT`。
157fn contains_keyword_word_boundary(haystack: &str, keyword: &str) -> bool {
158    let bytes = haystack.as_bytes();
159    let kw_bytes = keyword.as_bytes();
160    if kw_bytes.is_empty() || bytes.len() < kw_bytes.len() {
161        return false;
162    }
163    let mut i = 0usize;
164    while i + kw_bytes.len() <= bytes.len() {
165        if &bytes[i..i + kw_bytes.len()] == kw_bytes {
166            let before_ok = i == 0 || {
167                let b = bytes[i - 1];
168                !(b.is_ascii_alphanumeric() || b == b'_')
169            };
170            let after_pos = i + kw_bytes.len();
171            let after_ok = after_pos >= bytes.len() || {
172                let b = bytes[after_pos];
173                !(b.is_ascii_alphanumeric() || b == b'_')
174            };
175            if before_ok && after_ok {
176                return true;
177            }
178        }
179        i += 1;
180    }
181    false
182}
