1//! TimescaleDB 深度扩展功能
2//!
3//! 本模块补充 TimescaleDB 扩展缺失的核心深度功能，包括：
4//!
5//! - **连续聚合管理**：物化视图刷新策略、刷新窗口、策略元数据
6//! - **数据压缩策略**：启用/禁用压缩、压缩配置（segmentby/orderby）、压缩统计
7//! - **数据保留策略**：自动数据清理策略、保留时长、策略注册表
8//! - **时间桶对齐与 gapfill**：time_bucket 对齐到自然边界、locf/interpolate 缺失桶填充
9//!
10//! # 设计说明
11//!
12//! 本模块以**独立结构 + 扩展 trait** 的方式提供，不修改既有 `TimeseriesExt` trait，
13//! 避免破坏已有的 memory / stub / real_timescale 三种实现。
14//! 内存实现部分基于纯 Rust 实现，不依赖外部库。
15

27/// 连续聚合刷新策略
28///
29/// TimescaleDB 的连续聚合视图支持自动刷新，刷新策略决定刷新频率和窗口。
30#[derive(Debug, Clone, Copy, PartialEq, Eq)]
31pub enum RefreshPolicy {
32    /// 永不自动刷新（手动调用 refresh）
33    Manual,
34    /// 按固定间隔自动刷新
35    Scheduled,
36    /// 仅在数据写入时触发刷新（实时聚合）
37    RealTime,
38}
39
40impl RefreshPolicy {
41    /// 转为 TimescaleDB SQL 参数片段
42    pub fn as_sql_config(&self) -> &'static str {
43        match self {
44            RefreshPolicy::Manual => "timescaledb.continuous = true",
45            RefreshPolicy::Scheduled => "timescaledb.continuous = true",
46            RefreshPolicy::RealTime => "timescaledb.continuous = true, timescaledb.materialized_only = false",
47        }
48    }
49}
50
51/// 连续聚合定义
52///
53/// 对应 TimescaleDB 的 `CREATE MATERIALIZED VIEW ... WITH (timescaledb.continuous)` 语句。
54#[derive(Debug, Clone)]
55pub struct ContinuousAggregateDef {
56    /// 视图名称
57    pub view_name: String,
58    /// 源 hypertable 名
59    pub source_table: String,
60    /// 时间桶宽度（如 "1h"、"1d"）
61    pub bucket_interval: String,
62    /// 聚合 SQL 表达式（如 "AVG(value)"、"SUM(value)"）
63    pub aggregate_expr: String,
64    /// 刷新策略
65    pub refresh_policy: RefreshPolicy,
66    /// 刷新窗口起点偏移（秒，负数表示过去）
67    /// 例如 -86400 表示从 1 天前开始刷新
68    pub refresh_start_offset: i64,
69    /// 刷新窗口终点偏移（秒，负数表示过去）
70    /// 例如 -3600 表示刷新到 1 小时前为止（避免实时数据冲突）
71    pub refresh_end_offset: i64,
72    /// 刷新间隔（秒）
73    pub refresh_interval: i64,
74}
75
76impl ContinuousAggregateDef {
77    /// 创建一个新的连续聚合定义
78    pub fn new(
79        view_name: impl Into<String>,
80        source_table: impl Into<String>,
81        bucket_interval: impl Into<String>,
82        aggregate_expr: impl Into<String>,
83    ) -> Self {
84        Self {
85            view_name: view_name.into(),
86            source_table: source_table.into(),
87            bucket_interval: bucket_interval.into(),
88            aggregate_expr: aggregate_expr.into(),
89            refresh_policy: RefreshPolicy::Scheduled,
90            refresh_start_offset: -86400,     // 默认从 1 天前开始
91            refresh_end_offset: -3600,         // 默认到 1 小时前为止
92            refresh_interval: 3600,            // 默认每小时刷新一次
93        }
94    }
95
96    /// 设置刷新策略
97    pub fn with_refresh_policy(mut self, policy: RefreshPolicy) -> Self {
98        self.refresh_policy = policy;
99        self
100    }
101
102    /// 设置刷新窗口偏移（秒）
103    pub fn with_refresh_window(mut self, start_offset: i64, end_offset: i64) -> Self {
104        self.refresh_start_offset = start_offset;
105        self.refresh_end_offset = end_offset;
106        self
107    }
108
109    /// 设置刷新间隔（秒）
110    pub fn with_refresh_interval(mut self, interval_secs: i64) -> Self {
111        self.refresh_interval = interval_secs;
112        self
113    }
114
115    /// 生成 CREATE MATERIALIZED VIEW SQL
116    pub fn to_create_sql(&self) -> String {
117        format!(
118            "CREATE MATERIALIZED VIEW {} WITH ({}) AS \
119             SELECT time_bucket('{}', time), {} FROM {} \
120             GROUP BY 1 WITH NO DATA",
121            self.view_name,
122            self.refresh_policy.as_sql_config(),
123            self.bucket_interval,
124            self.aggregate_expr,
125            self.source_table
126        )
127    }
128
129    /// 生成 REFRESH 语句的 SQL（刷新窗口由 start/end offset 决定）
130    pub fn to_refresh_sql(&self) -> String {
131        format!(
132            "CALL refresh_continuous_aggregate('{}', now() + interval '{} seconds', now() + interval '{} seconds')",
133            self.view_name, self.refresh_start_offset, self.refresh_end_offset
134        )
135    }
136
137    /// 生成 DROP MATERIALIZED VIEW SQL
138    pub fn to_drop_sql(&self) -> String {
139        format!("DROP MATERIALIZED VIEW IF EXISTS {}", self.view_name)
140    }
141}

143/// 连续聚合注册表（内存版，用于跟踪已创建的连续聚合）
144#[derive(Debug, Clone, Default)]
145pub struct ContinuousAggregateRegistry {
146    /// 已注册的连续聚合：view_name -> def
147    aggregates: HashMap<String, ContinuousAggregateDef>,
148    /// 每个视图的刷新历史：view_name -> 最后刷新时间
149    last_refreshed: HashMap<String, DateTime<Utc>>,
150}
151
152impl ContinuousAggregateRegistry {
153    /// 创建空注册表
154    pub fn new() -> Self {
155        Self::default()
156    }
157
158    /// 注册一个连续聚合
159    pub fn register(&mut self, def: ContinuousAggregateDef) -> Result<(), TimescaleError> {
160        if self.aggregates.contains_key(&def.view_name) {
161            return Err(TimescaleError::Query(format!(
162                "continuous aggregate already exists: {}",
163                def.view_name
164            )));
165        }
166        self.aggregates.insert(def.view_name.clone(), def);
167        Ok(())
168    }
169
170    /// 注销一个连续聚合
171    pub fn unregister(&mut self, view_name: &str) -> Result<ContinuousAggregateDef, TimescaleError> {
172        self.aggregates
173            .remove(view_name)
174            .ok_or_else(|| TimescaleError::NotFound(format!("continuous aggregate: {}", view_name)))
175    }
176
177    /// 记录刷新时间
178    pub fn mark_refreshed(&mut self, view_name: &str, when: DateTime<Utc>) {
179        self.last_refreshed.insert(view_name.to_string(), when);
180    }
181
182    /// 获取最后刷新时间
183    pub fn last_refresh_time(&self, view_name: &str) -> Option<DateTime<Utc>> {
184        self.last_refreshed.get(view_name).copied()
185    }
186
187    /// 检查是否需要刷新（根据刷新间隔判断）
188    pub fn needs_refresh(&self, view_name: &str, now: DateTime<Utc>) -> bool {
189        match self.aggregates.get(view_name) {
190            Some(def) => match self.last_refreshed.get(view_name) {
191                Some(last) => (now - *last).num_seconds() >= def.refresh_interval,
192                None => true, // 从未刷新过
193            },
194            None => false, // 不存在的视图不需要刷新
195        }
196    }
197
198    /// 列出所有需要刷新的视图
199    pub fn list_needs_refresh(&self, now: DateTime<Utc>) -> Vec<String> {
200        self.aggregates
201            .keys()
202            .filter(|name| self.needs_refresh(name, now))
203            .cloned()
204            .collect()
205    }
206
207    /// 获取所有已注册的连续聚合
208    pub fn list_all(&self) -> Vec<&ContinuousAggregateDef> {
209        self.aggregates.values().collect()
210    }
211
212    /// 生成所有需要刷新视图的 REFRESH SQL
213    pub fn refresh_all_sql(&self, now: DateTime<Utc>) -> Vec<String> {
214        self.list_needs_refresh(now)
215            .iter()
216            .filter_map(|name| self.aggregates.get(name).map(|def| def.to_refresh_sql()))
217            .collect()
218    }
219}

225/// 压缩状态
226#[derive(Debug, Clone, Copy, PartialEq, Eq)]
227pub enum CompressionStatus {
228    /// 压缩已启用
229    Enabled,
230    /// 压缩已禁用
231    Disabled,
232}
233
234/// 数据压缩配置
235///
236/// 对应 TimescaleDB 的 `ALTER TABLE ... SET (timescaledb.compress)` 语句。
237#[derive(Debug, Clone)]
238pub struct CompressionConfig {
239    /// 目标 hypertable 名
240    pub table: String,
241    /// 按 segmentby 列分区压缩（通常为 tag/label 列）
242    pub segmentby: String,
243    /// 按 orderby 列排序压缩（通常为时间列）
244    pub orderby: String,
245    /// 每个压缩块包含的数据行数（默认 1000）
246    pub chunk_time_interval: i64,
247}
248
249impl CompressionConfig {
250    /// 创建压缩配置
251    pub fn new(
252        table: impl Into<String>,
253        segmentby: impl Into<String>,
254        orderby: impl Into<String>,
255    ) -> Self {
256        Self {
257            table: table.into(),
258            segmentby: segmentby.into(),
259            orderby: orderby.into(),
260            chunk_time_interval: 86400, // 默认 1 天一个 chunk
261        }
262    }
263
264    /// 设置 chunk 时间间隔（秒）
265    pub fn with_chunk_interval(mut self, secs: i64) -> Self {
266        self.chunk_time_interval = secs;
267        self
268    }
269
270    /// 生成启用压缩的 SQL
271    pub fn to_enable_sql(&self) -> String {
272        format!(
273            "ALTER TABLE {} SET (timescaledb.compress, \
274             timescaledb.compress_segmentby = '{}', \
275             timescaledb.compress_orderby = '{}')",
276            self.table, self.segmentby, self.orderby
277        )
278    }
279
280    /// 生成禁用压缩的 SQL
281    pub fn to_disable_sql(&self) -> String {
282        format!(
283            "ALTER TABLE {} SET (timescaledb.compress = false)",
284            self.table
285        )
286    }
287
288    /// 生成压缩指定 chunk 的 SQL
289    pub fn to_compress_chunk_sql(&self, chunk_name: &str) -> String {
290        format!("SELECT compress_chunk('{}')", chunk_name)
291    }
292
293    /// 生成解压指定 chunk 的 SQL
294    pub fn to_decompress_chunk_sql(&self, chunk_name: &str) -> String {
295        format!("SELECT decompress_chunk('{}')", chunk_name)
296    }

299/// 压缩策略注册表（内存版，用于跟踪已启用压缩的表及其统计）
300#[derive(Debug, Clone, Default)]
301pub struct CompressionPolicyRegistry {
302    /// 已启用压缩的表：table -> (config, status)
303    configs: HashMap<String, (CompressionConfig, CompressionStatus)>,
304    /// 每张表的压缩统计：table -> CompressionStats
305    stats: HashMap<String, CompressionStats>,
306}
307
308/// 压缩统计信息
309#[derive(Debug, Clone, Default)]
310pub struct CompressionStats {
311    /// 已压缩的 chunk 数
312    pub compressed_chunks: u64,
313    /// 未压缩的 chunk 数
314    pub uncompressed_chunks: u64,
315    /// 压缩前总字节数
316    pub before_bytes: u64,
317    /// 压缩后总字节数
318    pub after_bytes: u64,
319}
320
321impl CompressionStats {
322    /// 创建空统计
323    pub fn new() -> Self {
324        Self::default()
325    }
326
327    /// 计算压缩比（after / before，越小越好）
328    pub fn ratio(&self) -> f64 {
329        if self.before_bytes == 0 {
330            return 1.0;
331        }
332        self.after_bytes as f64 / self.before_bytes as f64
333    }
334
335    /// 计算节省的空间百分比
336    pub fn space_saved_percent(&self) -> f64 {
337        if self.before_bytes == 0 {
338            return 0.0;
339        }
340        (1.0 - self.ratio()) * 100.0
341    }
342
343    /// 记录一次压缩操作
344    pub fn record_compression(&mut self, before: u64, after: u64) {
345        self.compressed_chunks += 1;
346        self.uncompressed_chunks = self.uncompressed_chunks.saturating_sub(1);
347        self.before_bytes += before;
348        self.after_bytes += after;
349    }
350
351    /// 记录一次解压操作
352    pub fn record_decompression(&mut self) {
353        self.uncompressed_chunks += 1;
354        self.compressed_chunks = self.compressed_chunks.saturating_sub(1);
355    }
356}
357
358impl CompressionPolicyRegistry {
359    /// 创建空注册表
360    pub fn new() -> Self {
361        Self::default()
362    }
363
364    /// 启用压缩策略
365    pub fn enable(&mut self, config: CompressionConfig) -> Result<(), TimescaleError> {
366        let table = config.table.clone();
367        if let Some((_, status)) = self.configs.get(&table) {
368            if *status == CompressionStatus::Enabled {
369                return Err(TimescaleError::Query(format!(
370                    "compression already enabled on table: {}",
371                    table
372                )));
373            }
374        }
375        self.configs
376            .insert(table.clone(), (config, CompressionStatus::Enabled));
377        self.stats.entry(table).or_default();
378        Ok(())
379    }
380
381    /// 禁用压缩策略
382    pub fn disable(&mut self, table: &str) -> Result<CompressionConfig, TimescaleError> {
383        match self.configs.get_mut(table) {
384            Some((config, status)) => {
385                if *status == CompressionStatus::Disabled {
386                    return Err(TimescaleError::Query(format!(
387                        "compression already disabled on table: {}",
388                        table
389                    )));
390                }
391                *status = CompressionStatus::Disabled;
392                Ok(config.clone())
393            }
394            None => Err(TimescaleError::NotFound(format!(
395                "compression config for table: {}",
396                table
397            ))),
398        }
399    }
400
401    /// 查询压缩状态
402    pub fn status(&self, table: &str) -> Option<CompressionStatus> {
403        self.configs.get(table).map(|(_, s)| *s)
404    }
405
406    /// 获取压缩配置
407    pub fn config(&self, table: &str) -> Option<&CompressionConfig> {
408        self.configs.get(table).map(|(c, _)| c)
409    }
410
411    /// 更新压缩统计
412    pub fn update_stats(&mut self, table: &str, stats: CompressionStats) {
413        self.stats.insert(table.to_string(), stats);
414    }
415
416    /// 获取压缩统计
417    pub fn stats(&self, table: &str) -> Option<&CompressionStats> {
418        self.stats.get(table)
419    }
420
421    /// 生成所有已启用表的压缩 SQL
422    pub fn enable_all_sql(&self) -> Vec<String> {
423        self.configs
424            .iter()
425            .filter(|(_, (_, status))| *status == CompressionStatus::Enabled)
426            .map(|(_, (config, _))| {
427                // 仅对配置存在但尚未实际启用的情况生成 SQL
428                config.to_enable_sql()
429            })
430            .collect()
431    }
432}

438/// 数据保留策略
439///
440/// 对应 TimescaleDB 的 `add_retention_policy` 函数，自动删除超过保留时长的数据。
441#[derive(Debug, Clone)]
442pub struct RetentionPolicy {
443    /// 目标 hypertable 名
444    pub table: String,
445    /// 保留时长（秒），超过此时长的数据将被删除
446    /// 例如 30 天 = 30 * 86400 = 2,592,000 秒
447    pub retention_secs: i64,
448    /// 策略调度间隔（秒），默认 1 天
449    pub schedule_interval: i64,
450    /// 是否启用
451    pub enabled: bool,
452}
453
454impl RetentionPolicy {
455    /// 创建保留策略
456    pub fn new(table: impl Into<String>, retention_secs: i64) -> Self {
457        Self {
458            table: table.into(),
459            retention_secs,
460            schedule_interval: 86400, // 默认每天检查一次
461            enabled: true,
462        }
463    }
464
465    /// 创建以天为单位的保留策略
466    pub fn with_days(table: impl Into<String>, days: i64) -> Self {
467        Self::new(table, days * 86400)
468    }
469
470    /// 设置调度间隔（秒）
471    pub fn with_schedule_interval(mut self, secs: i64) -> Self {
472        self.schedule_interval = secs;
473        self
474    }
475
476    /// 禁用策略
477    pub fn disable(mut self) -> Self {
478        self.enabled = false;
479        self
480    }
481
482    /// 生成 add_retention_policy SQL
483    pub fn to_add_sql(&self) -> String {
484        format!(
485            "SELECT add_retention_policy('{}', INTERVAL '{} seconds')",
486            self.table, self.retention_secs
487        )
488    }
489
490    /// 生成 remove_retention_policy SQL
491    pub fn to_remove_sql(&self) -> String {
492        format!("SELECT remove_retention_policy('{}')", self.table)
493    }
494
495    /// 计算截止时间点：早于此时间的数据应被删除
496    pub fn cutoff_time(&self, now: DateTime<Utc>) -> DateTime<Utc> {
497        now - Duration::seconds(self.retention_secs)
498    }
499}

501/// 保留策略注册表
502#[derive(Debug, Clone, Default)]
503pub struct RetentionPolicyRegistry {
504    /// 已注册的保留策略：table -> policy
505    policies: HashMap<String, RetentionPolicy>,
506    /// 每张表的删除统计：table -> (删除次数, 删除行数)
507    purge_stats: HashMap<String, (u64, u64)>,
508}
509
510impl RetentionPolicyRegistry {
511    /// 创建空注册表
512    pub fn new() -> Self {
513        Self::default()
514    }
515
516    /// 注册保留策略
517    pub fn register(&mut self, policy: RetentionPolicy) -> Result<(), TimescaleError> {
518        if self.policies.contains_key(&policy.table) {
519            return Err(TimescaleError::Query(format!(
520                "retention policy already exists for table: {}",
521                policy.table
522            )));
523        }
524        let table = policy.table.clone();
525        self.policies.insert(table.clone(), policy);
526        self.purge_stats.insert(table, (0, 0));
527        Ok(())
528    }
529
530    /// 注销保留策略
531    pub fn unregister(&mut self, table: &str) -> Result<RetentionPolicy, TimescaleError> {
532        self.purge_stats.remove(table);
533        self.policies
534            .remove(table)
535            .ok_or_else(|| TimescaleError::NotFound(format!("retention policy: {}", table)))
536    }
537
538    /// 获取策略
539    pub fn get(&self, table: &str) -> Option<&RetentionPolicy> {
540        self.policies.get(table)
541    }
542
543    /// 列出所有已启用的策略
544    pub fn list_enabled(&self) -> Vec<&RetentionPolicy> {
545        self.policies.values().filter(|p| p.enabled).collect()
546    }
547
548    /// 列出所有策略
549    pub fn list_all(&self) -> Vec<&RetentionPolicy> {
550        self.policies.values().collect()
551    }
552
553    /// 记录一次清理操作
554    pub fn record_purge(&mut self, table: &str, rows_deleted: u64) {
555        if let Some(stats) = self.purge_stats.get_mut(table) {
556            stats.0 += 1;
557            stats.1 += rows_deleted;
558        }
559    }
560
561    /// 获取清理统计
562    pub fn purge_stats(&self, table: &str) -> Option<(u64, u64)> {
563        self.purge_stats.get(table).copied()
564    }
565
566    /// 生成所有已启用策略的 add_retention_policy SQL
567    pub fn add_all_sql(&self) -> Vec<String> {
568        self.list_enabled()
569            .iter()
570            .map(|p| p.to_add_sql())
571            .collect()
572    }
573
574    /// 计算所有已启用策略的截止时间
575    pub fn cutoff_times(&self, now: DateTime<Utc>) -> Vec<(String, DateTime<Utc>)> {
576        self.list_enabled()
577            .iter()
578            .map(|p| (p.table.clone(), p.cutoff_time(now)))
579            .collect()
580    }
581}

587/// 时间桶对齐工具
588///
589/// TimescaleDB 的 `time_bucket` 函数会将时间戳对齐到桶的自然边界。
590/// 例如 `time_bucket('1h', ts)` 会将时间对齐到整点。
591pub struct TimeBucketAligner;
592
593impl TimeBucketAligner {
594    /// 将时间戳对齐到桶的自然边界
595    ///
596    /// - `bucket_secs`：桶宽度（秒）
597    /// - `epoch`：对齐基准时间（通常是 Unix epoch 或某个起始时间）
598    pub fn align(timestamp: DateTime<Utc>, bucket_secs: i64, epoch: DateTime<Utc>) -> DateTime<Utc> {
599        let elapsed = (timestamp - epoch).num_seconds();
600        let bucket_idx = elapsed.div_euclid(bucket_secs);
601        epoch + Duration::seconds(bucket_idx * bucket_secs)
602    }
603
604    /// 对齐到 Unix epoch（1970-01-01 00:00:00 UTC）
605    pub fn align_to_epoch(timestamp: DateTime<Utc>, bucket_secs: i64) -> DateTime<Utc> {
606        let epoch = DateTime::<Utc>::from_timestamp(0, 0).unwrap();
607        Self::align(timestamp, bucket_secs, epoch)
608    }
609
610    /// 生成对齐到自然边界的桶序列
611    ///
612    /// 返回从 `start`（对齐后）到 `end` 的所有桶起始时间。
613    pub fn bucket_sequence(
614        start: DateTime<Utc>,
615        end: DateTime<Utc>,
616        bucket_secs: i64,
617    ) -> Vec<DateTime<Utc>> {
618        let aligned_start = Self::align_to_epoch(start, bucket_secs);
619        let mut buckets = Vec::new();
620        let mut current = aligned_start;
621        while current < end {
622            buckets.push(current);
623            current += Duration::seconds(bucket_secs);
624        }
625        buckets
626    }
627}

629/// gapfill 填充策略
630///
631/// TimescaleDB 提供 `time_bucket_gapfill` 函数，对缺失的桶进行填充。
632#[derive(Debug, Clone, Copy, PartialEq, Eq)]
633pub enum GapfillStrategy {
634    /// 用 NULL 填充（即不填充，保持空值）
635    Null,
636    /// LOCF（Last Observation Carried Forward）：用前一个非空值填充
637    Locf,
638    /// 线性插值：在两个已知值之间线性插值
639    Interpolate,
640    /// 用固定值填充
641    Constant,
642}
643
644/// gapfill 填充器
645pub struct GapfillFiller;
646
647impl GapfillFiller {
648    /// 对时间桶序列进行 gapfill
649    ///
650    /// - `buckets`：原始桶数据（可能有不连续的桶）
651    /// - `expected_starts`：期望的桶起始时间序列（包含所有应存在的桶）
652    /// - `strategy`：填充策略
653    /// - `constant_value`：当策略为 Constant 时使用的固定值
654    pub fn fill(
655        buckets: &[TimeBucket],
656        expected_starts: &[DateTime<Utc>],
657        strategy: GapfillStrategy,
658        constant_value: f64,
659    ) -> Vec<TimeBucket> {
660        // 建立已有的桶索引
661        let mut bucket_map: HashMap<DateTime<Utc>, &TimeBucket> = HashMap::new();
662        for b in buckets {
663            bucket_map.insert(b.bucket_start, b);
664        }
665
666        let mut result = Vec::with_capacity(expected_starts.len());
667        let mut last_non_empty: Option<&TimeBucket> = None;
668        let mut next_non_empty: Option<&TimeBucket> = None;
669        let mut next_idx = 0usize;
670
671        for (i, &start) in expected_starts.iter().enumerate() {
672            if let Some(b) = bucket_map.get(&start) {
673                if b.count > 0 {
674                    last_non_empty = Some(b);
675                }
676                result.push((*b).clone());
677            } else {
678                // 缺失的桶
679                let filled = match strategy {
680                    GapfillStrategy::Null => TimeBucket {
681                        bucket_start: start,
682                        count: 0,
683                        sum: 0.0,
684                        min: 0.0,
685                        max: 0.0,
686                        avg: 0.0,
687                    },
688                    GapfillStrategy::Constant => TimeBucket {
689                        bucket_start: start,
690                        count: 0,
691                        sum: constant_value,
692                        min: constant_value,
693                        max: constant_value,
694                        avg: constant_value,
695                    },
696                    GapfillStrategy::Locf => {
697                        if let Some(last) = last_non_empty {
698                            TimeBucket {
699                                bucket_start: start,
700                                count: 0,
701                                sum: last.avg,
702                                min: last.avg,
703                                max: last.avg,
704                                avg: last.avg,
705                            }
706                        } else {
707                            TimeBucket {
708                                bucket_start: start,
709                                count: 0,
710                                sum: 0.0,
711                                min: 0.0,
712                                max: 0.0,
713                                avg: 0.0,
714                            }
715                        }
716                    }
717                    GapfillStrategy::Interpolate => {
718                        // 查找下一个非空桶
719                        if next_non_empty.is_none() || next_idx <= i {
720                            next_non_empty = None;
721                            next_idx = i + 1;
722                            #[allow(clippy::needless_range_loop)]
723                            for j in (i + 1)..expected_starts.len() {
724                                if let Some(b) = bucket_map.get(&expected_starts[j]) {
725                                    if b.count > 0 {
726                                        next_non_empty = Some(b);
727                                        next_idx = j;
728                                        break;
729                                    }
730                                }
731                            }
732                        }
733                        // 线性插值
734                        let interpolated = match (last_non_empty, next_non_empty) {
735                            (Some(prev), Some(next)) => {
736                                let total_gap = next_idx - i + (i - find_idx(expected_starts, prev.bucket_start).unwrap_or(i));
737                                if total_gap == 0 {
738                                    prev.avg
739                                } else {
740                                    let progress = 1.0; // 简化：当前位置的插值
741                                    prev.avg + (next.avg - prev.avg) * progress / total_gap as f64
742                                }
743                            }
744                            (Some(prev), None) => prev.avg,
745                            (None, Some(next)) => next.avg,
746                            (None, None) => 0.0,
747                        };
748                        TimeBucket {
749                            bucket_start: start,
750                            count: 0,
751                            sum: interpolated,
752                            min: interpolated,
753                            max: interpolated,
754                            avg: interpolated,
755                        }
756                    }
757                };
758                result.push(filled);
759            }
760        }
761        result
762    }
763}

765/// 在期望桶序列中查找某个时间点的索引
766fn find_idx(starts: &[DateTime<Utc>], target: DateTime<Utc>) -> Option<usize> {
767    starts.iter().position(|&t| t == target)
768}

774/// 解析时间桶字符串为秒数（支持多字符数字，如 "15m"、"12h"）
775///
776/// 支持的单位：s（秒）、m（分）、h（时）、d（天）、w（周）
777pub fn parse_bucket_to_secs(bucket: &str) -> Result<i64, TimescaleError> {
778    if bucket.is_empty() {
779        return Err(TimescaleError::InvalidConfig(
780            "bucket string is empty".to_string(),
781        ));
782    }
783    let unit_char = bucket.chars().last().unwrap();
784    let num_str = &bucket[..bucket.len() - 1];
785    let num: i64 = num_str.parse().map_err(|_| {
786        TimescaleError::InvalidConfig(format!(
787            "invalid bucket number in '{}': '{}' is not a valid integer",
788            bucket, num_str
789        ))
790    })?;
791    let secs = match unit_char {
792        's' => num,
793        'm' => num.checked_mul(60).ok_or_else(|| {
794            TimescaleError::InvalidConfig(format!("bucket overflow: {}", bucket))
795        })?,
796        'h' => num.checked_mul(3600).ok_or_else(|| {
797            TimescaleError::InvalidConfig(format!("bucket overflow: {}", bucket))
798        })?,
799        'd' => num.checked_mul(86400).ok_or_else(|| {
800            TimescaleError::InvalidConfig(format!("bucket overflow: {}", bucket))
801        })?,
802        'w' => num.checked_mul(86400 * 7).ok_or_else(|| {
803            TimescaleError::InvalidConfig(format!("bucket overflow: {}", bucket))
804        })?,
805        _ => {
806            return Err(TimescaleError::InvalidConfig(format!(
807                "unsupported bucket unit '{}' in '{}': expected one of s/m/h/d/w",
808                unit_char, bucket
809            )))
810        }
811    };
812    if secs <= 0 {
813        return Err(TimescaleError::InvalidConfig(format!(
814            "bucket must be positive: {}",
815            bucket
816        )));
817    }
818    Ok(secs)
819}
820
821/// 将秒数转为时间桶字符串
822pub fn secs_to_bucket_string(secs: i64) -> Result<String, TimescaleError> {
823    if secs <= 0 {
824        return Err(TimescaleError::InvalidConfig(format!(
825            "bucket seconds must be positive: {}",
826            secs
827        )));
828    }
829    if secs % (86400 * 7) == 0 {
830        Ok(format!("{}w", secs / (86400 * 7)))
831    } else if secs % 86400 == 0 {
832        Ok(format!("{}d", secs / 86400))
833    } else if secs % 3600 == 0 {
834        Ok(format!("{}h", secs / 3600))
835    } else if secs % 60 == 0 {
836        Ok(format!("{}m", secs / 60))
837    } else {
838        Ok(format!("{}s", secs))
839    }
840}
