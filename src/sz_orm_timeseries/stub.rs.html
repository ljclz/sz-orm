1//! Stub 实现：生成 TimescaleDB SQL 字符串但不执行
2//!
3//! # 安全性（v0.2.2 修复 C-4）
4//!
5//! 所有 SQL 标识符（表名/列名/视图名/指标名）经 `safety::validate_identifier` 校验，
6//! time_bucket 参数经 `safety::validate_time_bucket` 校验，杜绝 SQL 注入。
7
8use crate::error::TimescaleError;
9use crate::safety::{
10    validate_continuous_aggregate_query, validate_identifier, validate_time_bucket,
11};
12use crate::timeseries::TimeseriesExt;
13use crate::types::{Aggregation, DownsampleConfig, Metric, TimeBucket};
14use async_trait::async_trait;
15use chrono::{DateTime, Utc};
16use std::sync::Mutex;

18/// Stub TimescaleDB 实现
19pub struct StubTimeseries {
20    pub sql_log: Mutex<Vec<String>>,
21}
22
23impl StubTimeseries {
24    pub fn new() -> Self {
25        Self {
26            sql_log: Mutex::new(Vec::new()),
27        }
28    }
29
30    pub fn sql_history(&self) -> Vec<String> {
31        self.sql_log.lock().unwrap().clone()
32    }
33
34    pub fn clear(&self) {
35        self.sql_log.lock().unwrap().clear();
36    }
37
38    fn log(&self, sql: String) {
39        self.sql_log.lock().unwrap().push(sql);
40    }
41}
42
43impl Default for StubTimeseries {
44    fn default() -> Self {
45        Self::new()
46    }
47}

49#[async_trait]
50impl TimeseriesExt for StubTimeseries {
51    async fn create_hypertable(
52        &self,
53        table: &str,
54        time_column: &str,
55    ) -> Result<(), TimescaleError> {
56        // v0.2.2 修复 C-4：表名/时间列名严格校验
57        validate_identifier(table, "table")?;
58        validate_identifier(time_column, "time column")?;
59        let sql = format!("SELECT create_hypertable('{}', '{}')", table, time_column);
60        self.log(sql);
61        Ok(())
62    }
63
64    async fn insert_metric(&self, metric: &Metric) -> Result<(), TimescaleError> {
65        // v0.2.2 修复 C-4：metric.name 严格校验
66        validate_identifier(&metric.name, "metric name")?;
67        let tags_json = serde_json::to_string(&metric.tags).unwrap_or_default();
68        let sql = format!(
69            "INSERT INTO {} (timestamp, value, tags) VALUES ('{}', {}, '{}')",
70            metric.name, metric.timestamp, metric.value, tags_json
71        );
72        self.log(sql);
73        Ok(())
74    }
75
76    async fn query_range(
77        &self,
78        metric: &str,
79        start: DateTime<Utc>,
80        end: DateTime<Utc>,
81    ) -> Result<Vec<Metric>, TimescaleError> {
82        // v0.2.2 修复 C-4：metric 严格校验
83        validate_identifier(metric, "metric")?;
84        let sql = format!(
85            "SELECT * FROM {} WHERE timestamp >= '{}' AND timestamp < '{}'",
86            metric, start, end
87        );
88        self.log(sql);
89        Ok(Vec::new())
90    }
91
92    async fn time_bucket_aggregate(
93        &self,
94        metric: &str,
95        bucket: &str,
96        aggregation: Aggregation,
97        start: DateTime<Utc>,
98        end: DateTime<Utc>,
99    ) -> Result<Vec<TimeBucket>, TimescaleError> {
100        // v0.2.2 修复 C-4：metric 与 bucket 严格校验
101        validate_identifier(metric, "metric")?;
102        validate_time_bucket(bucket)?;
103        let sql = format!(
104            "SELECT time_bucket('{}', timestamp) AS bucket, {}(value) AS value FROM {} WHERE timestamp >= '{}' AND timestamp < '{}' GROUP BY bucket ORDER BY bucket",
105            bucket, aggregation.as_sql(), metric, start, end
106        );
107        self.log(sql);
108        Ok(Vec::new())
109    }
110
111    async fn create_continuous_aggregate(
112        &self,
113        view_name: &str,
114        query: &str,
115    ) -> Result<(), TimescaleError> {
116        // v0.2.2 修复 C-4：视图名严格校验
117        // v0.2.2 修复 P0-9（第二次审查发现）：query 严格校验（防 SQL 注入）
118        validate_identifier(view_name, "view name")?;
119        validate_continuous_aggregate_query(query)?;
120        let sql = format!(
121            "CREATE MATERIALIZED VIEW {} WITH (timescaledb.continuous) AS {} WITH NO DATA",
122            view_name, query
123        );
124        self.log(sql);
125        Ok(())
126    }
127
128    async fn downsample(&self, config: &DownsampleConfig) -> Result<(), TimescaleError> {
129        // v0.2.2 修复 C-4：target_metric/source_metric/interval 严格校验
130        validate_identifier(&config.target_metric, "target metric")?;
131        validate_identifier(&config.source_metric, "source metric")?;
132        validate_time_bucket(&config.interval)?;
133        let sql = format!(
134            "INSERT INTO {} SELECT time_bucket('{}', timestamp), {}(value) FROM {} GROUP BY 1",
135            config.target_metric,
136            config.interval,
137            config.aggregation.as_sql(),
138            config.source_metric
139        );
140        self.log(sql);
141        Ok(())
142    }
143
144    async fn drop_metric(&self, metric: &str) -> Result<(), TimescaleError> {
145        // v0.2.2 修复 C-4：metric 严格校验
146        validate_identifier(metric, "metric")?;
147        let sql = format!("DROP TABLE IF EXISTS {}", metric);
148        self.log(sql);
149        Ok(())
150    }
151}

307    #[tokio::test]
308    async fn test_stub_rejects_sql_injection_in_continuous_aggregate_query() {
309        // v0.2.2 修复 P0-9（第二次审查发现）：query 参数 SQL 注入防护
310        let stub = StubTimeseries::new();
311        // 分号注入
312        let result = stub
313            .create_continuous_aggregate("cpu_view", "SELECT 1; DROP TABLE metrics")
314            .await;
315        assert!(
316            result.is_err(),
317            "should reject semicolon injection in query"
318        );
319        // 非 SELECT 开头
320        let result = stub
321            .create_continuous_aggregate("cpu_view", "DROP TABLE metrics")
322            .await;
323        assert!(result.is_err(), "should reject non-SELECT query");
324        // 行注释
325        let result = stub
326            .create_continuous_aggregate("cpu_view", "SELECT 1 -- comment")
327            .await;
328        assert!(result.is_err(), "should reject line comment in query");
329        // 块注释
330        let result = stub
331            .create_continuous_aggregate("cpu_view", "SELECT /* x */ 1")
332            .await;
333        assert!(result.is_err(), "should reject block comment in query");
334        // DDL 关键字
335        let result = stub
336            .create_continuous_aggregate("cpu_view", "SELECT * FROM m WHERE x = DROP")
337            .await;
338        assert!(result.is_err(), "should reject DDL keyword in query");
339        // 合法 query 应通过
340        let result = stub
341            .create_continuous_aggregate(
342                "cpu_view",
343                "SELECT time_bucket('1h', ts), avg(v) FROM m GROUP BY 1",
344            )
345            .await;
346        assert!(result.is_ok(), "should accept valid SELECT query");
347        // 合法 CTE 应通过
348        let result = stub
349            .create_continuous_aggregate("cpu_view", "WITH x AS (SELECT 1) SELECT * FROM x")
350            .await;
351        assert!(result.is_ok(), "should accept valid WITH CTE query");
352    }
