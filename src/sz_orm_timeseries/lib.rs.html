1//! SZ-ORM TimescaleDB 扩展
2//!
3//! 提供时序数据存储、查询和聚合能力，支持三种实现：
4//!
5//! - **内存实现**（`Memory`）：纯 Rust 时序数据存储，不连接数据库
6//! - **Stub 实现**（`Stub`）：生成 TimescaleDB SQL 字符串但不执行
7//! - **真实实现**（`RealTimescale`，需启用 `real-timescale` feature）：通过 tokio-postgres 连接 TimescaleDB
8//!
9//! # 支持的操作
10//!
11//! | 方法 | SQL 等价 | 说明 |
12//! |------|---------|------|
13//! | `create_hypertable` | `SELECT create_hypertable(...)` | 创建 hypertable |
14//! | `insert_metric` | `INSERT INTO ...` | 插入指标 |
15//! | `query_range` | `SELECT ... WHERE timestamp BETWEEN ...` | 范围查询 |
16//! | `time_bucket_aggregate` | `SELECT time_bucket(...), AGG(value) ...` | 时间桶聚合 |
17//! | `create_continuous_aggregate` | `CREATE MATERIALIZED VIEW ...` | 连续聚合视图 |
18//! | `downsample` | `INSERT INTO target SELECT time_bucket ...` | 降采样 |
19//! | `drop_metric` | `DROP TABLE ...` | 删除指标 |
20//!
21//! # 快速入门
22//!
23//! ```rust
24//! use sz_orm_timeseries::{TimeseriesBuilder, TimeseriesExt, TimeseriesProvider, Metric, Aggregation};
25//! use chrono::{Utc, TimeZone};
26//!
27//! # #[tokio::main]
28//! # async fn main() -> Result<(), Box<dyn std::error::Error>> {
29//! let wrapper = TimeseriesBuilder::new(TimeseriesProvider::Memory).build()?;
30//! wrapper.create_hypertable("cpu_usage", "ts").await?;
31//!
32//! let now = Utc::now();
33//! wrapper.insert_metric(&Metric::new("cpu_usage", now, 0.75)).await?;
34//!
35//! let buckets = wrapper.time_bucket_aggregate(
36//!     "cpu_usage", "1m", Aggregation::Avg,
37//!     now - chrono::Duration::minutes(5), now
38//! ).await?;
39//! println!("buckets: {}", buckets.len());
40//! # Ok(())
41//! # }
42//! ```

74/// M-17 修复：查询时间范围最大跨度（秒）
75///
76/// 限制单次 `query_range` / `time_bucket_aggregate` 的时间范围不能超过 366 天，
77/// 防止调用方误用（如查询 100 年的数据）导致 OOM 或数据库性能问题。
78///
79/// - 366 天 ≈ 1 年（含闰年），足以覆盖常见的监控/分析场景
80/// - 366 * 86400 = 31,622,400 秒
81pub const MAX_QUERY_RANGE_SECS: i64 = 366 * 86400;
82
83/// M-17 修复：校验时间范围是否在合理跨度内
84///
85/// - `start >= end`：返回 `InvalidTimeRange` 错误
86/// - `(end - start) > MAX_QUERY_RANGE_SECS`：返回 `InvalidTimeRange` 错误
87/// - 其他情况：返回 Ok
88pub fn validate_time_range(
89    start: chrono::DateTime<chrono::Utc>,
90    end: chrono::DateTime<chrono::Utc>,
91) -> Result<(), TimescaleError> {
92    if start >= end {
93        return Err(TimescaleError::InvalidTimeRange {
94            start: start.to_rfc3339(),
95            end: end.to_rfc3339(),
96        });
97    }
98    let duration_secs = (end - start).num_seconds();
99    if duration_secs > MAX_QUERY_RANGE_SECS {
100        return Err(TimescaleError::InvalidTimeRange {
101            start: start.to_rfc3339(),
102            end: end.to_rfc3339(),
103        });
104    }
105    Ok(())
106}
