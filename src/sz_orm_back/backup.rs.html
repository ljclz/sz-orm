9/// A serialized backup manifest written to disk by [`BackupManager::backup`]
10/// and read back by [`crate::RestoreManager::restore`].
11///
12/// The format is JSON so that restore can validate the pool name and inspect
13/// row counts without depending on any external tooling.
14#[derive(Debug, Clone, Serialize, Deserialize)]
15pub struct BackupManifest {
16    /// Magic header so restore can fail early on unrelated files.
17    pub format: String,
18    pub format_version: u32,
19    pub pool_name: String,
20    pub created_at: i64,
21    pub compressed: bool,
22    pub tables: Vec<BackupTable>,
23    /// `true` for incremental backups produced by
24    /// [`BackupManager::incremental_backup`]. Full backups keep this `false`
25    /// so older readers can deserialize the manifest unchanged.
26    #[serde(default)]
27    pub is_incremental: bool,
28    /// Identifier of the full backup an incremental backup is built on top
29    /// of. `None` for full backups or when the incremental has no base.
30    #[serde(default)]
31    pub base_backup_id: Option<String>,
32}
33
34impl BackupManifest {
35    pub const FORMAT: &'static str = "sz-orm-back";
36    pub const VERSION: u32 = 1;
37
38    pub fn new(pool_name: impl Into<String>, compressed: bool, tables: Vec<BackupTable>) -> Self {
39        Self {
40            format: Self::FORMAT.to_string(),
41            format_version: Self::VERSION,
42            pool_name: pool_name.into(),
43            created_at: current_timestamp_millis(),
44            compressed,
45            tables,
46            is_incremental: false,
47            base_backup_id: None,
48        }
49    }
50
51    /// Computes the SHA-256 hex digest of the manifest's canonical JSON
52    /// serialization. The digest is deterministic for byte-identical
53    /// manifests, so it can be used to detect tampering or corruption in
54    /// transit.
55    pub fn checksum_sha256(&self) -> String {
56        use sha2::{Digest, Sha256};
57        // serde_json serializes struct fields in declaration order and
58        // serde_json::Map uses BTreeMap by default (no `preserve_order`
59        // feature), so the byte output is stable across processes.
60        let bytes = serde_json::to_vec(self).unwrap_or_default();
61        let mut hasher = Sha256::new();
62        hasher.update(&bytes);
63        let digest = hasher.finalize();
64        let mut out = String::with_capacity(digest.len() * 2);
65        for byte in digest.iter() {
66            out.push_str(&format!("{:02x}", byte));
67        }
68        out
69    }
70
71    /// Builder-style setter for `base_backup_id`. Useful for associating an
72    /// incremental backup manifest with the full backup it was built on top
73    /// of. Does not mutate any other field.
74    pub fn with_base_backup_id(mut self, base_backup_id: impl Into<String>) -> Self {
75        self.base_backup_id = Some(base_backup_id.into());
76        self
77    }
78}

92/// Internal record tracking a registered table's rows and the wall-clock
93/// time of its last mutation. The modification timestamp powers
94/// [`BackupManager::incremental_backup`].
95struct RegisteredTable {
96    rows: Vec<serde_json::Value>,
97    modified_at: chrono::DateTime<chrono::Utc>,
98}
99
100impl BackupManager {
101    pub fn new(config: BackupConfig) -> Self {
102        Self {
103            config,
104            tables: RwLock::new(HashMap::new()),
105        }
106    }
107
108    /// Registers (or replaces) the rows for a virtual table that will be
109    /// included in the next [`Self::backup`] / [`Self::export_sql`] call.
110    /// The table's modification timestamp is updated to `Utc::now()` so
111    /// subsequent [`Self::incremental_backup`] calls can pick it up.
112    pub fn register_table(&self, name: impl Into<String>, rows: Vec<serde_json::Value>) {
113        // lock poisoned 时降级为 no-op，避免级联 panic 拖垮整个进程。
114        // lock poisoned 意味着另一线程已 panic，该数据结构状态不可信。
115        if let Ok(mut tables) = self.tables.write() {
116            tables.insert(
117                name.into(),
118                RegisteredTable {
119                    rows,
120                    modified_at: chrono::Utc::now(),
121                },
122            );
123        }
124    }
125
126    /// Returns a snapshot of the table names currently registered, in
127    /// deterministic (sorted) order.
128    pub fn registered_tables(&self) -> Vec<String> {
129        // lock poisoned 时返回空 Vec，避免级联 panic。
130        let names = match self.tables.read() {
131            Ok(tables) => {
132                let mut names: Vec<String> = tables.keys().cloned().collect();
133                names.sort();
134                names
135            }
136            Err(_) => Vec::new(),
137        };
138        names
139    }
140
141    /// Writes a JSON [`BackupManifest`] containing every registered table to
142    /// `output_dir`. The on-disk file is real - subsequent calls to
143    /// [`crate::RestoreManager::restore`] can read it back.
144    ///
145    /// When `config.compress == true` the JSON payload is gzip-compressed
146    /// (magic bytes `0x1f 0x8b`) before being written to disk. Restore
147    /// auto-detects the encoding, so callers do not need to track it.
148    pub async fn backup(
149        &self,
150        pool_name: &str,
151        output_dir: &Path,
152    ) -> Result<BackupResult, BkError> {
153        let start = std::time::Instant::now();
154
155        if !output_dir.exists() {
156            tokio::fs::create_dir_all(output_dir).await?;
157        }
158
159        let filename = format!("{}_backup_{}.json", pool_name, timestamp_secs());
160        let output_path = output_dir.join(filename);
161
162        let tables_snapshot = self.snapshot_all_tables()?;
163
164        let manifest =
165            BackupManifest::new(pool_name, self.config.compress, tables_snapshot.clone());
166
167        let json_bytes = if self.config.compress {
168            serde_json::to_vec(&manifest).map_err(|e| BkError::Backup(e.to_string()))?
169        } else {
170            serde_json::to_vec_pretty(&manifest).map_err(|e| BkError::Backup(e.to_string()))?
171        };
172
173        let bytes = if self.config.compress {
174            gzip_encode(&json_bytes, self.config.compression_level)?
175        } else {
176            json_bytes
177        };
178
179        let file_size = bytes.len() as u64;
180        tokio::fs::write(&output_path, &bytes).await?;
181
182        let total_tables = tables_snapshot.len();
183        let total_rows: u64 = tables_snapshot.iter().map(|t| t.row_count).sum();
184
185        Ok(BackupResult {
186            pool_name: pool_name.to_string(),
187            output_path,
188            total_tables,
189            backed_tables: total_tables,
190            total_rows,
191            duration_ms: start.elapsed().as_millis() as u64,
192            compressed: self.config.compress,
193            file_size,
194        })
195    }
196
197    /// Produces an incremental [`BackupManifest`] containing only tables
198    /// whose `modified_at >= since`. The returned manifest has
199    /// `is_incremental = true` and `base_backup_id = None`; callers can
200    /// chain [`BackupManifest::with_base_backup_id`] to associate it with
201    /// a full backup.
202    ///
203    /// No file is written - the manifest is returned in-memory so callers
204    /// can decide where (and whether) to persist it.
205    pub async fn incremental_backup(
206        &self,
207        since: chrono::DateTime<chrono::Utc>,
208    ) -> Result<BackupManifest, BkError> {
209        let mut snapshot = self.snapshot_tables_modified_since(since)?;
210        // Deterministic ordering, matching full backup semantics.
211        snapshot.sort_by(|a, b| a.name.cmp(&b.name));
212
213        let mut manifest = BackupManifest::new("", self.config.compress, snapshot);
214        manifest.is_incremental = true;
215        manifest.base_backup_id = None;
216        Ok(manifest)
217    }
218
219    /// Snapshots every registered table into a sorted [`BackupTable`] vec.
220    fn snapshot_all_tables(&self) -> Result<Vec<BackupTable>, BkError> {
221        let tables = self
222            .tables
223            .read()
224            .map_err(|e| BkError::Backup(e.to_string()))?;
225        let mut entries: Vec<(&String, &RegisteredTable)> = tables.iter().collect();
226        entries.sort_by(|a, b| a.0.cmp(b.0));
227        Ok(entries
228            .into_iter()
229            .map(|(name, registered)| BackupTable {
230                name: name.clone(),
231                row_count: registered.rows.len() as u64,
232                rows: registered.rows.clone(),
233            })
234            .collect())
235    }
236
237    /// Snapshots only tables whose `modified_at >= since` into an unsorted
238    /// [`BackupTable`] vec. Callers that need deterministic ordering should
239    /// sort the result.
240    fn snapshot_tables_modified_since(
241        &self,
242        since: chrono::DateTime<chrono::Utc>,
243    ) -> Result<Vec<BackupTable>, BkError> {
244        let tables = self
245            .tables
246            .read()
247            .map_err(|e| BkError::Backup(e.to_string()))?;
248        Ok(tables
249            .iter()
250            .filter(|(_, registered)| registered.modified_at >= since)
251            .map(|(name, registered)| BackupTable {
252                name: name.clone(),
253                row_count: registered.rows.len() as u64,
254                rows: registered.rows.clone(),
255            })
256            .collect())
257    }

259    /// Writes a UTF-8 SQL file containing one `INSERT` statement per registered
260    /// row. The output is real - it can be re-read by
261    /// [`crate::RestoreManager::import_sql`].
262    pub async fn export_sql(
263        &self,
264        pool_name: &str,
265        output_dir: &Path,
266    ) -> Result<ExportResult, BkError> {
267        let start = std::time::Instant::now();
268
269        if !output_dir.exists() {
270            tokio::fs::create_dir_all(output_dir).await?;
271        }
272
273        let output_path = output_dir.join(format!("{}_export.sql", pool_name));
274
275        let mut sql = String::new();
276        sql.push_str(&format!("-- SZ-ORM export for pool: {}\n", pool_name));
277        sql.push_str(&format!(
278            "-- Generated at: {}\n",
279            current_timestamp_millis()
280        ));
281        sql.push_str("-- Format: sz-orm-back/sql-export v1\n\n");
282
283        let (total_tables, total_rows) = {
284            let tables = self
285                .tables
286                .read()
287                .map_err(|e| BkError::Export(e.to_string()))?;
288            let mut entries: Vec<(&String, &RegisteredTable)> = tables.iter().collect();
289            entries.sort_by(|a, b| a.0.cmp(b.0));
290            let mut tables_count = 0usize;
291            let mut rows_count = 0u64;
292            for (name, registered) in entries {
293                tables_count += 1;
294                sql.push_str(&format!("-- Table: {}\n", name));
295                for row in &registered.rows {
296                    let json_str =
297                        serde_json::to_string(row).map_err(|e| BkError::Export(e.to_string()))?;
298                    sql.push_str(&format!(
299                        "INSERT INTO \"{}\" VALUES ({});\n",
300                        name, json_str
301                    ));
302                    rows_count += 1;
303                }
304                sql.push('\n');
305            }
306            (tables_count, rows_count)
307        };
308
309        let file_size = sql.len() as u64;
310        tokio::fs::write(&output_path, sql.as_bytes()).await?;
311
312        Ok(ExportResult {
313            pool_name: pool_name.to_string(),
314            output_path,
315            file_size,
316            tables: total_tables,
317            total_rows,
318            duration_ms: start.elapsed().as_millis() as u64,
319        })
320    }
321}

417/// In-memory registry of [`BackupManifest`]s used to enforce retention
418/// policies. The catalog keeps manifests in insertion order so that
419/// [`Self::prune`] can return the evicted manifests in the same order they
420/// were registered.
421pub struct BackupCatalog {
422    manifests: Vec<BackupManifest>,
423}
424
425impl BackupCatalog {
426    pub fn new() -> Self {
427        Self {
428            manifests: Vec::new(),
429        }
430    }
431
432    /// Appends a manifest to the catalog. Manifests are stored in
433    /// insertion order; later calls to [`Self::list`] and
434    /// [`Self::prune`] preserve that order.
435    pub fn register(&mut self, manifest: BackupManifest) {
436        self.manifests.push(manifest);
437    }
438
439    /// Returns a snapshot of all registered manifests in insertion order.
440    pub fn list(&self) -> Vec<&BackupManifest> {
441        self.manifests.iter().collect()
442    }
443
444    /// Removes every manifest whose `created_at` is older than
445    /// `now - retention` and returns the evicted manifests in insertion
446    /// order. Manifests exactly at the cutoff boundary are kept (the
447    /// comparison is strict `<`).
448    pub fn prune(&mut self, retention: std::time::Duration) -> Vec<BackupManifest> {
449        let now_millis = current_timestamp_millis();
450        let retention_millis = i64::try_from(retention.as_millis()).unwrap_or(i64::MAX);
451        let cutoff = now_millis.saturating_sub(retention_millis);
452
453        let all = std::mem::take(&mut self.manifests);
454        let (removed, kept) = all
455            .into_iter()
456            .partition(|manifest: &BackupManifest| manifest.created_at < cutoff);
457        self.manifests = kept;
458        removed
459    }
460}
461
462impl Default for BackupCatalog {
463    fn default() -> Self {
464        Self::new()
465    }
466}

468fn timestamp_secs() -> String {
469    use std::time::{SystemTime, UNIX_EPOCH};
470    let now = SystemTime::now()
471        .duration_since(UNIX_EPOCH)
472        .unwrap_or_default()
473        .as_secs();
474    format!("{}", now)
475}

478    use std::time::{SystemTime, UNIX_EPOCH};

482        .as_millis() as i64

485/// Gzip-encodes `input` using the requested compression level. `level` of
486/// `None` falls back to flate2's default (currently 6). Levels outside the
487/// valid range (0..=9) are clamped.
488fn gzip_encode(input: &[u8], level: Option<u32>) -> Result<Vec<u8>, BkError> {
489    use flate2::write::GzEncoder;
490    use std::io::Write;
491    let level = level.unwrap_or(6).clamp(0, 9);
492    let compression = flate2::Compression::new(level);
493    let encoder = GzEncoder::new(Vec::with_capacity(input.len() / 4 + 32), compression);
494    let mut writer = encoder;
495    writer
496        .write_all(input)
497        .map_err(|e| BkError::Compression(e.to_string()))?;
498    writer
499        .finish()
500        .map_err(|e| BkError::Compression(e.to_string()))
501}

503/// Returns `true` if `bytes` begins with the gzip magic header `0x1f 0x8b`.
504pub(crate) fn looks_like_gzip(bytes: &[u8]) -> bool {
505    bytes.len() >= 2 && bytes[0] == 0x1f && bytes[1] == 0x8b
506}
507
508/// Gzip-decodes `input`. Caller is expected to have already verified the
509/// magic header via [`looks_like_gzip`].
510pub(crate) fn gzip_decode(input: &[u8]) -> Result<Vec<u8>, BkError> {
511    use std::io::Read;
512    let mut decoder = flate2::read::GzDecoder::new(input);
513    let mut out = Vec::with_capacity(input.len() * 4);
514    decoder
515        .read_to_end(&mut out)
516        .map_err(|e| BkError::Compression(e.to_string()))?;
517    Ok(out)
