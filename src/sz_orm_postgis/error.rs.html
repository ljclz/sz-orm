1//! PostGIS 错误类型
2
3use std::fmt;
4
5/// PostGIS 操作错误
6#[derive(Debug)]
7pub enum PostgisError {
8    /// 几何数据无效
9    InvalidGeometry(String),
10    /// 坐标参考系统（SRID）不匹配
11    SridMismatch { expected: i32, actual: i32 },
12    /// 操作不支持
13    Unsupported(String),
14    /// SQL 执行错误
15    Query(String),
16    /// 连接错误
17    Connection(String),
18    /// 配置错误
19    InvalidConfig(String),
20    /// 几何类型不匹配
21    TypeMismatch {
22        expected: &'static str,
23        actual: &'static str,
24    },
25}
26
27impl fmt::Display for PostgisError {
28    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
29        match self {
30            PostgisError::InvalidGeometry(msg) => {
31                write!(f, "invalid geometry: {}", msg)
32            }
33            PostgisError::SridMismatch { expected, actual } => {
34                write!(f, "SRID mismatch: expected {}, got {}", expected, actual)
35            }
36            PostgisError::Unsupported(msg) => {
37                write!(f, "unsupported operation: {}", msg)
38            }
39            PostgisError::Query(msg) => write!(f, "query error: {}", msg),
40            PostgisError::Connection(msg) => write!(f, "connection error: {}", msg),
41            PostgisError::InvalidConfig(msg) => write!(f, "invalid config: {}", msg),
42            PostgisError::TypeMismatch { expected, actual } => {
43                write!(
44                    f,
45                    "geometry type mismatch: expected {}, got {}",
46                    expected, actual
47                )
48            }
49        }
50    }
51}
52
53impl std::error::Error for PostgisError {}
54
55impl From<std::io::Error> for PostgisError {
56    fn from(err: std::io::Error) -> Self {
57        PostgisError::Query(err.to_string())
58    }
59}
